    25→//// Convolution Functions
    26→
    27→int32_t riscv_nn_conv_HWC_s8_s8_s8_asym_bias_any(const int8_t * in_tensor,
    28→                                                 const uint16_t in_tensor_dim_x,
    29→                                                 const uint16_t in_tensor_dim_y,
    30→                                                 const uint16_t in_tensor_ch,
    31→                                                 const uint16_t in_tensor_batch,
    32→                                                 const int8_t * ker_weight,
    33→                                                 const uint16_t out_tensor_ch,
    34→                                                 const uint16_t ker_dim_x,
    35→                                                 const uint16_t ker_dim_y,
    36→                                                 const uint16_t pad_x,
    37→                                                 const uint16_t pad_y,
    38→                                                 const uint16_t stride_x,
    39→                                                 const uint16_t stride_y,
    40→                                                 const int32_t * bias,
    41→                                                 int8_t * out_tensor,
    42→                                                 const int32_t * out_shift,
    43→                                                 const int32_t * out_scale,
    44→                                                 const int32_t out_offset,    //value is in the range of [-128, 127]
    45→                                                 const int32_t in_offset,     //value is in the range of [-127, 128]
    46→                                                 const int32_t act_min,
    47→                                                 const int32_t act_max,
    48→                                                 const uint16_t out_tensor_dim_x,
    49→                                                 const uint16_t out_tensor_dim_y,
    50→                                                 int16_t * in_tmp_buf)
    51→{
    52→#ifdef ENA_EMPTY_FUN
    53→    //empty body
    54→#elif defined(ENA_VEC_ISA) && (NDS_VEC_RVV_VERSION >= 100) && defined(ENA_NDS_V5_VEC_DOT_PROD) && defined(ENA_VEC_ELEN64) && defined(ENA_VEC_INTRINSIC)
    55→    int i_batch;
    56→    for (i_batch = 0; i_batch < in_tensor_batch; i_batch++)
    57→    {
    58→        int32_t i_out_y, i_out_x, i_ker_y, i_ker_x;
    59→
    60→        /* Generate two columns from the input tensor a GEMM computation */
    61→        int8_t *two_column_buf = (int8_t *)in_tmp_buf;
    62→        int8_t *out = out_tensor;
    63→        long vl;
    64→
    65→        /* This part implements the im2col function */
    66→        for (i_out_y = 0; i_out_y < out_tensor_dim_y; i_out_y++)
    67→        {
    68→            for (i_out_x = 0; i_out_x < out_tensor_dim_x; i_out_x++)
    69→            {
    70→                for (i_ker_y = i_out_y * stride_y - pad_y; i_ker_y < i_out_y * stride_y - pad_y + ker_dim_y; i_ker_y++)
    71→                {
    72→                    for (i_ker_x = i_out_x * stride_x - pad_x; i_ker_x < i_out_x * stride_x - pad_x + ker_dim_x; i_ker_x++)
    73→                    {
    74→                        if (i_ker_y < 0 || i_ker_y >= in_tensor_dim_y || i_ker_x < 0 || i_ker_x >= in_tensor_dim_x)
    75→                        {
    76→                            nn_set_val_s8(two_column_buf, -in_offset, in_tensor_ch);
    77→                        }
    78→                        else
    79→                        {
    80→                            nn_dup_s8(in_tensor + (i_ker_y * in_tensor_dim_x + i_ker_x) * in_tensor_ch, two_column_buf, in_tensor_ch);
    81→                        }
    82→                        two_column_buf += in_tensor_ch;
    83→                    }
    84→                }
    85→
    86→                /* Computation is filed for every 2 columns */
    87→                if (two_column_buf == (int8_t *)in_tmp_buf + 2 * in_tensor_ch * ker_dim_y * ker_dim_x)
    88→                {
    89→                    out = nn_mat_mult_kernel_s8_offset(ker_weight,
    90→                                                       (int8_t *)in_tmp_buf,
    91→                                                       out_tensor_ch,
    92→                                                       out_shift,
    93→                                                       out_scale,
    94→                                                       in_offset,
    95→                                                       out_offset,
    96→                                                       act_min,
    97→                                                       act_max,
    98→                                                       in_tensor_ch * ker_dim_y * ker_dim_x,
    99→                                                       bias,
   100→                                                       out);
   101→
   102→                    /* counter reset */
   103→                    two_column_buf = (int8_t *)in_tmp_buf;
   104→                }
   105→            }
   106→        }
   107→
   108→        /* left-over because odd number of output pixels */
   109→        if (two_column_buf != (int8_t *)in_tmp_buf)
   110→        {
   111→            const int8_t *ker_a = ker_weight;
   112→            int i;
   113→            long col_count = (in_tensor_ch * ker_dim_y * ker_dim_x);
   114→            long vlenb = NDS_VEC_CSR_GET_VLENB();
   115→
   116→            vl = __riscv_vsetvl_e8m4(col_count);
   117→            vint8m4_t vOne = __riscv_vmv_v_x_i8m4(1, vl);                   //const one for vd4dot
   118→
   119→            //set SEW for 1st time VMV_S_X
   120→            vl = __riscv_vsetvl_e32m4(col_count);
   121→
   122→            for (i = 0; i < out_tensor_ch; i++)
   123→            {
   124→                int32_t sum = 0;
   125→                if (bias)
   126→                {
   127→                    sum = bias[i];
   128→                }
   129→
   130→                /* Point to the beginning of the im2col buffer where the input is available as a rearranged column */
   131→                const int8_t *ip_as_col = (int8_t *)in_tmp_buf;
   132→                long col_count2 = col_count;
   133→                int32_t sum_ker_a = 0;
   134→
   135→                //init
   136→                vint32m1_t vSum_e0 = __riscv_vmv_s_x_i32m1(sum, 1);
   137→                vint32m1_t vSum_ker_a_e0 = __riscv_vmv_s_x_i32m1(sum_ker_a, 1);
   138→
   139→                //clear the accumulator
   140→                vint32m4_t vAcc0 = __riscv_vand(__riscv_vundefined_i32m4(), 0, vl);
   141→                vint32m4_t vAcc1 = __riscv_vand(__riscv_vundefined_i32m4(), 0, vl);
   142→
   143→                while (col_count2 >= (vlenb * NDS_VEC_VTYPE_LMUL_ID_M4))
   144→                {
   145→                    vint8m4_t vWt_i8, vIn_i8;
   146→                    vl = __riscv_vsetvl_e8m4(col_count2);
   147→                    vWt_i8 = __riscv_vle8_v_i8m4(ker_a, vl);
   148→                    vIn_i8 = __riscv_vle8_v_i8m4(ip_as_col, vl);
   149→                    col_count2 -= vl;
   150→                    ker_a += vl;
   151→                    ip_as_col += vl;
   152→
   153→                    vl = __riscv_vsetvl_e32m4(vl);
   154→                    vAcc0 = __riscv_vd4dots_vv_i32m4_tu(vAcc0, vWt_i8, vIn_i8, vl);
   155→                    vAcc1 = __riscv_vd4dots_vv_i32m4_tu(vAcc1, vWt_i8, vOne, vl);
   156→                }
   157→
   158→                if(col_count2 > 0)
   159→                {
   160→                    vint8m4_t vWt_i8, vIn_i8;
   161→                    vl = __riscv_vsetvl_e8m4(col_count+3);
   162→                    vWt_i8 = __riscv_vand(__riscv_vundefined_i8m4(), 0, vl);
   163→
   164→                    vl = __riscv_vsetvl_e8m4(col_count2);
   165→                    vWt_i8 = __riscv_vle8_v_i8m4_tu(vWt_i8, ker_a, vl);
   166→                    vIn_i8 = __riscv_vle8_v_i8m4(ip_as_col, vl);
   167→                    col_count2 -= vl;
   168→                    ker_a += vl;
   169→                    ip_as_col += vl;
   170→
   171→                    vl = (vl + 3) >> 2;
   172→                    vl = __riscv_vsetvl_e32m4(vl);
   173→                    vAcc0 = __riscv_vd4dots_vv_i32m4_tu(vAcc0, vWt_i8, vIn_i8, vl);
   174→                    vAcc1 = __riscv_vd4dots_vv_i32m4_tu(vAcc1, vWt_i8, vOne, vl);
   175→                }
   176→
   177→                vl = __riscv_vsetvl_e32m4(col_count);       //note the avl here
   178→                vSum_e0 = __riscv_vredsum_tu(vSum_e0, vAcc0, vSum_e0, vl);
   179→                vSum_ker_a_e0 = __riscv_vredsum_tu(vSum_ker_a_e0, vAcc1, vSum_ker_a_e0, vl);
   180→
   181→                //move vr results to gpr
   182→                sum = __riscv_vmv_x(vSum_e0);
   183→                sum_ker_a = __riscv_vmv_x(vSum_ker_a_e0);
   184→
   185→                //add offset back
   186→                sum += in_offset * sum_ker_a;
   187→
   188→                sum = nn_requantize(sum, out_scale[i], out_shift[i]);
   189→                sum += out_offset;
   190→                sum = MAX(sum, act_min);
   191→                sum = MIN(sum, act_max);
   192→                *out++ = (int8_t)sum;
   193→            }
   194→        }
   195→        in_tensor += (in_tensor_dim_x * in_tensor_dim_y * in_tensor_ch);
   196→        out_tensor += (out_tensor_dim_x * out_tensor_dim_y * out_tensor_ch);
   197→    }
   198→#elif defined(ENA_VEC_ISA) && (NDS_VEC_RVV_VERSION >= 100) && defined(ENA_VEC_ELEN64) && defined(ENA_VEC_INTRINSIC)
   199→    int i_batch;
   200→    for (i_batch = 0; i_batch < in_tensor_batch; i_batch++)
   201→    {
   202→        int32_t i_out_y, i_out_x, i_ker_y, i_ker_x;
   203→
   204→        /* Generate two columns from the input tensor a GEMM computation */
   205→        int8_t *two_column_buf = (int8_t *)in_tmp_buf;
   206→        int8_t *out = out_tensor;
   207→        long vl;
   208→
   209→        /* This part implements the im2col function */
   210→        for (i_out_y = 0; i_out_y < out_tensor_dim_y; i_out_y++)
   211→        {
   212→            for (i_out_x = 0; i_out_x < out_tensor_dim_x; i_out_x++)
   213→            {
   214→                for (i_ker_y = i_out_y * stride_y - pad_y; i_ker_y < i_out_y * stride_y - pad_y + ker_dim_y; i_ker_y++)
   215→                {
   216→                    for (i_ker_x = i_out_x * stride_x - pad_x; i_ker_x < i_out_x * stride_x - pad_x + ker_dim_x; i_ker_x++)
   217→                    {
   218→                        if (i_ker_y < 0 || i_ker_y >= in_tensor_dim_y || i_ker_x < 0 || i_ker_x >= in_tensor_dim_x)
   219→                        {
   220→                            nn_set_val_s8(two_column_buf, -in_offset, in_tensor_ch);
   221→                        }
   222→                        else
   223→                        {
   224→                            nn_dup_s8(in_tensor + (i_ker_y * in_tensor_dim_x + i_ker_x) * in_tensor_ch, two_column_buf, in_tensor_ch);
   225→                        }
   226→                        two_column_buf += in_tensor_ch;
   227→                    }
   228→                }
   229→
   230→                /* Computation is filed for every 2 columns */
   231→                if (two_column_buf == (int8_t *)in_tmp_buf + 2 * in_tensor_ch * ker_dim_y * ker_dim_x)
   232→                {
   233→                    out = nn_mat_mult_kernel_s8_offset(ker_weight,
   234→                                                       (int8_t *)in_tmp_buf,
   235→                                                       out_tensor_ch,
   236→                                                       out_shift,
   237→                                                       out_scale,
   238→                                                       in_offset,
   239→                                                       out_offset,
   240→                                                       act_min,
   241→                                                       act_max,
   242→                                                       in_tensor_ch * ker_dim_y * ker_dim_x,
   243→                                                       bias,
   244→                                                       out);
   245→
   246→                    /* counter reset */
   247→                    two_column_buf = (int8_t *)in_tmp_buf;
   248→                }
   249→            }
   250→        }
   251→
   252→        /* left-over because odd number of output pixels */
   253→        if (two_column_buf != (int8_t *)in_tmp_buf)
   254→        {
   255→            const int8_t *ptr_wt = ker_weight;
   256→            int i;
   257→            long col_count = (in_tensor_ch * ker_dim_y * ker_dim_x);
   258→            const long in_offset2 = -in_offset;
   259→
   260→            vl = __riscv_vsetvl_e32m1(1);
   261→            vint32m1_t vzero = __riscv_vand(__riscv_vundefined_i32m1(), 0, vl);   //const zero for redsum
   262→
   263→            vl = __riscv_vsetvl_e32m8(col_count);
   264→
   265→            for (i = 0; i < out_tensor_ch; i++)
   266→            {
   267→                int32_t sum = 0;
   268→
   269→                /* Point to the beginning of the im2col buffer where the input is available as a rearranged column */
   270→                const int8_t *ptr_in = (int8_t *)in_tmp_buf;
   271→                long col_count2 = col_count;
   272→
   273→                //clear v0~v7
   274→                vint32m8_t vaccumulator = __riscv_vand(__riscv_vundefined_i32m8(), 0x0, vl);
   275→
   276→                while (col_count2)
   277→                {
   278→                    vl = __riscv_vsetvl_e8m2(col_count2);
   279→                    vint8m2_t vptr_wt = __riscv_vle8_v_i8m2(ptr_wt, vl);
   280→                    vint8m2_t vptr_in = __riscv_vle8_v_i8m2(ptr_in, vl);
   281→                    vint16m4_t vptr_in_i16m4 = __riscv_vwsub_vx_i16m4(vptr_in, in_offset2, vl);
   282→                    vl = __riscv_vsetvl_e16m4(col_count2);
   283→                    vint16m4_t vptr_wt_i16m4 = __riscv_vsext_vf2(vptr_wt, vl);
   284→                    vaccumulator = __riscv_vwmacc(vaccumulator, vptr_wt_i16m4, vptr_in_i16m4, vl);
   285→                    col_count2 -= vl;
   286→                    ptr_wt += vl;
   287→                    ptr_in += vl;
   288→                }
   289→
   290→                vl = __riscv_vsetvl_e32m8(col_count);   //note the avl here
   291→                vint32m1_t vsum = __riscv_vredsum(vaccumulator, vzero, vl);
   292→                //move vr results to gpr
   293→                sum = __riscv_vmv_x(vsum);
   294→
   295→                //add bias back
   296→                if (bias)
   297→                {
   298→                    sum += bias[i];
   299→                }
   300→
   301→                sum = nn_requantize(sum, out_scale[i], out_shift[i]);
   302→                sum += out_offset;
   303→                sum = MAX(sum, act_min);
   304→                sum = MIN(sum, act_max);
   305→                *out++ = (int8_t)sum;
   306→            }
   307→        }
   308→        in_tensor += (in_tensor_dim_x * in_tensor_dim_y * in_tensor_ch);
   309→        out_tensor += (out_tensor_dim_x * out_tensor_dim_y * out_tensor_ch);
   310→    }
   311→#elif defined(ENA_VEC_ISA) && (NDS_VEC_RVV_VERSION >= 100) && defined(ENA_NDS_V5_VEC_DOT_PROD) && defined(ENA_TILING)
   312→    const uint32_t col = ker_dim_y * ker_dim_x * in_tensor_ch;
   313→    const uint32_t tiling_size = 256;
   314→    long maxvl_per_loop;
   315→
   316→    //--- calculate the buffer address ---
   317→    const long align_byte = 64;
   318→    // buf_A
   319→    int8_t *src1 = (int8_t*)in_tmp_buf;
   320→    src1 = (int8_t*)nn_align((unsigned long)src1, align_byte);                  // align the starting addresss
   321→    // buf_B: converted src2
   322→    long src1_size = tiling_size * in_tensor_ch * ker_dim_x * ker_dim_y;
   323→    int8_t *src2 = (int8_t*)(src1 + src1_size);
   324→    src2 = (int8_t*)nn_align((unsigned long)src2, align_byte);                  // align the starting addresss
   325→    // buf_C: temp buffer for int32_t tiling accumulator
   326→    long src2_size = out_tensor_ch * in_tensor_ch * ker_dim_x * ker_dim_y;
   327→    int32_t *dst_i32 = (int32_t*)(src2 + src2_size);
   328→    dst_i32 = (int32_t*)nn_align((unsigned long)dst_i32, align_byte);           // align the starting addresss
   329→    // buf_contri
   330→    long dst_i32_size = tiling_size * out_tensor_ch;
   331→    int32_t *contri_buf = (int32_t*)(dst_i32 + dst_i32_size);
   332→    contri_buf = (int32_t*)nn_align((unsigned long)contri_buf, align_byte);     // align the starting addresss
   333→
   334→    // calculate the weight contribution
   335→    const int8_t *pWt = ker_weight;
   336→    int32_t *pContri_buf = contri_buf;
   337→    const int32_t *pBias = bias;
   338→    NDS_VEC_VSETVLI(maxvl_per_loop, reg_x0, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   339→    if (bias != NULL)
   340→    {
   341→        long out_tensor_ch2 = out_tensor_ch;
   342→        long vl;
   343→        NDS_VEC_VSETVLI(vl, 1, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M1);
   344→        NDS_VEC_VAND_VI(NDS_VEC_V31, NDS_VEC_V31, 0);   // const zero for redsum
   345→
   346→        while (out_tensor_ch2 > 0)
   347→        {
   348→            NDS_VEC_VSETVLI(vl, col, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   349→            NDS_VEC_VAND_VI(NDS_VEC_V0, NDS_VEC_V0, 0);
   350→            NDS_VEC_VSETVLI(vl, col, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   351→            NDS_VEC_VMV_V_I(NDS_VEC_V8, 1);     // const one for vd4dot
   352→
   353→            long rhs_cols2 = col;
   354→            while (rhs_cols2 >= maxvl_per_loop)
   355→            {
   356→                NDS_VEC_VSETVLI(vl, rhs_cols2, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   357→                NDS_VEC_VLE8_V(NDS_VEC_V16, pWt);
   358→                pWt += vl;
   359→                rhs_cols2 -= vl;
   360→
   361→                NDS_VEC_VSETVLI(vl, vl, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   362→                NDS_VEC_VD4DOTS_VV(NDS_VEC_V0, NDS_VEC_V16, NDS_VEC_V8);
   363→            }
   364→
   365→            //clean one of the dot product source
   366→            long tmp_avl;
   367→            tmp_avl = rhs_cols2 + 3;
   368→            NDS_VEC_VSETVLI(vl, tmp_avl, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   369→            NDS_VEC_VAND_VI(NDS_VEC_V8, NDS_VEC_V8, 0);
   370→            NDS_VEC_VSETVLI(vl, rhs_cols2, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   371→            NDS_VEC_VADD_VI(NDS_VEC_V8, NDS_VEC_V8, 1);     // const one for vd4dot
   372→
   373→            // remain loop
   374→            while (rhs_cols2 > 0)
   375→            {
   376→                NDS_VEC_VSETVLI(vl, rhs_cols2, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   377→                NDS_VEC_VLE8_V(NDS_VEC_V16, pWt);
   378→                pWt += vl;
   379→                rhs_cols2 -= vl;
   380→
   381→                vl = (vl + 3) >> 2;
   382→                NDS_VEC_VSETVLI(vl, vl, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   383→                NDS_VEC_VD4DOTS_VV(NDS_VEC_V0, NDS_VEC_V16, NDS_VEC_V8);
   384→            }
   385→
   386→            NDS_VEC_VSETVLI(vl, col, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);   //note the avl here
   387→            NDS_VEC_VREDSUM_VS(NDS_VEC_V0, NDS_VEC_V0, NDS_VEC_V31);
   388→
   389→            NDS_VEC_VSETVLI(vl, 1, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M1);
   390→            NDS_VEC_VLE32_V(NDS_VEC_V1, pBias);
   391→            NDS_VEC_VMUL_VX(NDS_VEC_V0, NDS_VEC_V0, in_offset);     // add input_offset contribution
   392→            NDS_VEC_VADD_VV(NDS_VEC_V0, NDS_VEC_V0, NDS_VEC_V1);    // add bias
   393→            NDS_VEC_VSE32_V(NDS_VEC_V0, pContri_buf);
   394→            pBias += vl;
   395→            pContri_buf += vl;
   396→            out_tensor_ch2--;
   397→        }
   398→    }
   399→    else
   400→    {
   401→        long out_tensor_ch2 = out_tensor_ch;
   402→        long vl;
   403→        NDS_VEC_VSETVLI(vl, 1, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M1);

  1200→    int8_t *two_column_buf = (int8_t*)in_tmp_buf;
  1201→    two_column_buf = (int8_t*)nn_align((unsigned long)two_column_buf, align_byte);  // align the starting addresss
  1202→    // buffer-2
  1203→    long two_column_buf_size = 2 * in_tensor_ch * ker_dim_x * ker_dim_y * sizeof(*in_tensor);
  1204→    int32_t *contri_buf = (int32_t*)(two_column_buf + two_column_buf_size);
  1205→    contri_buf = (int32_t*)nn_align((unsigned long)contri_buf, align_byte);         // align the starting addresss
  1206→
  1207→    // calculate the weight contribution
  1208→    const int8_t *pWt = ker_weight;
  1209→    const long cst_one = 0x01010101;
  1210→
  1211→    for (long cur_out_ch = 0; cur_out_ch < out_tensor_ch; cur_out_ch++)
  1212→    {
  1213→        long sum = 0;
  1214→        long rhs_cols2 = rhs_cols;
  1215→
  1216→        while (rhs_cols2 >= SIMD_NUM_Q7)
  1217→        {
  1218→            suni_t wt_s8 = simd_load_8((s8_t*)pWt, 0);
  1219→            sum = NDS_DSP_SMAQA(sum, wt_s8.l0, cst_one);
  1220→            rhs_cols2 -= SIMD_NUM_Q7;
  1221→            pWt += SIMD_NUM_Q7;
  1222→        }
  1223→
  1224→        // rest loop
  1225→        while (rhs_cols2 > 0)
  1226→        {
  1227→            sum += *pWt;
  1228→            rhs_cols2--;
  1229→            pWt++;
  1230→        }
  1231→
  1232→        sum *= in_offset;
  1233→
  1234→        if (bias != NULL)
  1235→        {
  1236→            sum += bias[cur_out_ch];
  1237→        }
  1238→        contri_buf[cur_out_ch] = sum;
  1239→    }
  1240→
  1241→    for (long cur_batch = 0; cur_batch < in_tensor_batch; cur_batch++)
  1242→    {
  1243→        int8_t *two_column_buf2 = (int8_t *)two_column_buf;
  1244→        int8_t *out = out_tensor;
  1245→
  1246→        /* This part implements the im2col function */
  1247→        for (long cur_out_y = 0; cur_out_y < out_tensor_dim_y; cur_out_y++)
  1248→        {
  1249→            for (long cur_out_x = 0; cur_out_x < out_tensor_dim_x; cur_out_x++)
  1250→            {
  1251→                for (long cur_ker_y = cur_out_y * stride_y - pad_y; cur_ker_y < cur_out_y * stride_y - pad_y + ker_dim_y; cur_ker_y++)
  1252→                {
  1253→                    for (long cur_ker_x = cur_out_x * stride_x - pad_x; cur_ker_x < cur_out_x * stride_x - pad_x + ker_dim_x; cur_ker_x++)
  1254→                    {
  1255→                        if (cur_ker_y < 0 || cur_ker_y >= in_tensor_dim_y || cur_ker_x < 0 || cur_ker_x >= in_tensor_dim_x)
  1256→                        {
  1257→                            nn_set_val_s8(two_column_buf2, -in_offset, in_tensor_ch);
  1258→                        }
  1259→                        else
  1260→                        {
  1261→                            nn_dup_s8(in_tensor + (cur_ker_y * in_tensor_dim_x + cur_ker_x) * in_tensor_ch, two_column_buf2, in_tensor_ch);
  1262→                        }
  1263→                        two_column_buf2 += in_tensor_ch;
  1264→                    }
  1265→                }
  1266→
  1267→                /* Computation is filed for every 2 columns */
  1268→                if (two_column_buf2 == (int8_t *)in_tmp_buf + 2 * in_tensor_ch * ker_dim_y * ker_dim_x)
  1269→                {
  1270→                    out = nn_mat_mult_kernel_s8_offset_v2(ker_weight,
  1271→                                                          two_column_buf,
  1272→                                                          out_tensor_ch,
  1273→                                                          out_shift,
  1274→                                                          out_scale,
  1275→                                                          in_offset,
  1276→                                                          out_offset,
  1277→                                                          act_min,
  1278→                                                          act_max,
  1279→                                                          rhs_cols,
  1280→                                                          contri_buf,
  1281→                                                          out);
  1282→
  1283→                    /* counter reset */
  1284→                    two_column_buf2 = (int8_t *)two_column_buf;
  1285→                }
  1286→            }
  1287→        }
  1288→
  1289→        /* left-over because odd number of out_tensor pixels */
  1290→        if (two_column_buf2 != (int8_t *)two_column_buf)
  1291→        {
  1292→            const int8_t *pWt = ker_weight;
  1293→
  1294→            for (long cur_out_ch = 0; cur_out_ch < out_tensor_ch; cur_out_ch++)
  1295→            {
  1296→                const int8_t *pIn = (int8_t *)two_column_buf;
  1297→                long acc = 0;
  1298→                long rhs_cols2 = rhs_cols;
  1299→
  1300→                while (rhs_cols2 >= SIMD_NUM_Q7)
  1301→                {
  1302→                    suni_t in_val = simd_load_8((s8_t*)pIn, 0);
  1303→                    suni_t wt_val = simd_load_8((s8_t*)pWt, 0);
  1304→                    acc = NDS_DSP_SMAQA(acc, in_val.l0, wt_val.l0);
  1305→                    pIn += SIMD_NUM_Q7;
  1306→                    pWt += SIMD_NUM_Q7;
  1307→                    rhs_cols2 -= SIMD_NUM_Q7;
  1308→                }
  1309→
  1310→                // rest loop
  1311→                while (rhs_cols2 > 0)
  1312→                {
  1313→                    acc += *pIn * *pWt;
  1314→                    pIn++;
  1315→                    pWt++;
  1316→                    rhs_cols2--;
  1317→                }
  1318→
  1319→                // add contribution back
  1320→                acc += contri_buf[cur_out_ch];
  1321→
  1322→                acc = nn_requantize(acc, out_scale[cur_out_ch], out_shift[cur_out_ch]);
  1323→                acc += out_offset;
  1324→                acc = MAX(acc, act_min);
  1325→                acc = MIN(acc, act_max);
  1326→                *out++ = (int8_t)acc;
  1327→            }
  1328→        }
  1329→
  1330→        in_tensor += (in_tensor_dim_x * in_tensor_dim_y * in_tensor_ch);
  1331→        out_tensor += (out_tensor_dim_x * out_tensor_dim_y * out_tensor_ch);
  1332→    }
  1333→#else
  1334→    int i_batch;
  1335→    for (i_batch = 0; i_batch < in_tensor_batch; i_batch++)
  1336→    {
  1337→        (void)in_tmp_buf;
  1338→        int32_t i_out_ch, i_out_y, i_out_x, i_input_ch, i_ker_y, i_ker_x;
  1339→        int32_t conv_out;
  1340→
  1341→        for (i_out_ch = 0; i_out_ch < out_tensor_ch; i_out_ch++)
  1342→        {
  1343→            for (i_out_y = 0; i_out_y < out_tensor_dim_y; i_out_y++)
  1344→            {
  1345→                for (i_out_x = 0; i_out_x < out_tensor_dim_x; i_out_x++)
  1346→                {
  1347→                    conv_out = 0;
  1348→                    if (bias)
  1349→                    {
  1350→                        conv_out = bias[i_out_ch];
  1351→                    }
  1352→
  1353→                    const int32_t base_idx_y = stride_y * i_out_y - pad_y;
  1354→                    const int32_t base_idx_x = stride_x * i_out_x - pad_x;
  1355→
  1356→                    const int32_t ker_y_start = MAX(0, -base_idx_y);
  1357→                    const int32_t ker_x_start = MAX(0, -base_idx_x);
  1358→
  1359→                    const int32_t ker_y_end = MIN(ker_dim_y, in_tensor_dim_y - base_idx_y);
  1360→                    const int32_t ker_x_end = MIN(ker_dim_x, in_tensor_dim_x - base_idx_x);
  1361→
  1362→                    for (i_ker_y = ker_y_start; i_ker_y < ker_y_end; i_ker_y++)
  1363→                    {
  1364→                        for (i_ker_x = ker_x_start; i_ker_x < ker_x_end; i_ker_x++)
  1365→                        {
  1366→                            const int32_t in_row = base_idx_y + i_ker_y;
  1367→                            const int32_t in_col = base_idx_x + i_ker_x;
  1368→                            for (i_input_ch = 0; i_input_ch < in_tensor_ch; i_input_ch++)
  1369→                            {
  1370→                                conv_out +=
  1371→                                    (in_tensor[(in_row * in_tensor_dim_x + in_col) * in_tensor_ch + i_input_ch] + in_offset) *
  1372→                                    ker_weight[i_out_ch * in_tensor_ch * ker_dim_y * ker_dim_x +
  1373→                                           (i_ker_y * ker_dim_x + i_ker_x) * in_tensor_ch + i_input_ch];
  1374→                            }
  1375→                        }
  1376→                    }
  1377→                    conv_out = nn_requantize(conv_out, out_scale[i_out_ch], out_shift[i_out_ch]);
  1378→                    conv_out += out_offset;
  1379→                    conv_out = MAX(conv_out, act_min);
  1380→                    conv_out = MIN(conv_out, act_max);
  1381→                    out_tensor[i_out_ch + (i_out_y * out_tensor_dim_x + i_out_x) * out_tensor_ch] = (int8_t)conv_out;
  1382→                }
  1383→            }
  1384→        }
  1385→        in_tensor += (in_tensor_dim_x * in_tensor_dim_y * in_tensor_ch);
  1386→        out_tensor += (out_tensor_dim_x * out_tensor_dim_y * out_tensor_ch);
  1387→    }
  1388→#endif  //ENA_EMPTY_FUN
  1389→
  1390→    /* Return to application */
  1391→    return 0;
  1392→}
