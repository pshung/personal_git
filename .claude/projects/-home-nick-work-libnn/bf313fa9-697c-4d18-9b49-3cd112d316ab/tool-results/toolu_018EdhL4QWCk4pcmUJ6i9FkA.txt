    32→/**
    33→ * @defgroup FullyConnect Fully-Connected Layer Functions
    34→ * @brief Fully-connected layer functions multiply the input vector by a weight
    35→ *        matrix and add a bias, if any, to the results.
    36→ *        The supported combinations of input vector and weight matrix are
    37→ *        (signed 8-bit integer, signed 8-bit integer), (unsigned 8-bit integer,
    38→ *        signed 8-bit integer), (signed 16-bit integer, signed 8-bit integer),
    39→ *        (signed 16-bit integer, signed 16-bit integer), and (half-precision
    40→ *        floating-point, half-precision floating-point).
    41→ *
    42→ * @{
    43→ */

    45→/**
    46→ * @brief           This function performs batch matrix multiplication with
    47→ *                  signed 8-bit integers for both inputs and outputs.
    48→ * @param[in]       in_lhs          Pointer to the left-hand side input tensor
    49→ * @param[in]       in_rhs          Pointer to the right-hand side input tensor
    50→ * @param[in]       lhs_offset      Offset value for the left-hand side inputs.
    51→ *                                  It should be in the range of -128 to 127.
    52→ * @param[in]       rhs_offset      Offset value for the right-hand side inputs.
    53→ *                                  It should be in the range of -128 to 127.
    54→ * @param[in]       bias            Pointer to the bias vector
    55→ * @param[in]       dst             Pointer to the output tensor
    56→ * @param[in]       out_offset      Offset value for the output tensor. It
    57→ *                                  should be in the range of -128 to 127.
    58→ * @param[in]       out_scale       Scaling value for the quantization on the
    59→ *                                  outputs
    60→ * @param[in]       out_shift       Shift amount for the quantization on the
    61→ *                                  outputs
    62→ * @param[in]       lhs_dim_n       N dimension of the left-hand side input
    63→ *                                  tensor
    64→ * @param[in]       lhs_dim_h       H dimension of the left-hand side input
    65→ *                                  tensor
    66→ * @param[in]       lhs_dim_w       W dimension of the left-hand side input
    67→ *                                  tensor
    68→ * @param[in]       rhs_dim_n       N dimension of the right-hand side input
    69→ *                                  tensor
    70→ * @param[in]       rhs_dim_h       H dimension of the right-hand side input
    71→ *                                  tensor
    72→ * @param[in]       rhs_dim_w       W dimension of the right-hand side input
    73→ *                                  tensor
    74→ * @param[in]       rhs_dim_c       C dimension of the right-hand side input
    75→ *                                  tensor
    76→ * @param[in]       out_dim_n       N dimension of the output tensor
    77→ * @param[in]       out_dim_h       H dimension of the output tensor
    78→ * @param[in]       act_min         Minimum value that the output tensor is
    79→ *                                  limited to. It should be in the range of
    80→ *                                  -128 to 127.
    81→ * @param[in]       act_max         Maximum value that the output tensor is
    82→ *                                  limited to. It should be in the range of
    83→ *                                  -128 to 127.
    84→ * @return          This function only returns 0.
    85→ *
    86→ * @note
    87→ *  - bias could be a null pointer as the bias vector is optional for this
    88→ *    function.
    89→ *  - During the quantization process, a positive out_shift value is used to
    90→ *    left shift calculation results whereas a negative one is used to right
    91→ *    shift.
    92→ */
    93→int32_t riscv_nn_batch_matmul_s8_s8_s8(const int8_t * in_lhs,

    95→                                       const int16_t lhs_offset,

    97→                                       const int32_t * bias,
    98→                                       int8_t * dst,
    99→                                       const int16_t out_offset,

   112→                                       const int32_t act_max);

   180→/**
   181→ * @brief           This function performs calculation on signed 8-bit integers
   182→ *                  for inputs, applying shift-based quantization to the outputs.
   183→ * @param[in]       in_vec          Pointer to the input vector
   184→ * @param[in]       wt_mat          Pointer to the weight matrix
   185→ * @param[in]       size            Number of elements in the input vector
   186→ * @param[in]       wt_row_num      Number of rows in the weight matrix
   187→ * @param[in]       bias_lshift     Left shift amount for the bias
   188→ * @param[in]       out_rshift      Right shift amount for the output
   189→ * @param[in]       bias            Pointer to the bias vector
   190→ * @param[out]      out_vec         Pointer to the output vector
   191→ * @param[in]       in_tmp_buf      Dummy
   192→ * @return          This function only returns 0.
   193→ *
   194→ * @b Example:
   195→ * @code
   196→ * #define IN_SIZE 2048
   197→ * #define OUT_SIZE 256
   198→ * #define BIAS_LSHIFT 9        //Scale up the bias by 2^9
   199→ * #define OUT_RSHIFT 9         //Scale down the outputs by 1/2^9
   200→ *
   201→ * q7_t in_vec[IN_SIZE] = {...};;
   202→ * q7_t wt_mat[IN_SIZE * OUT_SIZE] {...};
   203→ * q7_t bias[OUT_SIZE] = {...};
   204→ * q7_t out_vec[OUT_SIZE];
   205→ *
   206→ * riscv_nn_fc_s8_s8_s8_sft_bias(in_vec, wt_mat, IN_SIZE, OUT_SIZE, BIAS_LSHIFT,
   207→ *     OUT_RSHIFT, bias, out_vec, NULL);
   208→ * @endcode
   209→ */
   210→int32_t riscv_nn_fc_s8_s8_s8_sft_bias(const q7_t * in_vec,
   211→                                      const q7_t * wt_mat,
   212→                                      const uint16_t size,
   213→                                      const uint16_t wt_row_num,
   214→                                      const uint16_t bias_lshift,
   215→                                      const uint16_t out_rshift,
   216→                                      const q7_t * bias,
   217→                                      q7_t * out_vec,
   218→                                      q15_t * in_tmp_buf);

   220→/**

   231→ * @param[out]      out_vec         Pointer to the output vector
