    25→//// Convolution Functions
    26→int32_t riscv_nn_conv_HWC_s8_s8_s8_sym_bias_fast(const q7_t * in_tensor,
    27→                                        const uint16_t in_tensor_dim,
    28→                                        const uint16_t in_tensor_ch,
    29→                                        const q7_t * ker_weight,
    30→                                        const uint16_t out_tensor_ch,
    31→                                        const uint16_t ker_dim,
    32→                                        const uint16_t pad,
    33→                                        const uint16_t stride,
    34→                                        const q31_t * bias,
    35→                                        const uint16_t pre_rshift,
    36→                                        const uint16_t out_scale,
    37→                                        const uint16_t post_rshift,
    38→                                        q7_t * out_tensor,
    39→                                        const uint16_t out_tensor_dim,
    40→                                        q15_t * in_tmp_buf)
    41→{
    42→#ifdef ENA_EMPTY_FUN
    43→    //empty body
    44→#else
    45→    if (in_tensor_ch % 4 != 0 || out_tensor_ch % 2 != 0)
    46→    {
    47→        /* check if the input dimension meets the constraints */
    48→        return -1;
    49→    }
    50→
    51→#if defined(ENA_VEC_ISA) && defined(ENA_TILING)
    52→    int16_t   i_out_y, i_out_x, i_ker_y, i_ker_x;
    53→
    54→    /*
    55→     *  Use im2col with 4-column batching and nn_mat_mul_kernel_q7_bias_2sft_unroll4
    56→     *  which implements a 4x4 tiled GEMM with vd4dots, vectorizing across the
    57→     *  reduction dimension K (=in_tensor_ch * ker_dim * ker_dim). This achieves
    58→     *  full vector utilization even when out_tensor_ch is small.
    59→     */
    60→
    61→    q7_t     *pBuffer = (q7_t*)in_tmp_buf;
    62→    q7_t     *pOut = out_tensor;
    63→    int bound_begin = (pad + stride - 1) / stride;
    64→    int bound_end   = (in_tensor_dim + pad - ker_dim + stride) / stride;
    65→
    66→    /* top part */
    67→    for (i_out_y = 0; i_out_y < bound_begin; i_out_y++)
    68→    {
    69→        for (i_out_x = 0; i_out_x < out_tensor_dim; i_out_x++)
    70→        {
    71→            for (i_ker_y = i_out_y * stride - pad; i_ker_y < i_out_y * stride - pad + ker_dim; i_ker_y++)
    72→            {
    73→                for (i_ker_x = i_out_x * stride - pad; i_ker_x < i_out_x * stride - pad + ker_dim; i_ker_x++)
    74→                {
    75→                    if (i_ker_y < 0 || i_ker_y >= in_tensor_dim || i_ker_x < 0 || i_ker_x >= in_tensor_dim)
    76→                    {
    77→                        nn_set_zero_s8(pBuffer, sizeof(q7_t)*in_tensor_ch);
    78→                    }
    79→                    else
    80→                    {
    81→                        nn_dup_s8
    82→                        ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim + i_ker_x) * in_tensor_ch, pBuffer, in_tensor_ch);
    83→                    }
    84→                    pBuffer += in_tensor_ch;
    85→                }
    86→            }
    87→
    88→            if (pBuffer == (q7_t *)in_tmp_buf + 4 * in_tensor_ch * ker_dim * ker_dim)
    89→            {
    90→                pOut = nn_mat_mul_kernel_q7_bias_2sft_unroll4(ker_weight,
    91→                                                      (q7_t*)in_tmp_buf,
    92→                                                      out_tensor_ch,
    93→                                                      in_tensor_ch * ker_dim * ker_dim,
    94→                                                      pre_rshift,
    95→                                                      out_scale,
    96→                                                      post_rshift,
    97→                                                      bias,
    98→                                                      pOut);
    99→                pBuffer = (q7_t *)in_tmp_buf;
   100→            }
   101→        }
   102→    }
   103→
   104→    /* middle part — split x into left, mid, right */
   105→    for (; i_out_y < bound_end; i_out_y++)
   106→    {
   107→        /* left part */
   108→        for (i_out_x = 0; i_out_x < bound_begin; i_out_x++)
   109→        {
   110→            for (i_ker_y = i_out_y * stride - pad; i_ker_y < i_out_y * stride - pad + ker_dim; i_ker_y++)
   111→            {
   112→                for (i_ker_x = i_out_x * stride - pad; i_ker_x < i_out_x * stride - pad + ker_dim; i_ker_x++)
   113→                {
   114→                    if (i_ker_x < 0 || i_ker_x >= in_tensor_dim)
   115→                    {
   116→                        nn_set_zero_s8(pBuffer, sizeof(q7_t)*in_tensor_ch);
   117→                    }
   118→                    else
   119→                    {
   120→                        nn_dup_s8
   121→                        ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim + i_ker_x) * in_tensor_ch, pBuffer, in_tensor_ch);
   122→                    }
   123→                    pBuffer += in_tensor_ch;
   124→                }
   125→            }
   126→
   127→            if (pBuffer == (q7_t *)in_tmp_buf + 4 * in_tensor_ch * ker_dim * ker_dim)
   128→            {
   129→                pOut = nn_mat_mul_kernel_q7_bias_2sft_unroll4(ker_weight,
   130→                                                      (q7_t*)in_tmp_buf,
   131→                                                      out_tensor_ch,
   132→                                                      in_tensor_ch * ker_dim * ker_dim,
   133→                                                      pre_rshift,
   134→                                                      out_scale,
   135→                                                      post_rshift,
   136→                                                      bias,
   137→                                                      pOut);
   138→                pBuffer = (q7_t *)in_tmp_buf;
   139→            }
   140→        }
   141→
   142→        /* mid part — no padding, use bulk copy */
   143→        for (; i_out_x < bound_end; i_out_x++)
   144→        {
   145→            for (i_ker_y = i_out_y * stride - pad; i_ker_y < i_out_y * stride - pad + ker_dim; i_ker_y++)
   146→            {
   147→                nn_dup_s8
   148→                ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim + i_out_x * stride - pad) * in_tensor_ch, pBuffer, in_tensor_ch * ker_dim);
   149→                pBuffer += in_tensor_ch * ker_dim;
   150→            }
   151→
   152→            if (pBuffer == (q7_t *)in_tmp_buf + 4 * in_tensor_ch * ker_dim * ker_dim)
   153→            {
   154→                pOut = nn_mat_mul_kernel_q7_bias_2sft_unroll4(ker_weight,
   155→                                                      (q7_t*)in_tmp_buf,
   156→                                                      out_tensor_ch,
   157→                                                      in_tensor_ch * ker_dim * ker_dim,
   158→                                                      pre_rshift,
   159→                                                      out_scale,
   160→                                                      post_rshift,
   161→                                                      bias,
   162→                                                      pOut);
   163→                pBuffer = (q7_t *)in_tmp_buf;
   164→            }
   165→        }
   166→
   167→        /* right part */
   168→        for (; i_out_x < out_tensor_dim; i_out_x++)
   169→        {
   170→            for (i_ker_y = i_out_y * stride - pad; i_ker_y < i_out_y * stride - pad + ker_dim; i_ker_y++)
   171→            {
   172→                for (i_ker_x = i_out_x * stride - pad; i_ker_x < i_out_x * stride - pad + ker_dim; i_ker_x++)
   173→                {
   174→                    if (i_ker_x < 0 || i_ker_x >= in_tensor_dim)
   175→                    {
   176→                        nn_set_zero_s8(pBuffer, sizeof(q7_t)*in_tensor_ch);
   177→                    }
   178→                    else
   179→                    {
   180→                        nn_dup_s8
   181→                        ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim + i_ker_x) * in_tensor_ch, pBuffer, in_tensor_ch);
   182→                    }
   183→                    pBuffer += in_tensor_ch;
   184→                }
   185→            }
   186→
   187→            if (pBuffer == (q7_t *)in_tmp_buf + 4 * in_tensor_ch * ker_dim * ker_dim)
   188→            {
   189→                pOut = nn_mat_mul_kernel_q7_bias_2sft_unroll4(ker_weight,
   190→                                                      (q7_t*)in_tmp_buf,
   191→                                                      out_tensor_ch,
   192→                                                      in_tensor_ch * ker_dim * ker_dim,
   193→                                                      pre_rshift,
   194→                                                      out_scale,
   195→                                                      post_rshift,
   196→                                                      bias,
   197→                                                      pOut);
   198→                pBuffer = (q7_t *)in_tmp_buf;
   199→            }
   200→        }
   201→    }
   202→
   203→    /* bottom part */
   204→    for (; i_out_y < out_tensor_dim; i_out_y++)
   205→    {
   206→        for (i_out_x = 0; i_out_x < out_tensor_dim; i_out_x++)
   207→        {
   208→            for (i_ker_y = i_out_y * stride - pad; i_ker_y < i_out_y * stride - pad + ker_dim; i_ker_y++)
   209→            {
   210→                for (i_ker_x = i_out_x * stride - pad; i_ker_x < i_out_x * stride - pad + ker_dim; i_ker_x++)
   211→                {
   212→                    if (i_ker_y < 0 || i_ker_y >= in_tensor_dim || i_ker_x < 0 || i_ker_x >= in_tensor_dim)
   213→                    {
   214→                        nn_set_zero_s8(pBuffer, sizeof(q7_t)*in_tensor_ch);
   215→                    }
   216→                    else
   217→                    {
   218→                        nn_dup_s8
   219→                        ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim + i_ker_x) * in_tensor_ch, pBuffer, in_tensor_ch);
   220→                    }
   221→                    pBuffer += in_tensor_ch;
   222→                }
   223→            }
   224→
   225→            if (pBuffer == (q7_t *)in_tmp_buf + 4 * in_tensor_ch * ker_dim * ker_dim)
   226→            {
   227→                pOut = nn_mat_mul_kernel_q7_bias_2sft_unroll4(ker_weight,
   228→                                                      (q7_t*)in_tmp_buf,
   229→                                                      out_tensor_ch,
   230→                                                      in_tensor_ch * ker_dim * ker_dim,
   231→                                                      pre_rshift,
   232→                                                      out_scale,
   233→                                                      post_rshift,
   234→                                                      bias,
   235→                                                      pOut);
   236→                pBuffer = (q7_t *)in_tmp_buf;
   237→            }
   238→        }
   239→    }
   240→
   241→    /* check if there are left-over pixels in the buffer (1..3 pixels) */
   242→    if (pBuffer != (q7_t *)in_tmp_buf)
   243→    {
   244→        int leftover_bytes = pBuffer - (q7_t *)in_tmp_buf;
   245→        int col_size = in_tensor_ch * ker_dim * ker_dim;
   246→
   247→        /* drain 2 pixels if we have 2 or 3 left over */
   248→        if (leftover_bytes >= 2 * col_size)
   249→        {
   250→            pOut = nn_mat_mul_kernel_q7_bias_2sft(ker_weight,
   251→                                                  (q7_t*)in_tmp_buf,
   252→                                                  out_tensor_ch,
   253→                                                  col_size,
   254→                                                  pre_rshift,
   255→                                                  out_scale,
   256→                                                  post_rshift,
   257→                                                  bias,
   258→                                                  pOut);
   259→            leftover_bytes -= 2 * col_size;
   260→        }
   261→
   262→        /* handle 1 remaining pixel with scalar path */
   263→        if (leftover_bytes > 0)
   264→        {
   265→            const q7_t *pA = ker_weight;
   266→            q7_t *pB = pBuffer - col_size;
   267→            int i;
   268→
   269→            for (i = 0; i < out_tensor_ch; i++)
   270→            {
   271→                q31_t sum = bias[i];
   272→                uint16_t colCnt = col_size;
   273→                q7_t *pB_iter = pB;
   274→
   275→                while (colCnt)
   276→                {
   277→                    sum += *pA++ * *pB_iter++;
   278→                    colCnt--;
   279→                }
   280→
   281→                sum = (sum >> pre_rshift) * out_scale + NN_ROUND(post_rshift);
   282→                *pOut++ = NDS_ISA_SATS((sum >> post_rshift), 8);
   283→            }
   284→        }
   285→    }
   286→#elif defined(ENA_VEC_ISA) || defined(ENA_DSP_ISA_V5)

   288→
   289→    /*
   290→     *  Here we use in_tmp_buf as q15_t internally as computation are done with q15_t level
   291→     *  im2col are done to output in q15_t format from q7_t input
   292→     */
   293→
   294→    q7_t     *pBuffer = (q7_t*)in_tmp_buf;
   295→    q7_t     *pOut = out_tensor;
   296→    int bound_begin = (pad + stride - 1) / stride;
   297→    int bound_end   = (in_tensor_dim + pad - ker_dim + stride) / stride;
   298→
   299→    /*
   300→     *  Here we split the entire matrix into three regions depending on the pad situation
   301→     *    Top: i_out_y from 0 to pad - 1

   324→                    }
   325→                    pBuffer += in_tensor_ch;
   326→                }
   327→            }
   328→
   329→            if (pBuffer == (q7_t *)in_tmp_buf + 2 * in_tensor_ch * ker_dim * ker_dim)
   330→            {
   331→                pOut = nn_mat_mul_kernel_q7_bias_2sft(ker_weight,
   332→                                                      (q7_t*)in_tmp_buf,
   333→                                                      out_tensor_ch,
   334→                                                      in_tensor_ch * ker_dim * ker_dim,
   335→                                                      pre_rshift,
   336→                                                      out_scale,
   337→                                                      post_rshift,
   338→                                                      bias,
   339→                                                      pOut);
   340→                /* counter reset */
   341→                pBuffer = (q7_t *)in_tmp_buf;
   342→            }
   343→        }
   344→    }
   345→
   346→    /* middle part, here we also divide the x into left, mid and right */
   347→    for (; i_out_y < bound_end; i_out_y++)
   348→    {
   349→        /* left part */
   350→        for (i_out_x = 0; i_out_x < bound_begin; i_out_x++)
   351→        {
   352→            /* This part implements the im2col function */
   353→            for (i_ker_y = i_out_y * stride - pad; i_ker_y < i_out_y * stride - pad + ker_dim; i_ker_y++)
   354→            {
   355→                for (i_ker_x = i_out_x * stride - pad; i_ker_x < i_out_x * stride - pad + ker_dim; i_ker_x++)
   356→                {
   357→                    if (i_ker_x < 0 || i_ker_x >= in_tensor_dim)
   358→                    {
   359→                        nn_set_zero_s8(pBuffer, sizeof(q7_t)*in_tensor_ch);
   360→                    }
   361→                    else
   362→                    {
   363→                        nn_dup_s8
   364→                        ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim + i_ker_x) * in_tensor_ch, pBuffer, in_tensor_ch);
   365→                    }
   366→                    pBuffer += in_tensor_ch;
   367→                }
   368→            }
   369→
   370→            if (pBuffer == (q7_t *)in_tmp_buf + 2 * in_tensor_ch * ker_dim * ker_dim)

   373→                                                      (q7_t*)in_tmp_buf,
   374→                                                      out_tensor_ch,

   376→                                                      pre_rshift,

   383→            }
   384→        }
   385→
   386→        /* mid part */
   387→        for (; i_out_x < bound_end; i_out_x++)
   388→        {
   389→            /* This part implements the im2col function */
   390→            for (i_ker_y = i_out_y * stride - pad; i_ker_y < i_out_y * stride - pad + ker_dim; i_ker_y++)
   391→            {
   392→                nn_dup_s8
   393→                ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim + i_out_x * stride - pad) * in_tensor_ch, pBuffer, in_tensor_ch * ker_dim);

   395→            }

   400→                                                      (q7_t*)in_tmp_buf,

   418→            {

   420→                {

   422→                    {

   424→                    }
   425→                    else
   426→                    {
   427→                        nn_dup_s8
   428→                        ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim + i_ker_x) * in_tensor_ch, pBuffer, in_tensor_ch);
   429→                    }
   430→                    pBuffer += in_tensor_ch;
   431→                }
   432→            }
   433→
   434→            if (pBuffer == (q7_t *)in_tmp_buf + 2 * in_tensor_ch * ker_dim * ker_dim)
   435→            {
   436→                pOut = nn_mat_mul_kernel_q7_bias_2sft(ker_weight,
   437→                                                      (q7_t*)in_tmp_buf,

   449→    }
   450→
   451→    for (; i_out_y < out_tensor_dim; i_out_y++)
   452→    {
   453→        for (i_out_x = 0; i_out_x < out_tensor_dim; i_out_x++)
   454→        {
   455→            /* This part implements the im2col function */
   456→            for (i_ker_y = i_out_y * stride - pad; i_ker_y < i_out_y * stride - pad + ker_dim; i_ker_y++)
   457→            {
   458→                for (i_ker_x = i_out_x * stride - pad; i_ker_x < i_out_x * stride - pad + ker_dim; i_ker_x++)
   459→                {
   460→                    if (i_ker_y < 0 || i_ker_y >= in_tensor_dim || i_ker_x < 0 || i_ker_x >= in_tensor_dim)

   468→                    }

   470→                }
   471→            }
   472→
   473→            if (pBuffer == (q7_t *)in_tmp_buf + 2 * in_tensor_ch * ker_dim * ker_dim)
   474→            {
   475→                pOut = nn_mat_mul_kernel_q7_bias_2sft(ker_weight,
   476→                                                      (q7_t*)in_tmp_buf,
   477→                                                      out_tensor_ch,
   478→                                                      in_tensor_ch * ker_dim * ker_dim,
   479→                                                      pre_rshift,
