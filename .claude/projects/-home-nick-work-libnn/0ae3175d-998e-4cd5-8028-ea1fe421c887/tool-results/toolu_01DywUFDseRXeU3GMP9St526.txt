    25→//// Convolution Functions
    26→int32_t riscv_nn_conv_HWC_s8_s8_s8_sym_bias_fast(const q7_t * in_tensor,
    27→                                        const uint16_t in_tensor_dim,
    28→                                        const uint16_t in_tensor_ch,
    29→                                        const q7_t * ker_weight,
    30→                                        const uint16_t out_tensor_ch,
    31→                                        const uint16_t ker_dim,
    32→                                        const uint16_t pad,
    33→                                        const uint16_t stride,
    34→                                        const q31_t * bias,
    35→                                        const uint16_t pre_rshift,
    36→                                        const uint16_t out_scale,
    37→                                        const uint16_t post_rshift,
    38→                                        q7_t * out_tensor,
    39→                                        const uint16_t out_tensor_dim,
    40→                                        q15_t * in_tmp_buf)
    41→{
    42→#ifdef ENA_EMPTY_FUN
    43→    //empty body
    44→#else
    45→    if (in_tensor_ch % 4 != 0 || out_tensor_ch % 2 != 0)
    46→    {
    47→        /* check if the input dimension meets the constraints */
    48→        return -1;
    49→    }
    50→
    51→#if defined(ENA_VEC_ISA) && defined(ENA_TILING)
    52→    int16_t   i_out_y, i_out_x, i_ker_y, i_ker_x;
    53→
    54→    /*
    55→     *  Use im2col with 2-column batching and nn_mat_mul_kernel_q7_bias_2sft
    56→     *  which implements a 4x2 tiled GEMM with vd4dots, vectorizing across the
    57→     *  reduction dimension K (=in_tensor_ch * ker_dim * ker_dim). This achieves
    58→     *  full vector utilization even when out_tensor_ch is small.
    59→     */
    60→
    61→    q7_t     *pBuffer = (q7_t*)in_tmp_buf;
    62→    q7_t     *pOut = out_tensor;

    64→    int bound_end   = (in_tensor_dim + pad - ker_dim + stride) / stride;
    65→
    66→    /* top part */
    67→    for (i_out_y = 0; i_out_y < bound_begin; i_out_y++)
    68→    {
    69→        for (i_out_x = 0; i_out_x < out_tensor_dim; i_out_x++)
    70→        {
    71→            for (i_ker_y = i_out_y * stride - pad; i_ker_y < i_out_y * stride - pad + ker_dim; i_ker_y++)
    72→            {
    73→                for (i_ker_x = i_out_x * stride - pad; i_ker_x < i_out_x * stride - pad + ker_dim; i_ker_x++)
    74→                {
    75→                    if (i_ker_y < 0 || i_ker_y >= in_tensor_dim || i_ker_x < 0 || i_ker_x >= in_tensor_dim)
    76→                    {
    77→                        nn_set_zero_s8(pBuffer, sizeof(q7_t)*in_tensor_ch);
    78→                    }
    79→                    else
    80→                    {
    81→                        nn_dup_s8
    82→                        ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim + i_ker_x) * in_tensor_ch, pBuffer, in_tensor_ch);
    83→                    }
    84→                    pBuffer += in_tensor_ch;
    85→                }
    86→            }
    87→
    88→            if (pBuffer == (q7_t *)in_tmp_buf + 2 * in_tensor_ch * ker_dim * ker_dim)
    89→            {
    90→                pOut = nn_mat_mul_kernel_q7_bias_2sft(ker_weight,
    91→                                                      (q7_t*)in_tmp_buf,
    92→                                                      out_tensor_ch,
    93→                                                      in_tensor_ch * ker_dim * ker_dim,
    94→                                                      pre_rshift,
    95→                                                      out_scale,
    96→                                                      post_rshift,
    97→                                                      bias,
    98→                                                      pOut);

   100→            }
   101→        }
   102→    }
   103→

   106→    {
   107→        /* left part */
   108→        for (i_out_x = 0; i_out_x < bound_begin; i_out_x++)
   109→        {
   110→            for (i_ker_y = i_out_y * stride - pad; i_ker_y < i_out_y * stride - pad + ker_dim; i_ker_y++)

   112→                for (i_ker_x = i_out_x * stride - pad; i_ker_x < i_out_x * stride - pad + ker_dim; i_ker_x++)

   114→                    if (i_ker_x < 0 || i_ker_x >= in_tensor_dim)
   115→                    {
   116→                        nn_set_zero_s8(pBuffer, sizeof(q7_t)*in_tensor_ch);
   117→                    }
   118→                    else
   119→                    {
   120→                        nn_dup_s8
   121→                        ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim + i_ker_x) * in_tensor_ch, pBuffer, in_tensor_ch);
   122→                    }
   123→                    pBuffer += in_tensor_ch;
   124→                }
   125→            }
   126→
   127→            if (pBuffer == (q7_t *)in_tmp_buf + 2 * in_tensor_ch * ker_dim * ker_dim)
   128→            {
   129→                pOut = nn_mat_mul_kernel_q7_bias_2sft(ker_weight,
   130→                                                      (q7_t*)in_tmp_buf,
   131→                                                      out_tensor_ch,
   132→                                                      in_tensor_ch * ker_dim * ker_dim,
   133→                                                      pre_rshift,
   134→                                                      out_scale,
   135→                                                      post_rshift,
   136→                                                      bias,
   137→                                                      pOut);
   138→                pBuffer = (q7_t *)in_tmp_buf;
   139→            }
   140→        }
   141→
   142→        /* mid part — no padding, use bulk copy */

   144→        {

   146→            {
   147→                nn_dup_s8
   148→                ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim + i_out_x * stride - pad) * in_tensor_ch, pBuffer, in_tensor_ch * ker_dim);
   149→                pBuffer += in_tensor_ch * ker_dim;
   150→            }
   151→
   152→            if (pBuffer == (q7_t *)in_tmp_buf + 2 * in_tensor_ch * ker_dim * ker_dim)
   153→            {
   154→                pOut = nn_mat_mul_kernel_q7_bias_2sft(ker_weight,
   155→                                                      (q7_t*)in_tmp_buf,
   156→                                                      out_tensor_ch,
   157→                                                      in_tensor_ch * ker_dim * ker_dim,
   158→                                                      pre_rshift,
   159→                                                      out_scale,
   160→                                                      post_rshift,
   161→                                                      bias,
   162→                                                      pOut);
   163→                pBuffer = (q7_t *)in_tmp_buf;
   164→            }
   165→        }
   166→
   167→        /* right part */
   168→        for (; i_out_x < out_tensor_dim; i_out_x++)
   169→        {
   170→            for (i_ker_y = i_out_y * stride - pad; i_ker_y < i_out_y * stride - pad + ker_dim; i_ker_y++)

   172→                for (i_ker_x = i_out_x * stride - pad; i_ker_x < i_out_x * stride - pad + ker_dim; i_ker_x++)

   174→                    if (i_ker_x < 0 || i_ker_x >= in_tensor_dim)
   175→                    {
   176→                        nn_set_zero_s8(pBuffer, sizeof(q7_t)*in_tensor_ch);
   177→                    }

   181→                        ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim + i_ker_x) * in_tensor_ch, pBuffer, in_tensor_ch);
   182→                    }
   183→                    pBuffer += in_tensor_ch;
   184→                }
   185→            }
   186→
   187→            if (pBuffer == (q7_t *)in_tmp_buf + 2 * in_tensor_ch * ker_dim * ker_dim)
   188→            {
   189→                pOut = nn_mat_mul_kernel_q7_bias_2sft(ker_weight,
   190→                                                      (q7_t*)in_tmp_buf,
   191→                                                      out_tensor_ch,
   192→                                                      in_tensor_ch * ker_dim * ker_dim,
   193→                                                      pre_rshift,
   194→                                                      out_scale,
   195→                                                      post_rshift,
   196→                                                      bias,
   197→                                                      pOut);
   198→                pBuffer = (q7_t *)in_tmp_buf;
   199→            }
   200→        }
   201→    }
   202→
   203→    /* bottom part */
   204→    for (; i_out_y < out_tensor_dim; i_out_y++)
   205→    {
   206→        for (i_out_x = 0; i_out_x < out_tensor_dim; i_out_x++)
   207→        {
   208→            for (i_ker_y = i_out_y * stride - pad; i_ker_y < i_out_y * stride - pad + ker_dim; i_ker_y++)
   209→            {
   210→                for (i_ker_x = i_out_x * stride - pad; i_ker_x < i_out_x * stride - pad + ker_dim; i_ker_x++)
   211→                {
   212→                    if (i_ker_y < 0 || i_ker_y >= in_tensor_dim || i_ker_x < 0 || i_ker_x >= in_tensor_dim)
   213→                    {
   214→                        nn_set_zero_s8(pBuffer, sizeof(q7_t)*in_tensor_ch);
   215→                    }
   216→                    else
   217→                    {
   218→                        nn_dup_s8
   219→                        ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim + i_ker_x) * in_tensor_ch, pBuffer, in_tensor_ch);
   220→                    }
   221→                    pBuffer += in_tensor_ch;
   222→                }
   223→            }
   224→
   225→            if (pBuffer == (q7_t *)in_tmp_buf + 2 * in_tensor_ch * ker_dim * ker_dim)
   226→            {
   227→                pOut = nn_mat_mul_kernel_q7_bias_2sft(ker_weight,
   228→                                                      (q7_t*)in_tmp_buf,
   229→                                                      out_tensor_ch,
   230→                                                      in_tensor_ch * ker_dim * ker_dim,
   231→                                                      pre_rshift,
   232→                                                      out_scale,
   233→                                                      post_rshift,
   234→                                                      bias,
   235→                                                      pOut);
   236→                pBuffer = (q7_t *)in_tmp_buf;
   237→            }
   238→        }
   239→    }
   240→
   241→    /* check if there is left-over for compute (odd number of output pixels) */
   242→    if (pBuffer != (q7_t *)in_tmp_buf)
   243→    {
   244→        const q7_t *pA = ker_weight;
   245→        int i;
   246→
   247→        for (i = 0; i < out_tensor_ch; i++)
   248→        {
   249→            q31_t sum = bias[i];
   250→            q7_t *pB = (q7_t*)in_tmp_buf;
   251→            uint16_t colCnt = in_tensor_ch * ker_dim * ker_dim;
   252→
   253→            while (colCnt)
   254→            {
   255→                sum += *pA++ * *pB++;
   256→                colCnt--;
   257→            }
   258→
   259→            sum = (sum >> pre_rshift) * out_scale + NN_ROUND(post_rshift);
   260→            *pOut++ = NDS_ISA_SATS((sum >> post_rshift), 8);
   261→        }
   262→    }
   263→#elif defined(ENA_VEC_ISA) || defined(ENA_DSP_ISA_V5)
   264→    int16_t   i_out_y, i_out_x, i_ker_y, i_ker_x;
   265→
   266→    /*
   267→     *  Here we use in_tmp_buf as q15_t internally as computation are done with q15_t level

   269→     */
   270→
   271→    q7_t     *pBuffer = (q7_t*)in_tmp_buf;
   272→    q7_t     *pOut = out_tensor;
   273→    int bound_begin = (pad + stride - 1) / stride;
   274→    int bound_end   = (in_tensor_dim + pad - ker_dim + stride) / stride;
   275→
   276→    /*

   280→     * Bottom: i_out_y from out_tensor_dim-pad to out_tensor_dim-1
   281→     */
   282→
   283→    /* top part */
   284→    for (i_out_y = 0; i_out_y < bound_begin; i_out_y++)
   285→    {
   286→        for (i_out_x = 0; i_out_x < out_tensor_dim; i_out_x++)
   287→        {
   288→            /* This part implements the im2col function */
   289→            for (i_ker_y = i_out_y * stride - pad; i_ker_y < i_out_y * stride - pad + ker_dim; i_ker_y++)
   290→            {
   291→                for (i_ker_x = i_out_x * stride - pad; i_ker_x < i_out_x * stride - pad + ker_dim; i_ker_x++)
   292→                {

   306→            if (pBuffer == (q7_t *)in_tmp_buf + 2 * in_tensor_ch * ker_dim * ker_dim)
   307→            {
   308→                pOut = nn_mat_mul_kernel_q7_bias_2sft(ker_weight,
   309→                                                      (q7_t*)in_tmp_buf,
   310→                                                      out_tensor_ch,
   311→                                                      in_tensor_ch * ker_dim * ker_dim,
   312→                                                      pre_rshift,
   313→                                                      out_scale,
   314→                                                      post_rshift,
   315→                                                      bias,
   316→                                                      pOut);
   317→                /* counter reset */
   318→                pBuffer = (q7_t *)in_tmp_buf;
   319→            }
   320→        }
   321→    }
   322→
   323→    /* middle part, here we also divide the x into left, mid and right */
   324→    for (; i_out_y < bound_end; i_out_y++)
   325→    {
   326→        /* left part */
   327→        for (i_out_x = 0; i_out_x < bound_begin; i_out_x++)
   328→        {
   329→            /* This part implements the im2col function */
   330→            for (i_ker_y = i_out_y * stride - pad; i_ker_y < i_out_y * stride - pad + ker_dim; i_ker_y++)
   331→            {
   332→                for (i_ker_x = i_out_x * stride - pad; i_ker_x < i_out_x * stride - pad + ker_dim; i_ker_x++)
   333→                {
   334→                    if (i_ker_x < 0 || i_ker_x >= in_tensor_dim)
   335→                    {
   336→                        nn_set_zero_s8(pBuffer, sizeof(q7_t)*in_tensor_ch);
   337→                    }
   338→                    else
   339→                    {
   340→                        nn_dup_s8
   341→                        ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim + i_ker_x) * in_tensor_ch, pBuffer, in_tensor_ch);
   342→                    }
   343→                    pBuffer += in_tensor_ch;
   344→                }
   345→            }
   346→
   347→            if (pBuffer == (q7_t *)in_tmp_buf + 2 * in_tensor_ch * ker_dim * ker_dim)
   348→            {
   349→                pOut = nn_mat_mul_kernel_q7_bias_2sft(ker_weight,
   350→                                                      (q7_t*)in_tmp_buf,
   351→                                                      out_tensor_ch,
   352→                                                      in_tensor_ch * ker_dim * ker_dim,
   353→                                                      pre_rshift,
   354→                                                      out_scale,

   360→            }
   361→        }
   362→
   363→        /* mid part */
   364→        for (; i_out_x < bound_end; i_out_x++)
   365→        {
   366→            /* This part implements the im2col function */
   367→            for (i_ker_y = i_out_y * stride - pad; i_ker_y < i_out_y * stride - pad + ker_dim; i_ker_y++)
   368→            {
   369→                nn_dup_s8
   370→                ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim + i_out_x * stride - pad) * in_tensor_ch, pBuffer, in_tensor_ch * ker_dim);

   373→

   377→                                                      (q7_t*)in_tmp_buf,

   392→        {

   395→            {
   396→                for (i_ker_x = i_out_x * stride - pad; i_ker_x < i_out_x * stride - pad + ker_dim; i_ker_x++)

   405→                        ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim + i_ker_x) * in_tensor_ch, pBuffer, in_tensor_ch);
   406→                    }
   407→                    pBuffer += in_tensor_ch;
   408→                }
   409→            }
   410→
   411→            if (pBuffer == (q7_t *)in_tmp_buf + 2 * in_tensor_ch * ker_dim * ker_dim)
   412→            {
   413→                pOut = nn_mat_mul_kernel_q7_bias_2sft(ker_weight,
   414→                                                      (q7_t*)in_tmp_buf,
   415→                                                      out_tensor_ch,
   416→                                                      in_tensor_ch * ker_dim * ker_dim,

   425→        }
   426→    }
   427→
   428→    for (; i_out_y < out_tensor_dim; i_out_y++)
   429→    {
   430→        for (i_out_x = 0; i_out_x < out_tensor_dim; i_out_x++)
   431→        {
   432→            /* This part implements the im2col function */
   433→            for (i_ker_y = i_out_y * stride - pad; i_ker_y < i_out_y * stride - pad + ker_dim; i_ker_y++)
   434→            {
   435→                for (i_ker_x = i_out_x * stride - pad; i_ker_x < i_out_x * stride - pad + ker_dim; i_ker_x++)
   436→                {

   438→                    {

   440→                    }
   441→                    else
   442→                    {

   444→                        ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim + i_ker_x) * in_tensor_ch, pBuffer, in_tensor_ch);
   445→                    }
   446→                    pBuffer += in_tensor_ch;
   447→                }
   448→            }
   449→
   450→            if (pBuffer == (q7_t *)in_tmp_buf + 2 * in_tensor_ch * ker_dim * ker_dim)
   451→            {
   452→                pOut = nn_mat_mul_kernel_q7_bias_2sft(ker_weight,
   453→                                                      (q7_t*)in_tmp_buf,
   454→                                                      out_tensor_ch,
   455→                                                      in_tensor_ch * ker_dim * ker_dim,
   456→                                                      pre_rshift,
   457→                                                      out_scale,
   458→                                                      post_rshift,
   459→                                                      bias,
   460→                                                      pOut);
   461→                /* counter reset */
   462→                pBuffer = (q7_t *)in_tmp_buf;
   463→            }
   464→        }
   465→    }
   466→
   467→    /* check if there is left-over for compute */
   468→#ifdef ENA_VEC_ISA
   469→#if (NDS_VEC_RVV_VERSION >= 100) && defined(ENA_NDS_V5_VEC_DOT_PROD) && defined(ENA_VEC_INTRINSIC)
   470→    if (pBuffer != (q7_t *)in_tmp_buf)
   471→    {
   472→        unsigned vl;
   473→        const q7_t *p_wt = ker_weight;
   474→        int       i;
   475→        uint16_t  colCnt = in_tensor_ch * ker_dim * ker_dim;
   476→
   477→        vl = __riscv_vsetvl_e32m1(1);
   478→        vint32m1_t vZero = __riscv_vand(__riscv_vundefined_i32m1(), 0x0, vl);     //const zero for redsum
   479→
   480→        for (i = 0; i < out_tensor_ch; i++)
   481→        {
   482→            q7_t      *p_in = (q7_t*)in_tmp_buf;
   483→            uint16_t colCnt2 = colCnt;
   484→
