    25→    25→//// Convolution Functions
    26→    26→int32_t riscv_nn_conv_HWC_s8_s8_s8_sym_bias_fast(const q7_t * in_tensor,
    27→    27→                                        const uint16_t in_tensor_dim,
    28→    28→                                        const uint16_t in_tensor_ch,
    29→    29→                                        const q7_t * ker_weight,
    30→    30→                                        const uint16_t out_tensor_ch,
    31→    31→                                        const uint16_t ker_dim,
    32→    32→                                        const uint16_t pad,
    33→    33→                                        const uint16_t stride,
    34→    34→                                        const q31_t * bias,
    35→    35→                                        const uint16_t pre_rshift,
    36→    36→                                        const uint16_t out_scale,
    37→    37→                                        const uint16_t post_rshift,
    38→    38→                                        q7_t * out_tensor,
    39→    39→                                        const uint16_t out_tensor_dim,
    40→    40→                                        q15_t * in_tmp_buf)
    41→    41→{
    42→    42→#ifdef ENA_EMPTY_FUN
    43→    43→    //empty body
    44→    44→#else
    45→    45→    if (in_tensor_ch % 4 != 0 || out_tensor_ch % 2 != 0)
    46→    46→    {
    47→    47→        /* check if the input dimension meets the constraints */
    48→    48→        return -1;
    49→    49→    }
    50→    50→
    51→    51→#if defined(ENA_VEC_ISA) && defined(ENA_TILING)
    52→    52→    int16_t   i_out_y, i_out_x, i_ker_y, i_ker_x;
    53→    53→
    54→    54→    /*
    55→    55→     *  Use im2col with 4-column batching and nn_mat_mul_kernel_q7_bias_2sft_unroll4
    56→    56→     *  which implements a 4x4 tiled GEMM with vd4dots, vectorizing across the
    57→    57→     *  reduction dimension K (=in_tensor_ch * ker_dim * ker_dim). This achieves
    58→    58→     *  full vector utilization even when out_tensor_ch is small.
    59→    59→     */

    61→    61→    q7_t     *pBuffer = (q7_t*)in_tmp_buf;
    62→    62→    q7_t     *pOut = out_tensor;
    63→    63→    int bound_begin = (pad + stride - 1) / stride;
    64→    64→    int bound_end   = (in_tensor_dim + pad - ker_dim + stride) / stride;
    65→    65→
    66→    66→    /* top part */
    67→    67→    for (i_out_y = 0; i_out_y < bound_begin; i_out_y++)
    68→    68→    {
    69→    69→        for (i_out_x = 0; i_out_x < out_tensor_dim; i_out_x++)
    70→    70→        {
    71→    71→            for (i_ker_y = i_out_y * stride - pad; i_ker_y < i_out_y * stride - pad + ker_dim; i_ker_y++)
    72→    72→            {
    73→    73→                for (i_ker_x = i_out_x * stride - pad; i_ker_x < i_out_x * stride - pad + ker_dim; i_ker_x++)
    74→    74→                {
    75→    75→                    if (i_ker_y < 0 || i_ker_y >= in_tensor_dim || i_ker_x < 0 || i_ker_x >= in_tensor_dim)
    76→    76→                    {
    77→    77→                        nn_set_zero_s8(pBuffer, sizeof(q7_t)*in_tensor_ch);
    78→    78→                    }
    79→    79→                    else
    80→    80→                    {
    81→    81→                        nn_dup_s8
    82→    82→                        ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim + i_ker_x) * in_tensor_ch, pBuffer, in_tensor_ch);
    83→    83→                    }
    84→    84→                    pBuffer += in_tensor_ch;
    85→    85→                }
    86→    86→            }
    87→    87→
    88→    88→            if (pBuffer == (q7_t *)in_tmp_buf + 4 * in_tensor_ch * ker_dim * ker_dim)
    89→    89→            {
    90→    90→                pOut = nn_mat_mul_kernel_q7_bias_2sft_unroll4(ker_weight,
    91→    91→                                                      (q7_t*)in_tmp_buf,
    92→    92→                                                      out_tensor_ch,
    93→    93→                                                      in_tensor_ch * ker_dim * ker_dim,
    94→    94→                                                      pre_rshift,
    95→    95→                                                      out_scale,
    96→    96→                                                      post_rshift,
    97→    97→                                                      bias,
    98→    98→                                                      pOut);
    99→    99→                pBuffer = (q7_t *)in_tmp_buf;
   100→   100→            }
   101→   101→        }
   102→   102→    }
   103→   103→
   104→   104→    /* middle part — split x into left, mid, right */
   105→   105→    for (; i_out_y < bound_end; i_out_y++)
   106→   106→    {
   107→   107→        /* left part */

   109→   109→        {
   110→   110→            for (i_ker_y = i_out_y * stride - pad; i_ker_y < i_out_y * stride - pad + ker_dim; i_ker_y++)

   112→   112→                for (i_ker_x = i_out_x * stride - pad; i_ker_x < i_out_x * stride - pad + ker_dim; i_ker_x++)
   113→   113→                {
   114→   114→                    if (i_ker_x < 0 || i_ker_x >= in_tensor_dim)
   115→   115→                    {
   116→   116→                        nn_set_zero_s8(pBuffer, sizeof(q7_t)*in_tensor_ch);
   117→   117→                    }
   118→   118→                    else
   119→   119→                    {
   120→   120→                        nn_dup_s8
   121→   121→                        ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim + i_ker_x) * in_tensor_ch, pBuffer, in_tensor_ch);
   122→   122→                    }
   123→   123→                    pBuffer += in_tensor_ch;
   124→   124→                }
   125→   125→            }
   126→   126→
   127→   127→            if (pBuffer == (q7_t *)in_tmp_buf + 4 * in_tensor_ch * ker_dim * ker_dim)
   128→   128→            {
   129→   129→                pOut = nn_mat_mul_kernel_q7_bias_2sft_unroll4(ker_weight,
   130→   130→                                                      (q7_t*)in_tmp_buf,
   131→   131→                                                      out_tensor_ch,
   132→   132→                                                      in_tensor_ch * ker_dim * ker_dim,
   133→   133→                                                      pre_rshift,
   134→   134→                                                      out_scale,
   135→   135→                                                      post_rshift,
   136→   136→                                                      bias,
   137→   137→                                                      pOut);
   138→   138→                pBuffer = (q7_t *)in_tmp_buf;
   139→   139→            }
   140→   140→        }
   141→   141→
   142→   142→        /* mid part — no padding, use bulk copy */
   143→   143→        for (; i_out_x < bound_end; i_out_x++)
   144→   144→        {
   145→   145→            for (i_ker_y = i_out_y * stride - pad; i_ker_y < i_out_y * stride - pad + ker_dim; i_ker_y++)
   146→   146→            {
   147→   147→                nn_dup_s8
   148→   148→                ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim + i_out_x * stride - pad) * in_tensor_ch, pBuffer, in_tensor_ch * ker_dim);
   149→   149→                pBuffer += in_tensor_ch * ker_dim;
   150→   150→            }
   151→   151→
   152→   152→            if (pBuffer == (q7_t *)in_tmp_buf + 4 * in_tensor_ch * ker_dim * ker_dim)
   153→   153→            {
   154→   154→                pOut = nn_mat_mul_kernel_q7_bias_2sft_unroll4(ker_weight,
   155→   155→                                                      (q7_t*)in_tmp_buf,
   156→   156→                                                      out_tensor_ch,
   157→   157→                                                      in_tensor_ch * ker_dim * ker_dim,
   158→   158→                                                      pre_rshift,
   159→   159→                                                      out_scale,
   160→   160→                                                      post_rshift,
   161→   161→                                                      bias,
   162→   162→                                                      pOut);
   163→   163→                pBuffer = (q7_t *)in_tmp_buf;
   164→   164→            }
   165→   165→        }
   166→   166→
   167→   167→        /* right part */
   168→   168→        for (; i_out_x < out_tensor_dim; i_out_x++)
   169→   169→        {
   170→   170→            for (i_ker_y = i_out_y * stride - pad; i_ker_y < i_out_y * stride - pad + ker_dim; i_ker_y++)
   171→   171→            {
   172→   172→                for (i_ker_x = i_out_x * stride - pad; i_ker_x < i_out_x * stride - pad + ker_dim; i_ker_x++)
   173→   173→                {
   174→   174→                    if (i_ker_x < 0 || i_ker_x >= in_tensor_dim)
   175→   175→                    {
   176→   176→                        nn_set_zero_s8(pBuffer, sizeof(q7_t)*in_tensor_ch);
   177→   177→                    }
   178→   178→                    else
   179→   179→                    {
   180→   180→                        nn_dup_s8
   181→   181→                        ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim + i_ker_x) * in_tensor_ch, pBuffer, in_tensor_ch);
   182→   182→                    }
   183→   183→                    pBuffer += in_tensor_ch;
   184→   184→                }
   185→   185→            }
   186→   186→
   187→   187→            if (pBuffer == (q7_t *)in_tmp_buf + 4 * in_tensor_ch * ker_dim * ker_dim)
   188→   188→            {
   189→   189→                pOut = nn_mat_mul_kernel_q7_bias_2sft_unroll4(ker_weight,
   190→   190→                                                      (q7_t*)in_tmp_buf,
   191→   191→                                                      out_tensor_ch,
   192→   192→                                                      in_tensor_ch * ker_dim * ker_dim,
   193→   193→                                                      pre_rshift,
   194→   194→                                                      out_scale,
   195→   195→                                                      post_rshift,
   196→   196→                                                      bias,
   197→   197→                                                      pOut);
   198→   198→                pBuffer = (q7_t *)in_tmp_buf;
   199→   199→            }
   200→   200→        }
   201→   201→    }
   202→   202→
   203→   203→    /* bottom part */
   204→   204→    for (; i_out_y < out_tensor_dim; i_out_y++)
   205→   205→    {
   206→   206→        for (i_out_x = 0; i_out_x < out_tensor_dim; i_out_x++)
   207→   207→        {
   208→   208→            for (i_ker_y = i_out_y * stride - pad; i_ker_y < i_out_y * stride - pad + ker_dim; i_ker_y++)
   209→   209→            {
   210→   210→                for (i_ker_x = i_out_x * stride - pad; i_ker_x < i_out_x * stride - pad + ker_dim; i_ker_x++)
   211→   211→                {
   212→   212→                    if (i_ker_y < 0 || i_ker_y >= in_tensor_dim || i_ker_x < 0 || i_ker_x >= in_tensor_dim)
   213→   213→                    {
   214→   214→                        nn_set_zero_s8(pBuffer, sizeof(q7_t)*in_tensor_ch);
   215→   215→                    }
   216→   216→                    else
   217→   217→                    {
   218→   218→                        nn_dup_s8
   219→   219→                        ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim + i_ker_x) * in_tensor_ch, pBuffer, in_tensor_ch);
   220→   220→                    }
   221→   221→                    pBuffer += in_tensor_ch;
   222→   222→                }
   223→   223→            }
   224→   224→
   225→   225→            if (pBuffer == (q7_t *)in_tmp_buf + 4 * in_tensor_ch * ker_dim * ker_dim)
   226→   226→            {
   227→   227→                pOut = nn_mat_mul_kernel_q7_bias_2sft_unroll4(ker_weight,
   228→   228→                                                      (q7_t*)in_tmp_buf,
   229→   229→                                                      out_tensor_ch,
   230→   230→                                                      in_tensor_ch * ker_dim * ker_dim,
   231→   231→                                                      pre_rshift,
   232→   232→                                                      out_scale,
   233→   233→                                                      post_rshift,
   234→   234→                                                      bias,
   235→   235→                                                      pOut);
   236→   236→                pBuffer = (q7_t *)in_tmp_buf;
   237→   237→            }
   238→   238→        }
   239→   239→    }
   240→   240→
   241→   241→    /* check if there are left-over pixels in the buffer (1..3 pixels) */
   242→   242→    if (pBuffer != (q7_t *)in_tmp_buf)
   243→   243→    {
   244→   244→        int leftover_bytes = pBuffer - (q7_t *)in_tmp_buf;
   245→   245→        int col_size = in_tensor_ch * ker_dim * ker_dim;
   246→   246→
   247→   247→        /* drain 2 pixels if we have 2 or 3 left over */
   248→   248→        if (leftover_bytes >= 2 * col_size)
   249→   249→        {
   250→   250→            pOut = nn_mat_mul_kernel_q7_bias_2sft(ker_weight,
   251→   251→                                                  (q7_t*)in_tmp_buf,
   252→   252→                                                  out_tensor_ch,
   253→   253→                                                  col_size,
   254→   254→                                                  pre_rshift,
   255→   255→                                                  out_scale,
   256→   256→                                                  post_rshift,
   257→   257→                                                  bias,
   258→   258→                                                  pOut);
   259→   259→            leftover_bytes -= 2 * col_size;
   260→   260→        }
   261→   261→
   262→   262→        /* handle 1 remaining pixel with scalar path */
   263→   263→        if (leftover_bytes > 0)
   264→   264→        {
   265→   265→            const q7_t *pA = ker_weight;
   266→   266→            q7_t *pB = pBuffer - col_size;
   267→   267→            int i;
   268→   268→
   269→   269→            for (i = 0; i < out_tensor_ch; i++)
   270→   270→            {
   271→   271→                q31_t sum = bias[i];
   272→   272→                uint16_t colCnt = col_size;
   273→   273→                q7_t *pB_iter = pB;
   274→   274→
   275→   275→                while (colCnt)
   276→   276→                {
   277→   277→                    sum += *pA++ * *pB_iter++;
   278→   278→                    colCnt--;
   279→   279→                }
   280→   280→
   281→   281→                sum = (sum >> pre_rshift) * out_scale + NN_ROUND(post_rshift);
   282→   282→                *pOut++ = NDS_ISA_SATS((sum >> post_rshift), 8);
   283→   283→            }
   284→   284→        }
   285→   285→    }
   286→   286→#elif defined(ENA_VEC_ISA) || defined(ENA_DSP_ISA_V5)
   287→   287→    int16_t   i_out_y, i_out_x, i_ker_y, i_ker_x;
   288→   288→
   289→   289→    /*
   290→   290→     *  Here we use in_tmp_buf as q15_t internally as computation are done with q15_t level
   291→   291→     *  im2col are done to output in q15_t format from q7_t input
   292→   292→     */
   293→   293→
   294→   294→    q7_t     *pBuffer = (q7_t*)in_tmp_buf;
   295→   295→    q7_t     *pOut = out_tensor;
   296→   296→    int bound_begin = (pad + stride - 1) / stride;

   302→   302→     * Middle: i_out_y from pad to out_tensor_dim-pad-1
   303→   303→     * Bottom: i_out_y from out_tensor_dim-pad to out_tensor_dim-1
   304→   304→     */

   327→   327→            }

   332→   332→                                                      (q7_t*)in_tmp_buf,

   334→   334→                                                      in_tensor_ch * ker_dim * ker_dim,
   335→   335→                                                      pre_rshift,

   337→   337→                                                      post_rshift,

   342→   342→            }
   343→   343→        }
   344→   344→    }
   345→   345→
   346→   346→    /* middle part, here we also divide the x into left, mid and right */
   347→   347→    for (; i_out_y < bound_end; i_out_y++)
   348→   348→    {
   349→   349→        /* left part */
   350→   350→        for (i_out_x = 0; i_out_x < bound_begin; i_out_x++)
   351→   351→        {

   358→   358→                    {

   362→   362→                    {
   363→   363→                        nn_dup_s8
   364→   364→                        ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim + i_ker_x) * in_tensor_ch, pBuffer, in_tensor_ch);
   365→   365→                    }
   366→   366→                    pBuffer += in_tensor_ch;
   367→   367→                }
   368→   368→            }
   369→   369→
   370→   370→            if (pBuffer == (q7_t *)in_tmp_buf + 2 * in_tensor_ch * ker_dim * ker_dim)
   371→   371→            {
   372→   372→                pOut = nn_mat_mul_kernel_q7_bias_2sft(ker_weight,
   373→   373→                                                      (q7_t*)in_tmp_buf,
   374→   374→                                                      out_tensor_ch,
   375→   375→                                                      in_tensor_ch * ker_dim * ker_dim,
   376→   376→                                                      pre_rshift,
   377→   377→                                                      out_scale,
   378→   378→                                                      post_rshift,
   379→   379→                                                      bias,
   380→   380→                                                      pOut);
   381→   381→                /* counter reset */
   382→   382→                pBuffer = (q7_t *)in_tmp_buf;
   383→   383→            }
   384→   384→        }
   385→   385→
   386→   386→        /* mid part */
   387→   387→        for (; i_out_x < bound_end; i_out_x++)
   388→   388→        {
   389→   389→            /* This part implements the im2col function */
   390→   390→            for (i_ker_y = i_out_y * stride - pad; i_ker_y < i_out_y * stride - pad + ker_dim; i_ker_y++)
   391→   391→            {
   392→   392→                nn_dup_s8
   393→   393→                ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim + i_out_x * stride - pad) * in_tensor_ch, pBuffer, in_tensor_ch * ker_dim);
   394→   394→                pBuffer += in_tensor_ch * ker_dim;
   395→   395→            }
   396→   396→
   397→   397→            if (pBuffer == (q7_t *)in_tmp_buf + 2 * in_tensor_ch * ker_dim * ker_dim)
   398→   398→            {
   399→   399→                pOut = nn_mat_mul_kernel_q7_bias_2sft(ker_weight,
   400→   400→                                                      (q7_t*)in_tmp_buf,
   401→   401→                                                      out_tensor_ch,
   402→   402→                                                      in_tensor_ch * ker_dim * ker_dim,
   403→   403→                                                      pre_rshift,
   404→   404→                                                      out_scale,
   405→   405→                                                      post_rshift,
   406→   406→                                                      bias,
   407→   407→                                                      pOut);
   408→   408→                /* counter reset */

   410→   410→            }
   411→   411→        }
   412→   412→
   413→   413→        /* right part */
   414→   414→        for (; i_out_x < out_tensor_dim; i_out_x++)
