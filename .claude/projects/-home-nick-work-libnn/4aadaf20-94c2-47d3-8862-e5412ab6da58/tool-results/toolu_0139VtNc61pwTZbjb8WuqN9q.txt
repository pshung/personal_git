    25→//// Convolution Functions
    26→int32_t riscv_nn_conv_HWC_s8_s8_s8_asym_bias_any_dilated(const int8_t * in_tensor,
    27→                                                         const uint16_t in_tensor_dim_x,
    28→                                                         const uint16_t in_tensor_dim_y,
    29→                                                         const uint16_t in_tensor_ch,
    30→                                                         const uint16_t in_tensor_batch,
    31→                                                         const int8_t * ker_weight,
    32→                                                         const uint16_t out_tensor_ch,
    33→                                                         const uint16_t ker_dim_x,
    34→                                                         const uint16_t ker_dim_y,
    35→                                                         const uint16_t ker_ch,
    36→                                                         const uint16_t pad_x,
    37→                                                         const uint16_t pad_y,
    38→                                                         const uint16_t stride_x,
    39→                                                         const uint16_t stride_y,
    40→                                                         const int32_t * bias,
    41→                                                         int8_t * out_tensor,
    42→                                                         const int32_t * out_shift,
    43→                                                         const int32_t * out_scale,
    44→                                                         const int32_t out_offset,    //value is in the range of [-128, 127]
    45→                                                         const int32_t in_offset,     //value is in the range of [-127, 128]
    46→                                                         const int32_t act_min,
    47→                                                         const int32_t act_max,
    48→                                                         const uint16_t out_tensor_dim_x,
    49→                                                         const uint16_t out_tensor_dim_y,
    50→                                                         const int32_t dilation_x,
    51→                                                         const int32_t dilation_y,
    52→                                                         int16_t * in_tmp_buf)
    53→{
    54→#ifdef ENA_EMPTY_FUN
    55→    //empty body
    56→#elif defined(ENA_VEC_ISA) && (NDS_VEC_RVV_VERSION >= 100) && defined(ENA_NDS_V5_VEC_DOT_PROD) && defined(ENA_VEC_ELEN64) && defined(ENA_VEC_INTRINSIC)
    57→    int i_batch;
    58→    for (i_batch = 0; i_batch < in_tensor_batch; i_batch++)
    59→    {
    60→        int32_t i_out_y, i_out_x, i_ker_y, i_ker_x;
    61→
    62→        /* Generate two columns from the input tensor a GEMM computation */
    63→        int8_t *two_column_buf = (int8_t *)in_tmp_buf;
    64→        int8_t *out = out_tensor;
    65→        long vl;
    66→
    67→        /* This part implements the im2col function */
    68→        for (i_out_y = 0; i_out_y < out_tensor_dim_y; i_out_y++)
    69→        {
    70→            for (i_out_x = 0; i_out_x < out_tensor_dim_x; i_out_x++)
    71→            {
    72→                for (i_ker_y = i_out_y * stride_y - pad_y; i_ker_y < i_out_y * stride_y - pad_y + (ker_dim_y - 1) * dilation_y + 1; i_ker_y += dilation_y)
    73→                {
    74→                    for (i_ker_x = i_out_x * stride_x - pad_x; i_ker_x < i_out_x * stride_x - pad_x + (ker_dim_x - 1) * dilation_x + 1; i_ker_x += dilation_x)
    75→                    {
    76→                        if (i_ker_y < 0 || i_ker_y >= in_tensor_dim_y || i_ker_x < 0 || i_ker_x >= in_tensor_dim_x)
    77→                        {
    78→                            nn_set_val_s8(two_column_buf, -in_offset, in_tensor_ch);
    79→                        }
    80→                        else
    81→                        {
    82→                            /* Copying the pixel data to column */
    83→                            nn_dup_s8(in_tensor + (i_ker_y * in_tensor_dim_x + i_ker_x) * in_tensor_ch, two_column_buf, in_tensor_ch);
    84→                        }
    85→                        two_column_buf += in_tensor_ch;
    86→                    }
    87→                }
    88→
    89→                /* Computation is filed for every 2 columns */
    90→                if (two_column_buf == (int8_t *)in_tmp_buf + 2 * in_tensor_ch * ker_dim_y * ker_dim_x)
    91→                {
    92→                    out = nn_mat_mult_kernel_s8_offset(ker_weight,
    93→                                                       (int8_t *)in_tmp_buf,
    94→                                                       out_tensor_ch,
    95→                                                       out_shift,
    96→                                                       out_scale,
    97→                                                       in_offset,
    98→                                                       out_offset,
    99→                                                       act_min,
   100→                                                       act_max,
   101→                                                       in_tensor_ch * ker_dim_y * ker_dim_x,
   102→                                                       bias,
   103→                                                       out);
   104→
   105→                    /* counter reset */
   106→                    two_column_buf = (int8_t *)in_tmp_buf;
   107→                }
   108→            }
   109→        }
   110→
   111→        /* left-over because odd number of output pixels */
   112→        if (two_column_buf != (int8_t *)in_tmp_buf)
   113→        {
   114→            const int8_t *ker_a = ker_weight;
   115→            int i;
   116→            long col_count = (in_tensor_ch * ker_dim_y * ker_dim_x);
   117→            long vlenb = NDS_VEC_CSR_GET_VLENB();
   118→
   119→            vl = __riscv_vsetvl_e8m4(col_count);
   120→            vint8m4_t vOne_i8 = __riscv_vmv_v_x_i8m4(1, vl);                   //const one for vd4dot
   121→
   122→            //set SEW for 1st time VMV_S_X
   123→            vl = __riscv_vsetvl_e32m4(col_count);
   124→
   125→            for (i = 0; i < out_tensor_ch; i++)
   126→            {
   127→                int32_t sum = 0;
   128→                if (bias)
   129→                {
   130→                    sum = bias[i];
   131→                }
   132→
   133→                /* Point to the beginning of the im2col buffer where the input is available as a rearranged column */
   134→                const int8_t *ip_as_col = (int8_t *)in_tmp_buf;
   135→                long col_count2 = col_count;
   136→                int32_t sum_ker_a = 0;
   137→
   138→                //init
   139→                vint32m1_t vSum_e0 = __riscv_vmv_s_x_i32m1(sum, 1);
   140→                vint32m1_t vSum_ker_a_e0 = __riscv_vmv_s_x_i32m1(sum_ker_a, 1);
   141→
   142→                //clear the accumulator
   143→                vint32m4_t vAcc0 = __riscv_vand(__riscv_vundefined_i32m4(), 0, vl);
   144→                vint32m4_t vAcc1 = __riscv_vand(__riscv_vundefined_i32m4(), 0, vl);
   145→
   146→                while (col_count2 >= (vlenb * NDS_VEC_VTYPE_LMUL_ID_M4))
   147→                {
   148→                    vint8m4_t vWt_i8, vIn_i8;
   149→                    vl = __riscv_vsetvl_e8m4(col_count2);
   150→                    vWt_i8 = __riscv_vle8_v_i8m4(ker_a, vl);
   151→                    vIn_i8 = __riscv_vle8_v_i8m4(ip_as_col, vl);
   152→                    col_count2 -= vl;
   153→                    ker_a += vl;
   154→                    ip_as_col += vl;
   155→
   156→                    vl = __riscv_vsetvl_e32m4(vl);
   157→                    vAcc0 = __riscv_vd4dots_vv_i32m4_tu(vAcc0, vWt_i8, vIn_i8, vl);
   158→                    vAcc1 = __riscv_vd4dots_vv_i32m4_tu(vAcc1, vWt_i8, vOne_i8, vl);
   159→                }
   160→
   161→                if(col_count2 > 0)
   162→                {
   163→                    vint8m4_t vWt_i8, vIn_i8;
   164→                    vl = __riscv_vsetvl_e8m4(col_count+3);
   165→                    vWt_i8 = __riscv_vand(__riscv_vundefined_i8m4(), 0, vl);
   166→
   167→                    vl = __riscv_vsetvl_e8m4(col_count2);
   168→                    vWt_i8 = __riscv_vle8_v_i8m4_tu(vWt_i8, ker_a, vl);
   169→                    vIn_i8 = __riscv_vle8_v_i8m4(ip_as_col, vl);
   170→                    col_count2 -= vl;
   171→                    ker_a += vl;
   172→                    ip_as_col += vl;
   173→
   174→                    vl = (vl + 3) >> 2;
   175→                    vl = __riscv_vsetvl_e32m4(vl);
   176→                    vAcc0 = __riscv_vd4dots_vv_i32m4_tu(vAcc0, vWt_i8, vIn_i8, vl);
   177→                    vAcc1 = __riscv_vd4dots_vv_i32m4_tu(vAcc1, vWt_i8, vOne_i8, vl);
   178→                }
   179→
   180→                vl = __riscv_vsetvl_e32m4(col_count);       //note the avl here
   181→                vSum_e0 = __riscv_vredsum_tu(vSum_e0, vAcc0, vSum_e0, vl);
   182→                vSum_ker_a_e0 = __riscv_vredsum_tu(vSum_ker_a_e0, vAcc1, vSum_ker_a_e0, vl);
   183→
   184→                //move vr results to gpr
   185→                sum = __riscv_vmv_x(vSum_e0);
   186→                sum_ker_a = __riscv_vmv_x(vSum_ker_a_e0);
   187→
   188→                //add offset back
   189→                sum += in_offset * sum_ker_a;
   190→
   191→                sum = nn_requantize(sum, out_scale[i], out_shift[i]);
   192→                sum += out_offset;
   193→                sum = MAX(sum, act_min);
   194→                sum = MIN(sum, act_max);
   195→                *out++ = (int8_t)sum;
   196→            }
   197→        }
   198→        in_tensor += (in_tensor_dim_x * in_tensor_dim_y * in_tensor_ch);
   199→        out_tensor += (out_tensor_dim_x * out_tensor_dim_y * out_tensor_ch);
   200→    }
   201→#elif defined(ENA_VEC_ISA) && (NDS_VEC_RVV_VERSION >= 100) && defined(ENA_NDS_V5_VEC_DOT_PROD)
   202→    const long groups = in_tensor_ch / ker_ch;
   203→    const long out_ch_per_group = out_tensor_ch / groups;
   204→
   205→    if (in_tensor_ch % groups != 0 || out_tensor_ch % groups != 0)
   206→    {
   207→        return -1;
   208→    }
   209→
   210→    //--- calculate the buffer address ---
   211→    const long align_byte = sizeof(long);
   212→    // buffer-1
   213→    int8_t *two_column_buf = (int8_t*)in_tmp_buf;
   214→    two_column_buf = (int8_t*)nn_align((unsigned long)two_column_buf, align_byte);  // align the starting addresss
   215→    // buffer-2
   216→    long two_column_buf_size = 2 * ker_ch * ker_dim_x * ker_dim_y * sizeof(*in_tensor);
   217→    int32_t *contri_buf = (int32_t*)(two_column_buf + two_column_buf_size);
   218→    contri_buf = (int32_t*)nn_align((unsigned long)contri_buf, align_byte);         // align the starting addresss
   219→
   220→    const long rhs_cols = ker_dim_y * ker_dim_x * ker_ch;
   221→    const long rhs_cols_aligned = nn_align((long)rhs_cols, align_byte);
   222→    long maxvl_per_loop;
   223→
   224→    // calculate the weight contribution
   225→    const int8_t *pWt = ker_weight;
   226→    int32_t *pContri_buf = contri_buf;
   227→    const int32_t *pBias = bias;
   228→    NDS_VEC_VSETVLI(maxvl_per_loop, reg_x0, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   229→    if (bias != NULL)
   230→    {
   231→        long out_tensor_ch2 = out_tensor_ch;
   232→        long vl;
   233→        NDS_VEC_VSETVLI(vl, 1, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M1);
   234→        NDS_VEC_VAND_VI(NDS_VEC_V31, NDS_VEC_V31, 0);   // const zero for redsum
   235→
   236→        while (out_tensor_ch2 > 0)
   237→        {
   238→            NDS_VEC_VSETVLI(vl, rhs_cols, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   239→            NDS_VEC_VAND_VI(NDS_VEC_V0, NDS_VEC_V0, 0);
   240→            NDS_VEC_VSETVLI(vl, rhs_cols, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   241→            NDS_VEC_VMV_V_I(NDS_VEC_V8, 1);     // const one for vd4dot
   242→
   243→            long rhs_cols2 = rhs_cols;
   244→            while (rhs_cols2 >= maxvl_per_loop)
   245→            {
   246→                NDS_VEC_VSETVLI(vl, rhs_cols2, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   247→                NDS_VEC_VLE8_V(NDS_VEC_V16, pWt);
   248→                pWt += vl;
   249→                rhs_cols2 -= vl;
   250→
   251→                NDS_VEC_VSETVLI(vl, vl, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   252→                NDS_VEC_VD4DOTS_VV(NDS_VEC_V0, NDS_VEC_V16, NDS_VEC_V8);
   253→            }
   254→
   255→            //clean one of the dot product source
   256→            long tmp_avl;
   257→            tmp_avl = rhs_cols2 + 3;
   258→            NDS_VEC_VSETVLI(vl, tmp_avl, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   259→            NDS_VEC_VAND_VI(NDS_VEC_V8, NDS_VEC_V8, 0);
   260→            NDS_VEC_VSETVLI(vl, rhs_cols2, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   261→            NDS_VEC_VADD_VI(NDS_VEC_V8, NDS_VEC_V8, 1);     // const one for vd4dot
   262→
   263→            // remain loop
   264→            while (rhs_cols2 > 0)
   265→            {
   266→                NDS_VEC_VSETVLI(vl, rhs_cols2, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   267→                NDS_VEC_VLE8_V(NDS_VEC_V16, pWt);
   268→                pWt += vl;
   269→                rhs_cols2 -= vl;
   270→
   271→                vl = (vl + 3) >> 2;
   272→                NDS_VEC_VSETVLI(vl, vl, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   273→                NDS_VEC_VD4DOTS_VV(NDS_VEC_V0, NDS_VEC_V16, NDS_VEC_V8);
   274→            }
   275→
   276→            NDS_VEC_VSETVLI(vl, rhs_cols, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);   //note the avl here
   277→            NDS_VEC_VREDSUM_VS(NDS_VEC_V0, NDS_VEC_V0, NDS_VEC_V31);
   278→
   279→            NDS_VEC_VSETVLI(vl, 1, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M1);
   280→            NDS_VEC_VLE32_V(NDS_VEC_V1, pBias);
   281→            NDS_VEC_VMUL_VX(NDS_VEC_V0, NDS_VEC_V0, in_offset);     // add input_offset contribution
   282→            NDS_VEC_VADD_VV(NDS_VEC_V0, NDS_VEC_V0, NDS_VEC_V1);    // add bias
   283→            NDS_VEC_VSE32_V(NDS_VEC_V0, pContri_buf);
   284→            pBias += vl;
   285→            pContri_buf += vl;
   286→            out_tensor_ch2--;
   287→        }
   288→    }
   289→    else
   290→    {
   291→        long out_tensor_ch2 = out_tensor_ch;
   292→        long vl;
   293→        NDS_VEC_VSETVLI(vl, 1, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M1);
   294→        NDS_VEC_VAND_VI(NDS_VEC_V31, NDS_VEC_V31, 0);   // const zero for redsum
   295→
   296→        while (out_tensor_ch2 > 0)
   297→        {
   298→            NDS_VEC_VSETVLI(vl, rhs_cols, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   299→            NDS_VEC_VAND_VI(NDS_VEC_V0, NDS_VEC_V0, 0);
   300→            NDS_VEC_VSETVLI(vl, rhs_cols, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   301→            NDS_VEC_VMV_V_I(NDS_VEC_V8, 1);     // const one for vd4dot
   302→
   303→            long rhs_cols2 = rhs_cols;
   304→            while (rhs_cols2 >= maxvl_per_loop)
   305→            {
   306→                NDS_VEC_VSETVLI(vl, rhs_cols2, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   307→                NDS_VEC_VLE8_V(NDS_VEC_V16, pWt);
   308→                pWt += vl;
   309→                rhs_cols2 -= vl;
   310→
   311→                NDS_VEC_VSETVLI(vl, vl, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   312→                NDS_VEC_VD4DOTS_VV(NDS_VEC_V0, NDS_VEC_V16, NDS_VEC_V8);
   313→            }
   314→
   315→            //clean one of the dot product source
   316→            long tmp_avl;
   317→            tmp_avl = rhs_cols2 + 3;
   318→            NDS_VEC_VSETVLI(vl, tmp_avl, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   319→            NDS_VEC_VAND_VI(NDS_VEC_V8, NDS_VEC_V8, 0);
   320→            NDS_VEC_VSETVLI(vl, rhs_cols2, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   321→            NDS_VEC_VADD_VI(NDS_VEC_V8, NDS_VEC_V8, 1);     // const one for vd4dot
   322→
   323→            // remain loop
   324→            while (rhs_cols2 > 0)
   325→            {
   326→                NDS_VEC_VSETVLI(vl, rhs_cols2, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   327→                NDS_VEC_VLE8_V(NDS_VEC_V16, pWt);
   328→                pWt += vl;
   329→                rhs_cols2 -= vl;
   330→
   331→                vl = (vl + 3) >> 2;
   332→                NDS_VEC_VSETVLI(vl, vl, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   333→                NDS_VEC_VD4DOTS_VV(NDS_VEC_V0, NDS_VEC_V16, NDS_VEC_V8);
   334→            }
   335→
   336→            NDS_VEC_VSETVLI(vl, rhs_cols, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);   //note the avl here
   337→            NDS_VEC_VREDSUM_VS(NDS_VEC_V0, NDS_VEC_V0, NDS_VEC_V31);
   338→
   339→            NDS_VEC_VSETVLI(vl, 1, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M1);
   340→            NDS_VEC_VMUL_VX(NDS_VEC_V0, NDS_VEC_V0, in_offset);
   341→            NDS_VEC_VSE32_V(NDS_VEC_V0, pContri_buf);
   342→            pContri_buf++;
   343→            out_tensor_ch2--;
   344→        }
   345→    }
   346→
   347→    for (long cur_batch = 0; cur_batch < in_tensor_batch; cur_batch++)
   348→    {
   349→        int8_t *two_column_buf2 = two_column_buf;
   350→        const int8_t *pKer_wt = &ker_weight[0];
   351→        const int32_t *pOut_scale = &out_scale[0];
   352→        const int32_t *pOut_shift = &out_shift[0];
   353→        const int32_t *pContri_buf = &contri_buf[0];
   354→
   355→        /* This part implements the im2col function */
   356→        for (long cur_group = 0; cur_group < groups; cur_group++)
   357→        {
   358→            int8_t *out = out_tensor + cur_group * out_ch_per_group;
   359→            for (long cur_out_y = 0; cur_out_y < out_tensor_dim_y; cur_out_y++)
   360→            {
   361→                for (long cur_out_x = 0; cur_out_x < out_tensor_dim_x; cur_out_x++)
   362→                {
   363→                    for (long cur_ker_y = cur_out_y * stride_y - pad_y; cur_ker_y < cur_out_y * stride_y - pad_y + (ker_dim_y - 1) * dilation_y + 1; cur_ker_y += dilation_y)
   364→                    {
   365→                        for (long cur_ker_x = cur_out_x * stride_x - pad_x; cur_ker_x < cur_out_x * stride_x - pad_x + (ker_dim_x - 1) * dilation_x + 1; cur_ker_x += dilation_x)
   366→                        {
   367→                            if (cur_ker_y < 0 || cur_ker_y >= in_tensor_dim_y || cur_ker_x < 0 || cur_ker_x >= in_tensor_dim_x)
   368→                            {
   369→                                nn_set_val_s8(two_column_buf2, -in_offset, ker_ch);
   370→                            }
   371→                            else
   372→                            {
   373→                                nn_dup_s8(in_tensor + (cur_ker_y * in_tensor_dim_x + cur_ker_x) * in_tensor_ch + cur_group * ker_ch,
   374→                                                two_column_buf2,
   375→                                                ker_ch * sizeof(*in_tensor));
   376→                            }
   377→                            two_column_buf2 += ker_ch;
   378→                        }
   379→                    }
   380→
   381→                    /* Computation is filed for every 2 columns */

   831→        while (rhs_cols2 >= SIMD_NUM_Q7)
   832→        {
   833→            suni_t wt_s8 = simd_load_8((s8_t*)pWt, 0);
   834→            sum = NDS_DSP_SMAQA(sum, wt_s8.l0, cst_one);
   835→            rhs_cols2 -= SIMD_NUM_Q7;
   836→            pWt += SIMD_NUM_Q7;
   837→        }
   838→
   839→        // rest loop
   840→        while (rhs_cols2 > 0)
   841→        {
   842→            sum += *pWt;
   843→            rhs_cols2--;
   844→            pWt++;
   845→        }
   846→
   847→        sum *= in_offset;
   848→
   849→        if (bias != NULL)
   850→        {
   851→            sum += bias[cur_out_ch];
   852→        }
   853→        contri_buf[cur_out_ch] = sum;
   854→    }
   855→
   856→    for (long cur_batch = 0; cur_batch < in_tensor_batch; cur_batch++)
   857→    {
   858→        int8_t *two_column_buf2 = (int8_t *)two_column_buf;
   859→        const int8_t *pKer_wt = &ker_weight[0];
   860→        const int32_t *pOut_scale = &out_scale[0];
   861→        const int32_t *pOut_shift = &out_shift[0];
   862→        const int32_t *pContri_buf = &contri_buf[0];
   863→
   864→        /* This part implements the im2col function */
   865→        for (long cur_group = 0; cur_group < groups; cur_group++)
   866→        {
   867→            int8_t *out = out_tensor + cur_group * out_ch_per_group;
   868→            for (long cur_out_y = 0; cur_out_y < out_tensor_dim_y; cur_out_y++)
   869→            {
   870→                for (long cur_out_x = 0; cur_out_x < out_tensor_dim_x; cur_out_x++)
   871→                {
   872→                    for (long cur_ker_y = cur_out_y * stride_y - pad_y; cur_ker_y < cur_out_y * stride_y - pad_y + (ker_dim_y - 1) * dilation_y + 1; cur_ker_y += dilation_y)
   873→                    {
   874→                        for (long cur_ker_x = cur_out_x * stride_x - pad_x; cur_ker_x < cur_out_x * stride_x - pad_x + (ker_dim_x - 1) * dilation_x + 1; cur_ker_x += dilation_x)
   875→                        {
   876→                            if (cur_ker_y < 0 || cur_ker_y >= in_tensor_dim_y || cur_ker_x < 0 || cur_ker_x >= in_tensor_dim_x)
   877→                            {
   878→                                nn_set_val_s8(two_column_buf2, -in_offset, ker_ch);
   879→                            }
   880→                            else
   881→                            {
   882→                                nn_dup_s8(in_tensor + (cur_ker_y * in_tensor_dim_x + cur_ker_x) * in_tensor_ch + cur_group * ker_ch,
   883→                                                two_column_buf2,
   884→                                                ker_ch * sizeof(*in_tensor));
   885→                            }
   886→                            two_column_buf2 += ker_ch;
   887→                        }
   888→                    }
   889→
   890→                    /* Computation is filed for every 2 columns */
   891→                    if (two_column_buf2 == (int8_t *)in_tmp_buf + 2 * ker_ch * ker_dim_y * ker_dim_x)
   892→                    {
   893→                        out = nn_mat_mult_kernel_s8_offset_v3(pKer_wt,
   894→                                                              two_column_buf,
   895→                                                              out_ch_per_group,
   896→                                                              out_tensor_ch,
   897→                                                              pOut_shift,
   898→                                                              pOut_scale,
   899→                                                              in_offset,
   900→                                                              out_offset,
   901→                                                              act_min,
   902→                                                              act_max,
   903→                                                              rhs_cols,
   904→                                                              rhs_cols_aligned,
   905→                                                              pContri_buf,
   906→                                                              out);
   907→
   908→                        /* counter reset */
   909→                        two_column_buf2 = (int8_t *)two_column_buf;
   910→                    }
   911→                }
   912→            }
   913→
   914→            /* left-over because odd number of out_tensor pixels */
   915→            if (two_column_buf2 != (int8_t *)two_column_buf)
   916→            {
   917→                const int8_t *pKer_wt2 = pKer_wt;
   918→                const int32_t *pOut_scale2 = pOut_scale;
   919→                const int32_t *pOut_shift2 = pOut_shift;
   920→                const int32_t *pContri_buf2 = pContri_buf;
   921→
   922→                for (long cur_out_ch_in_group = 0; cur_out_ch_in_group < out_ch_per_group; cur_out_ch_in_group++)
   923→                {
   924→                    const int8_t *pIn = (int8_t *)two_column_buf;
   925→                    long acc = 0;
   926→                    long rhs_cols2 = rhs_cols;
   927→
   928→                    while (rhs_cols2 >= SIMD_NUM_Q7)
   929→                    {
   930→                        suni_t in_val = simd_load_8((s8_t*)pIn, 0);
   931→                        suni_t wt_val = simd_load_8((s8_t*)pKer_wt2, 0);
   932→                        acc = NDS_DSP_SMAQA(acc, in_val.l0, wt_val.l0);
   933→                        pIn += SIMD_NUM_Q7;
   934→                        pKer_wt2 += SIMD_NUM_Q7;
   935→                        rhs_cols2 -= SIMD_NUM_Q7;
   936→                    }
   937→
   938→                    // rest loop
   939→                    while (rhs_cols2 > 0)
   940→                    {
   941→                        acc += *pIn * *pKer_wt2;
   942→                        pIn++;
   943→                        pKer_wt2++;
   944→                        rhs_cols2--;
   945→                    }
   946→
   947→                    // add contribution back
   948→                    acc += *pContri_buf2;
   949→
   950→                    acc = nn_requantize(acc, *pOut_scale2, *pOut_shift2);
   951→                    acc += out_offset;
   952→                    acc = MAX(acc, act_min);
   953→                    acc = MIN(acc, act_max);
   954→                    *out++ = (int8_t)acc;
   955→
   956→                    pContri_buf2++;
   957→                    pOut_scale2++;
   958→                    pOut_shift2++;
   959→                }
   960→                /* counter reset */
   961→                two_column_buf2 = (int8_t *)two_column_buf;
   962→            }
   963→            pKer_wt += out_ch_per_group * rhs_cols;
   964→            pOut_scale += out_ch_per_group;
   965→            pOut_shift += out_ch_per_group;
   966→            pContri_buf += out_ch_per_group;
   967→        }
   968→        in_tensor += (in_tensor_dim_x * in_tensor_dim_y * in_tensor_ch);
   969→        out_tensor += (out_tensor_dim_x * out_tensor_dim_y * out_tensor_ch);
   970→    }
   971→#else
   972→    (void)in_tmp_buf;
   973→    const int32_t groups = in_tensor_ch / ker_ch;
   974→    const int32_t out_ch_per_group = out_tensor_ch / groups;
   975→
   976→    if (in_tensor_ch % groups != 0 || out_tensor_ch % groups != 0)
   977→    {
   978→        return -1;
   979→    }
   980→
   981→    for (int32_t i_batch = 0; i_batch < in_tensor_batch; i_batch++)
   982→    {
   983→        for (int32_t i_group = 0; i_group < groups; i_group++)
   984→        {
   985→            for (int32_t i_out_ch_in_group = 0; i_out_ch_in_group < out_ch_per_group; i_out_ch_in_group++)
   986→            {
   987→                for (int32_t i_out_y = 0; i_out_y < out_tensor_dim_y; i_out_y++)
   988→                {
   989→                    for (int32_t i_out_x = 0; i_out_x < out_tensor_dim_x; i_out_x++)
   990→                    {
   991→                        const int32_t base_idx_y = stride_y * i_out_y - pad_y;
   992→                        const int32_t base_idx_x = stride_x * i_out_x - pad_x;
   993→                        const int32_t start_y_max = (-base_idx_y + dilation_y - 1) / dilation_y;
   994→                        const int32_t start_x_max = (-base_idx_x + dilation_x - 1) / dilation_x;
   995→                        const int32_t ker_y_start = MAX(0, start_y_max);
   996→                        const int32_t ker_x_start = MAX(0, start_x_max);
   997→                        const int32_t end_min_y = (in_tensor_dim_y - base_idx_y + dilation_y - 1) / dilation_y;
   998→                        const int32_t end_min_x = (in_tensor_dim_x - base_idx_x + dilation_x - 1) / dilation_x;
   999→                        const int32_t ker_y_end = MIN(ker_dim_y, end_min_y);
  1000→                        const int32_t ker_x_end = MIN(ker_dim_x, end_min_x);
  1001→                        const int32_t i_out_ch = i_group * out_ch_per_group + i_out_ch_in_group;
  1002→
  1003→                        int32_t conv_out = 0;
  1004→                        if (bias)
  1005→                        {
  1006→                            conv_out = bias[i_out_ch];
  1007→                        }
  1008→
  1009→                        for (int32_t i_ker_y = ker_y_start; i_ker_y < ker_y_end; i_ker_y++)
  1010→                        {
  1011→                            for (int32_t i_ker_x = ker_x_start; i_ker_x < ker_x_end; i_ker_x++)
  1012→                            {
  1013→                                const int32_t in_row = base_idx_y + dilation_y * i_ker_y;
  1014→                                const int32_t in_col = base_idx_x + dilation_x * i_ker_x;
  1015→                                for (int32_t i_ker_ch = 0; i_ker_ch < ker_ch; i_ker_ch++)
  1016→                                {
  1017→                                    int32_t i_input_ch = i_group * ker_ch + i_ker_ch;
  1018→                                    int32_t in_idx = (in_row * in_tensor_dim_x + in_col) * in_tensor_ch + i_input_ch;
  1019→                                    int32_t wt_idx = i_out_ch * ker_ch * ker_dim_y * ker_dim_x + (i_ker_y * ker_dim_x + i_ker_x) * ker_ch + i_ker_ch;
  1020→                                    int32_t in_val = in_tensor[in_idx] + in_offset;
  1021→                                    int32_t wt_val = ker_weight[wt_idx];
  1022→                                    conv_out += in_val * wt_val;
  1023→                                }
  1024→                            }
  1025→                        }
  1026→                        conv_out = nn_requantize(conv_out, out_scale[i_out_ch], out_shift[i_out_ch]);
  1027→                        conv_out += out_offset;
  1028→                        conv_out = MAX(conv_out, act_min);
  1029→                        conv_out = MIN(conv_out, act_max);
  1030→                        out_tensor[i_out_ch + (i_out_y * out_tensor_dim_x + i_out_x) * out_tensor_ch] = (int8_t)conv_out;
  1031→                    }
  1032→                }
  1033→            }
  1034→        }
  1035→        in_tensor += (in_tensor_dim_x * in_tensor_dim_y * in_tensor_ch);
  1036→        out_tensor += (out_tensor_dim_x * out_tensor_dim_y * out_tensor_ch);
  1037→    }
  1038→#endif  //ENA_EMPTY_FUN
  1039→
  1040→    /* Return to application */
  1041→    return 0;
  1042→}
