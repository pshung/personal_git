    25→//// Convolution Functions
    26→
    27→int32_t riscv_nn_conv_HWC_s8_s8_s8_sym_bias_fast_any(const q7_t * in_tensor,
    28→                                            const uint16_t in_tensor_dim_x,
    29→                                            const uint16_t in_tensor_dim_y,
    30→                                            const uint16_t in_tensor_ch,
    31→                                            const q7_t * ker_weight,
    32→                                            const uint16_t out_tensor_ch,
    33→                                            const uint16_t ker_dim_x,
    34→                                            const uint16_t ker_dim_y,
    35→                                            const uint16_t pad_x,
    36→                                            const uint16_t pad_y,
    37→                                            const uint16_t stride_x,
    38→                                            const uint16_t stride_y,
    39→                                            const q31_t * bias,
    40→                                            const uint16_t pre_rshift,
    41→                                            const uint16_t out_scale,
    42→                                            const uint16_t post_rshift,
    43→                                            q7_t * out_tensor,
    44→                                            const uint16_t out_tensor_dim_x,
    45→                                            const uint16_t out_tensor_dim_y,
    46→                                            q15_t * in_tmp_buf)
    47→{
    48→#ifdef ENA_EMPTY_FUN
    49→    //empty body
    50→#else
    51→    if (in_tensor_ch % 4 != 0 || out_tensor_ch % 2 != 0)
    52→    {
    53→        /* check if the input dimension meets the constraints */
    54→        return -1;
    55→    }
    56→
    57→#if defined(ENA_VEC_ISA) && defined(ENA_TILING)
    58→    int16_t   i_out_y, i_out_x, i_ker_y, i_ker_x;
    59→
    60→    /*
    61→     *  Use im2col with 4-column batching and nn_mat_mul_kernel_q7_bias_2sft_unroll4
    62→     *  which implements a 4x4 tiled GEMM with vd4dots, vectorizing across the
    63→     *  reduction dimension K (=in_tensor_ch * ker_dim_x * ker_dim_y).
    64→     */
    65→
    66→    q7_t     *pBuffer = (q7_t*)in_tmp_buf;
    67→    q7_t     *pOut = out_tensor;
    68→    int bound_x_begin = (pad_x + stride_x - 1) / stride_x;
    69→    int bound_x_end   = (in_tensor_dim_x + pad_x - ker_dim_x + stride_x) / stride_x;
    70→    int bound_y_begin = (pad_y + stride_y - 1) / stride_y;
    71→    int bound_y_end   = (in_tensor_dim_y + pad_y - ker_dim_y + stride_y) / stride_y;
    72→
    73→    /* top part */
    74→    for (i_out_y = 0; i_out_y < bound_y_begin; i_out_y++)
    75→    {
    76→        for (i_out_x = 0; i_out_x < out_tensor_dim_x; i_out_x++)
    77→        {
    78→            for (i_ker_y = i_out_y * stride_y - pad_y; i_ker_y < i_out_y * stride_y - pad_y + ker_dim_y; i_ker_y++)
    79→            {
    80→                for (i_ker_x = i_out_x * stride_x - pad_x; i_ker_x < i_out_x * stride_x - pad_x + ker_dim_x; i_ker_x++)
    81→                {
    82→                    if (i_ker_y < 0 || i_ker_y >= in_tensor_dim_y || i_ker_x < 0 || i_ker_x >= in_tensor_dim_x)
    83→                    {
    84→                        nn_set_zero_s8(pBuffer, sizeof(q7_t)*in_tensor_ch);
    85→                    }
    86→                    else
    87→                    {
    88→                        nn_dup_s8
    89→                        ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim_x + i_ker_x) * in_tensor_ch, pBuffer, in_tensor_ch);
    90→                    }
    91→                    pBuffer += in_tensor_ch;
    92→                }
    93→            }
    94→
    95→            if (pBuffer == (q7_t *)in_tmp_buf + 4 * in_tensor_ch * ker_dim_x * ker_dim_y)
    96→            {
    97→                pOut = nn_mat_mul_kernel_q7_bias_2sft_unroll4(ker_weight,
    98→                                                      (q7_t*)in_tmp_buf,
    99→                                                      out_tensor_ch,
   100→                                                      in_tensor_ch * ker_dim_x * ker_dim_y,
   101→                                                      pre_rshift,
   102→                                                      out_scale,
   103→                                                      post_rshift,
   104→                                                      bias,
   105→                                                      pOut);
   106→                pBuffer = (q7_t *)in_tmp_buf;
   107→            }
   108→        }
   109→    }
   110→
   111→    /* middle part -- split x into left, mid, right */
   112→    for (; i_out_y < bound_y_end; i_out_y++)
   113→    {
   114→        /* left part */
   115→        for (i_out_x = 0; i_out_x < bound_x_begin; i_out_x++)
   116→        {
   117→            for (i_ker_y = i_out_y * stride_y - pad_y; i_ker_y < i_out_y * stride_y - pad_y + ker_dim_y;
   118→                    i_ker_y++)
   119→            {
   120→                for (i_ker_x = i_out_x * stride_x - pad_x; i_ker_x < i_out_x * stride_x - pad_x + ker_dim_x;
   121→                        i_ker_x++)
   122→                {
   123→                    if (i_ker_x < 0 || i_ker_x >= in_tensor_dim_x)
   124→                    {
   125→                        nn_set_zero_s8(pBuffer, sizeof(q7_t)*in_tensor_ch);
   126→                    }
   127→                    else
   128→                    {
   129→                        nn_dup_s8
   130→                        ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim_x + i_ker_x) * in_tensor_ch, pBuffer, in_tensor_ch);
   131→                    }
   132→                    pBuffer += in_tensor_ch;
   133→                }
   134→            }
   135→
   136→            if (pBuffer == (q7_t *)in_tmp_buf + 4 * in_tensor_ch * ker_dim_x * ker_dim_y)
   137→            {
   138→                pOut = nn_mat_mul_kernel_q7_bias_2sft_unroll4(ker_weight,
   139→                                                      (q7_t*)in_tmp_buf,
   140→                                                      out_tensor_ch,
   141→                                                      in_tensor_ch * ker_dim_x * ker_dim_y,
   142→                                                      pre_rshift,
   143→                                                      out_scale,
   144→                                                      post_rshift,
   145→                                                      bias,
   146→                                                      pOut);
   147→                pBuffer = (q7_t *)in_tmp_buf;
   148→            }
   149→        }
   150→
   151→        /* mid part -- no padding, use bulk copy */
   152→        for (; i_out_x < bound_x_end; i_out_x++)
   153→        {
   154→            for (i_ker_y = i_out_y * stride_y - pad_y; i_ker_y < i_out_y * stride_y - pad_y + ker_dim_y;
   155→                    i_ker_y++)
   156→            {
   157→                nn_dup_s8
   158→                ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim_x + i_out_x * stride_x - pad_x) * in_tensor_ch, pBuffer, in_tensor_ch * ker_dim_x);
   159→                pBuffer += in_tensor_ch * ker_dim_x;
   160→            }
   161→
   162→            if (pBuffer == (q7_t *)in_tmp_buf + 4 * in_tensor_ch * ker_dim_x * ker_dim_y)
   163→            {
   164→                pOut = nn_mat_mul_kernel_q7_bias_2sft_unroll4(ker_weight,
   165→                                                      (q7_t*)in_tmp_buf,
   166→                                                      out_tensor_ch,
   167→                                                      in_tensor_ch * ker_dim_x * ker_dim_y,
   168→                                                      pre_rshift,
   169→                                                      out_scale,
   170→                                                      post_rshift,
   171→                                                      bias,
   172→                                                      pOut);
   173→                pBuffer = (q7_t *)in_tmp_buf;
   174→            }
   175→        }
   176→
   177→        /* right part */
   178→        for (; i_out_x < out_tensor_dim_x; i_out_x++)
   179→        {
   180→            for (i_ker_y = i_out_y * stride_y - pad_y; i_ker_y < i_out_y * stride_y - pad_y + ker_dim_y;
   181→                    i_ker_y++)
   182→            {
   183→                for (i_ker_x = i_out_x * stride_x - pad_x; i_ker_x < i_out_x * stride_x - pad_x + ker_dim_x;
   184→                        i_ker_x++)
   185→                {
   186→                    if (i_ker_x < 0 || i_ker_x >= in_tensor_dim_x)
   187→                    {
   188→                        nn_set_zero_s8(pBuffer, sizeof(q7_t)*in_tensor_ch);
   189→                    }
   190→                    else
   191→                    {
   192→                        nn_dup_s8
   193→                        ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim_x + i_ker_x) * in_tensor_ch, pBuffer, in_tensor_ch);
   194→                    }
   195→                    pBuffer += in_tensor_ch;
   196→                }
   197→            }
   198→
   199→            if (pBuffer == (q7_t *)in_tmp_buf + 4 * in_tensor_ch * ker_dim_x * ker_dim_y)
   200→            {
   201→                pOut = nn_mat_mul_kernel_q7_bias_2sft_unroll4(ker_weight,
   202→                                                      (q7_t*)in_tmp_buf,
   203→                                                      out_tensor_ch,
   204→                                                      in_tensor_ch * ker_dim_x * ker_dim_y,
   205→                                                      pre_rshift,
   206→                                                      out_scale,
   207→                                                      post_rshift,
   208→                                                      bias,
   209→                                                      pOut);
   210→                pBuffer = (q7_t *)in_tmp_buf;
   211→            }
   212→        }
   213→    }
   214→
   215→    /* bottom part */
   216→    for (; i_out_y < out_tensor_dim_y; i_out_y++)
   217→    {
   218→        for (i_out_x = 0; i_out_x < out_tensor_dim_x; i_out_x++)
   219→        {
   220→            for (i_ker_y = i_out_y * stride_y - pad_y; i_ker_y < i_out_y * stride_y - pad_y + ker_dim_y;
   221→                    i_ker_y++)
   222→            {
   223→                for (i_ker_x = i_out_x * stride_x - pad_x; i_ker_x < i_out_x * stride_x - pad_x + ker_dim_x;
   224→                        i_ker_x++)
   225→                {
   226→                    if (i_ker_y < 0 || i_ker_y >= in_tensor_dim_y || i_ker_x < 0 || i_ker_x >= in_tensor_dim_x)
   227→                    {
   228→                        nn_set_zero_s8(pBuffer, sizeof(q7_t)*in_tensor_ch);
   229→                    }
   230→                    else
   231→                    {
   232→                        nn_dup_s8
   233→                        ((q7_t *) in_tensor + (i_ker_y * in_tensor_dim_x + i_ker_x) * in_tensor_ch, pBuffer, in_tensor_ch);
   234→                    }
   235→                    pBuffer += in_tensor_ch;
   236→                }
   237→            }
   238→
   239→            if (pBuffer == (q7_t *)in_tmp_buf + 4 * in_tensor_ch * ker_dim_x * ker_dim_y)
   240→            {
   241→                pOut = nn_mat_mul_kernel_q7_bias_2sft_unroll4(ker_weight,
   242→                                                      (q7_t*)in_tmp_buf,
   243→                                                      out_tensor_ch,
   244→                                                      in_tensor_ch * ker_dim_x * ker_dim_y,
   245→                                                      pre_rshift,
   246→                                                      out_scale,
   247→                                                      post_rshift,
   248→                                                      bias,
   249→                                                      pOut);
   250→                pBuffer = (q7_t *)in_tmp_buf;
   251→            }
   252→        }
   253→    }
   254→
   255→    /* check if there are left-over pixels in the buffer (1..3 pixels) */
   256→    if (pBuffer != (q7_t *)in_tmp_buf)
   257→    {
   258→        int leftover_bytes = pBuffer - (q7_t *)in_tmp_buf;
   259→        int col_size = in_tensor_ch * ker_dim_x * ker_dim_y;
   260→
   261→        /* drain 2 pixels if we have 2 or 3 left over */
   262→        if (leftover_bytes >= 2 * col_size)
   263→        {
   264→            pOut = nn_mat_mul_kernel_q7_bias_2sft(ker_weight,
   265→                                                  (q7_t*)in_tmp_buf,
   266→                                                  out_tensor_ch,
   267→                                                  col_size,
   268→                                                  pre_rshift,
   269→                                                  out_scale,
   270→                                                  post_rshift,
   271→                                                  bias,
   272→                                                  pOut);
   273→            leftover_bytes -= 2 * col_size;
   274→        }
   275→
   276→        /* handle 1 remaining pixel with scalar path */
   277→        if (leftover_bytes > 0)
   278→        {
   279→            const q7_t *pA = ker_weight;
   280→            q7_t *pB = pBuffer - col_size;
   281→            int i;
   282→
   283→            for (i = 0; i < out_tensor_ch; i++)
   284→            {
   285→                q31_t sum = bias[i];
   286→                uint16_t colCnt = col_size;
   287→                q7_t *pB_iter = pB;
   288→
   289→                while (colCnt)
   290→                {
   291→                    sum += *pA++ * *pB_iter++;
   292→                    colCnt--;
   293→                }
   294→
   295→                sum = (sum >> pre_rshift) * out_scale + NN_ROUND(post_rshift);
   296→                *pOut++ = NDS_ISA_SATS((sum >> post_rshift), 8);
   297→            }
   298→        }
   299→    }
   300→#elif defined(ENA_VEC_ISA) || defined(ENA_DSP_ISA_V5)
   301→    int i_out_y, i_out_x, i_ker_y, i_ker_x;
   302→
   303→    /* -----------------------
   304→     *  Here we use in_tmp_buf as q15_t internally as computation are done with q15_t level
   305→     *  im2col are done to output in q15_t format from q7_t input
   306→     */
   307→
   308→    q7_t     *pBuffer = (q7_t*)in_tmp_buf;
   309→    q7_t     *pOut = out_tensor;
   310→    int bound_x_begin = (pad_x + stride_x - 1) / stride_x;
   311→    int bound_x_end   = (in_tensor_dim_x + pad_x - ker_dim_x + stride_x) / stride_x;
   312→    int bound_y_begin = (pad_y + stride_y - 1) / stride_y;
   313→    int bound_y_end   = (in_tensor_dim_y + pad_y - ker_dim_y + stride_y) / stride_y;
   314→
   315→    /*
   316→     *  Here we split the entire matrix into three regions depending on the padding situation
   317→     *    Top: i_out_y from 0 to padding - 1
   318→     * Middle: i_out_y from padding to dim_im_out-padding-1
   319→     * Bottom: i_out_y from dim_im_out-padding to dim_im_out-1
   320→     */
   321→
   322→    /* top part */
   323→    for (i_out_y = 0; i_out_y < bound_y_begin; i_out_y++)
   324→    {
   325→        for (i_out_x = 0; i_out_x < out_tensor_dim_x; i_out_x++)
   326→        {
   327→            /* This part implements the im2col function */
   328→            for (i_ker_y = i_out_y * stride_y - pad_y; i_ker_y < i_out_y * stride_y - pad_y + ker_dim_y; i_ker_y++)
   329→            {
   330→                for (i_ker_x = i_out_x * stride_x - pad_x; i_ker_x < i_out_x * stride_x - pad_x + ker_dim_x; i_ker_x++)
   331→                {
   332→                    if (i_ker_y < 0 || i_ker_y >= in_tensor_dim_y || i_ker_x < 0 || i_ker_x >= in_tensor_dim_x)
   333→                    {
   334→                        nn_set_zero_s8(pBuffer, sizeof(q7_t)*in_tensor_ch);
   335→                    }
   336→                    else
   337→                    {
   338→                        nn_dup_s8((q7_t *) in_tensor + (i_ker_y * in_tensor_dim_x + i_ker_x) * in_tensor_ch,
   339→                                    pBuffer, in_tensor_ch);
   340→                    }
   341→                    pBuffer += in_tensor_ch;
   342→                }
   343→            }
   344→
   345→            if (pBuffer == (q7_t*)in_tmp_buf + 2 * in_tensor_ch * ker_dim_x * ker_dim_y)
   346→            {
   347→                pOut = nn_mat_mul_kernel_q7_bias_2sft(ker_weight,
   348→                                                      (q7_t*)in_tmp_buf,
   349→                                                      out_tensor_ch,
   350→                                                      in_tensor_ch * ker_dim_x * ker_dim_y,
   351→                                                      pre_rshift,
   352→                                                      out_scale,
   353→                                                      post_rshift,
   354→                                                      bias,
   355→                                                      pOut);
   356→                /* counter reset */
   357→                pBuffer = (q7_t *)in_tmp_buf;
   358→            }
   359→        }
   360→    }
   361→
   362→    /* middle part, here we also divide the x into left, mid and right */
   363→    for (; i_out_y < bound_y_end; i_out_y++)
   364→    {
   365→        /* left part */
   366→        for (i_out_x = 0; i_out_x < bound_x_begin; i_out_x++)
   367→        {
   368→            /* This part implements the im2col function */
   369→            for (i_ker_y = i_out_y * stride_y - pad_y; i_ker_y < i_out_y * stride_y - pad_y + ker_dim_y;
   370→                    i_ker_y++)
   371→            {
   372→                for (i_ker_x = i_out_x * stride_x - pad_x; i_ker_x < i_out_x * stride_x - pad_x + ker_dim_x;
   373→                        i_ker_x++)
   374→                {
   375→                    if (i_ker_x < 0 || i_ker_x >= in_tensor_dim_x)
   376→                    {
   377→                        nn_set_zero_s8(pBuffer, sizeof(q7_t)*in_tensor_ch);
   378→                    }
   379→                    else
   380→                    {
   381→                        nn_dup_s8((q7_t *) in_tensor + (i_ker_y * in_tensor_dim_x + i_ker_x) * in_tensor_ch,
   382→                                    pBuffer, in_tensor_ch);
   383→                    }
   384→                    pBuffer += in_tensor_ch;
   385→                }
   386→            }
   387→
   388→            if (pBuffer == (q7_t*)in_tmp_buf + 2 * in_tensor_ch * ker_dim_x * ker_dim_y)
   389→            {
   390→                pOut = nn_mat_mul_kernel_q7_bias_2sft(ker_weight,
   391→                                                      (q7_t *)in_tmp_buf,
   392→                                                      out_tensor_ch,
   393→                                                      in_tensor_ch * ker_dim_x * ker_dim_y,
   394→                                                      pre_rshift,
   395→                                                      out_scale,
   396→                                                      post_rshift,
   397→                                                      bias,
   398→                                                      pOut);
   399→                /* counter reset */
   400→                pBuffer = (q7_t*)in_tmp_buf;
   401→            }
   402→        }
   403→
   404→        /* mid part */
   405→        for (; i_out_x < bound_x_end; i_out_x++)
   406→        {
   407→            /* This part implements the im2col function */
   408→            for (i_ker_y = i_out_y * stride_y - pad_y; i_ker_y < i_out_y * stride_y - pad_y + ker_dim_y;
   409→                    i_ker_y++)
   410→            {
   411→                nn_dup_s8((q7_t *) in_tensor + (i_ker_y * in_tensor_dim_x + i_out_x * stride_x - pad_x) * in_tensor_ch,
   412→                            pBuffer, in_tensor_ch * ker_dim_x);
   413→                pBuffer += in_tensor_ch * ker_dim_x;
   414→            }
   415→
   416→            if (pBuffer == (q7_t*)in_tmp_buf + 2 * in_tensor_ch * ker_dim_x * ker_dim_y)
   417→            {
   418→                pOut = nn_mat_mul_kernel_q7_bias_2sft(ker_weight,
   419→                                                      (q7_t*)in_tmp_buf,
   420→                                                      out_tensor_ch,
   421→                                                      in_tensor_ch * ker_dim_x * ker_dim_y,
   422→                                                      pre_rshift,
   423→                                                      out_scale,
   424→                                                      post_rshift,
   425→                                                      bias,
   426→                                                      pOut);
   427→                /* counter reset */
   428→                pBuffer = (q7_t*)in_tmp_buf;
   429→            }
   430→        }
   431→
   432→        /* right part */
   433→        for (; i_out_x < out_tensor_dim_x; i_out_x++)
   434→        {
   435→            /* This part implements the im2col function */
   436→            for (i_ker_y = i_out_y * stride_y - pad_y; i_ker_y < i_out_y * stride_y - pad_y + ker_dim_y;
   437→                    i_ker_y++)
   438→            {
   439→                for (i_ker_x = i_out_x * stride_x - pad_x; i_ker_x < i_out_x * stride_x - pad_x + ker_dim_x;
   440→                        i_ker_x++)
   441→                {
   442→                    if (i_ker_x < 0 || i_ker_x >= in_tensor_dim_x)
   443→                    {
   444→                        nn_set_zero_s8(pBuffer, sizeof(q7_t)*in_tensor_ch);
   445→                    }
   446→                    else
   447→                    {
   448→                        nn_dup_s8((q7_t *) in_tensor + (i_ker_y * in_tensor_dim_x + i_ker_x) * in_tensor_ch,
   449→                                    pBuffer, in_tensor_ch);
   450→                    }
   451→                    pBuffer += in_tensor_ch;
   452→                }
   453→            }
   454→
   455→            if (pBuffer == (q7_t*)in_tmp_buf + 2 * in_tensor_ch * ker_dim_x * ker_dim_y)
   456→            {
   457→                pOut = nn_mat_mul_kernel_q7_bias_2sft(ker_weight,
   458→                                                      (q7_t*)in_tmp_buf,
   459→                                                      out_tensor_ch,
   460→                                                      in_tensor_ch * ker_dim_x * ker_dim_y,
   461→                                                      pre_rshift,
   462→                                                      out_scale,
   463→                                                      post_rshift,
   464→                                                      bias,
   465→                                                      pOut);
   466→                /* counter reset */
   467→                pBuffer = (q7_t*)in_tmp_buf;
   468→            }
   469→        }
   470→    }
   471→
