    25→    25→//// Convolution Functions
    26→    26→int32_t riscv_nn_conv_HWC_s8_s8_s8_asym_bias_any_dilated(const int8_t * in_tensor,
    27→    27→                                                         const uint16_t in_tensor_dim_x,
    28→    28→                                                         const uint16_t in_tensor_dim_y,
    29→    29→                                                         const uint16_t in_tensor_ch,
    30→    30→                                                         const uint16_t in_tensor_batch,
    31→    31→                                                         const int8_t * ker_weight,
    32→    32→                                                         const uint16_t out_tensor_ch,
    33→    33→                                                         const uint16_t ker_dim_x,
    34→    34→                                                         const uint16_t ker_dim_y,
    35→    35→                                                         const uint16_t ker_ch,
    36→    36→                                                         const uint16_t pad_x,
    37→    37→                                                         const uint16_t pad_y,
    38→    38→                                                         const uint16_t stride_x,
    39→    39→                                                         const uint16_t stride_y,
    40→    40→                                                         const int32_t * bias,
    41→    41→                                                         int8_t * out_tensor,
    42→    42→                                                         const int32_t * out_shift,
    43→    43→                                                         const int32_t * out_scale,
    44→    44→                                                         const int32_t out_offset,    //value is in the range of [-128, 127]
    45→    45→                                                         const int32_t in_offset,     //value is in the range of [-127, 128]
    46→    46→                                                         const int32_t act_min,
    47→    47→                                                         const int32_t act_max,
    48→    48→                                                         const uint16_t out_tensor_dim_x,
    49→    49→                                                         const uint16_t out_tensor_dim_y,
    50→    50→                                                         const int32_t dilation_x,
    51→    51→                                                         const int32_t dilation_y,
    52→    52→                                                         int16_t * in_tmp_buf)
    53→    53→{
    54→    54→#ifdef ENA_EMPTY_FUN
    55→    55→    //empty body
    56→    56→#elif defined(ENA_VEC_ISA) && (NDS_VEC_RVV_VERSION >= 100) && defined(ENA_NDS_V5_VEC_DOT_PROD) && defined(ENA_VEC_ELEN64) && defined(ENA_VEC_INTRINSIC)
    57→    57→    int i_batch;
    58→    58→    for (i_batch = 0; i_batch < in_tensor_batch; i_batch++)
    59→    59→    {
    60→    60→        int32_t i_out_y, i_out_x, i_ker_y, i_ker_x;
    61→    61→
    62→    62→        /* Generate two columns from the input tensor a GEMM computation */
    63→    63→        int8_t *two_column_buf = (int8_t *)in_tmp_buf;
    64→    64→        int8_t *out = out_tensor;
    65→    65→        long vl;
    66→    66→
    67→    67→        /* This part implements the im2col function */
    68→    68→        for (i_out_y = 0; i_out_y < out_tensor_dim_y; i_out_y++)
    69→    69→        {
    70→    70→            for (i_out_x = 0; i_out_x < out_tensor_dim_x; i_out_x++)
    71→    71→            {
    72→    72→                for (i_ker_y = i_out_y * stride_y - pad_y; i_ker_y < i_out_y * stride_y - pad_y + (ker_dim_y - 1) * dilation_y + 1; i_ker_y += dilation_y)
    73→    73→                {
    74→    74→                    for (i_ker_x = i_out_x * stride_x - pad_x; i_ker_x < i_out_x * stride_x - pad_x + (ker_dim_x - 1) * dilation_x + 1; i_ker_x += dilation_x)
    75→    75→                    {
    76→    76→                        if (i_ker_y < 0 || i_ker_y >= in_tensor_dim_y || i_ker_x < 0 || i_ker_x >= in_tensor_dim_x)
    77→    77→                        {
    78→    78→                            nn_set_val_s8(two_column_buf, -in_offset, in_tensor_ch);
    79→    79→                        }
    80→    80→                        else
    81→    81→                        {
    82→    82→                            /* Copying the pixel data to column */
    83→    83→                            nn_dup_s8(in_tensor + (i_ker_y * in_tensor_dim_x + i_ker_x) * in_tensor_ch, two_column_buf, in_tensor_ch);
    84→    84→                        }
    85→    85→                        two_column_buf += in_tensor_ch;
    86→    86→                    }
    87→    87→                }
    88→    88→
    89→    89→                /* Computation is filed for every 2 columns */
    90→    90→                if (two_column_buf == (int8_t *)in_tmp_buf + 2 * in_tensor_ch * ker_dim_y * ker_dim_x)
    91→    91→                {
    92→    92→                    out = nn_mat_mult_kernel_s8_offset(ker_weight,
    93→    93→                                                       (int8_t *)in_tmp_buf,
    94→    94→                                                       out_tensor_ch,
    95→    95→                                                       out_shift,
    96→    96→                                                       out_scale,
    97→    97→                                                       in_offset,
    98→    98→                                                       out_offset,
    99→    99→                                                       act_min,
   100→   100→                                                       act_max,
   101→   101→                                                       in_tensor_ch * ker_dim_y * ker_dim_x,
   102→   102→                                                       bias,
   103→   103→                                                       out);
   104→   104→
   105→   105→                    /* counter reset */
   106→   106→                    two_column_buf = (int8_t *)in_tmp_buf;
   107→   107→                }
   108→   108→            }
   109→   109→        }
   110→   110→
   111→   111→        /* left-over because odd number of output pixels */
   112→   112→        if (two_column_buf != (int8_t *)in_tmp_buf)
   113→   113→        {
   114→   114→            const int8_t *ker_a = ker_weight;
   115→   115→            int i;
   116→   116→            long col_count = (in_tensor_ch * ker_dim_y * ker_dim_x);
   117→   117→            long vlenb = NDS_VEC_CSR_GET_VLENB();
   118→   118→
   119→   119→            vl = __riscv_vsetvl_e8m4(col_count);
   120→   120→            vint8m4_t vOne_i8 = __riscv_vmv_v_x_i8m4(1, vl);                   //const one for vd4dot
   121→   121→
   122→   122→            //set SEW for 1st time VMV_S_X
   123→   123→            vl = __riscv_vsetvl_e32m4(col_count);
   124→   124→
   125→   125→            for (i = 0; i < out_tensor_ch; i++)
   126→   126→            {
   127→   127→                int32_t sum = 0;
   128→   128→                if (bias)
   129→   129→                {
   130→   130→                    sum = bias[i];
   131→   131→                }
   132→   132→
   133→   133→                /* Point to the beginning of the im2col buffer where the input is available as a rearranged column */
   134→   134→                const int8_t *ip_as_col = (int8_t *)in_tmp_buf;
   135→   135→                long col_count2 = col_count;
   136→   136→                int32_t sum_ker_a = 0;
   137→   137→
   138→   138→                //init
   139→   139→                vint32m1_t vSum_e0 = __riscv_vmv_s_x_i32m1(sum, 1);
   140→   140→                vint32m1_t vSum_ker_a_e0 = __riscv_vmv_s_x_i32m1(sum_ker_a, 1);
   141→   141→
   142→   142→                //clear the accumulator
   143→   143→                vint32m4_t vAcc0 = __riscv_vand(__riscv_vundefined_i32m4(), 0, vl);
   144→   144→                vint32m4_t vAcc1 = __riscv_vand(__riscv_vundefined_i32m4(), 0, vl);
   145→   145→
   146→   146→                while (col_count2 >= (vlenb * NDS_VEC_VTYPE_LMUL_ID_M4))
   147→   147→                {
   148→   148→                    vint8m4_t vWt_i8, vIn_i8;
   149→   149→                    vl = __riscv_vsetvl_e8m4(col_count2);
   150→   150→                    vWt_i8 = __riscv_vle8_v_i8m4(ker_a, vl);
   151→   151→                    vIn_i8 = __riscv_vle8_v_i8m4(ip_as_col, vl);
   152→   152→                    col_count2 -= vl;
   153→   153→                    ker_a += vl;
   154→   154→                    ip_as_col += vl;
   155→   155→
   156→   156→                    vl = __riscv_vsetvl_e32m4(vl);
   157→   157→                    vAcc0 = __riscv_vd4dots_vv_i32m4_tu(vAcc0, vWt_i8, vIn_i8, vl);
   158→   158→                    vAcc1 = __riscv_vd4dots_vv_i32m4_tu(vAcc1, vWt_i8, vOne_i8, vl);
   159→   159→                }
   160→   160→
   161→   161→                if(col_count2 > 0)
   162→   162→                {
   163→   163→                    vint8m4_t vWt_i8, vIn_i8;
   164→   164→                    vl = __riscv_vsetvl_e8m4(col_count+3);
   165→   165→                    vWt_i8 = __riscv_vand(__riscv_vundefined_i8m4(), 0, vl);
   166→   166→
   167→   167→                    vl = __riscv_vsetvl_e8m4(col_count2);
   168→   168→                    vWt_i8 = __riscv_vle8_v_i8m4_tu(vWt_i8, ker_a, vl);
   169→   169→                    vIn_i8 = __riscv_vle8_v_i8m4(ip_as_col, vl);
   170→   170→                    col_count2 -= vl;
   171→   171→                    ker_a += vl;
   172→   172→                    ip_as_col += vl;
   173→   173→
   174→   174→                    vl = (vl + 3) >> 2;
   175→   175→                    vl = __riscv_vsetvl_e32m4(vl);
   176→   176→                    vAcc0 = __riscv_vd4dots_vv_i32m4_tu(vAcc0, vWt_i8, vIn_i8, vl);
   177→   177→                    vAcc1 = __riscv_vd4dots_vv_i32m4_tu(vAcc1, vWt_i8, vOne_i8, vl);
   178→   178→                }
   179→   179→
   180→   180→                vl = __riscv_vsetvl_e32m4(col_count);       //note the avl here
   181→   181→                vSum_e0 = __riscv_vredsum_tu(vSum_e0, vAcc0, vSum_e0, vl);
   182→   182→                vSum_ker_a_e0 = __riscv_vredsum_tu(vSum_ker_a_e0, vAcc1, vSum_ker_a_e0, vl);
   183→   183→
   184→   184→                //move vr results to gpr
   185→   185→                sum = __riscv_vmv_x(vSum_e0);
   186→   186→                sum_ker_a = __riscv_vmv_x(vSum_ker_a_e0);
   187→   187→
   188→   188→                //add offset back
   189→   189→                sum += in_offset * sum_ker_a;
   190→   190→
   191→   191→                sum = nn_requantize(sum, out_scale[i], out_shift[i]);
   192→   192→                sum += out_offset;
   193→   193→                sum = MAX(sum, act_min);
   194→   194→                sum = MIN(sum, act_max);
   195→   195→                *out++ = (int8_t)sum;
   196→   196→            }
   197→   197→        }
   198→   198→        in_tensor += (in_tensor_dim_x * in_tensor_dim_y * in_tensor_ch);
   199→   199→        out_tensor += (out_tensor_dim_x * out_tensor_dim_y * out_tensor_ch);
   200→   200→    }
   201→   201→#elif defined(ENA_VEC_ISA) && (NDS_VEC_RVV_VERSION >= 100) && defined(ENA_NDS_V5_VEC_DOT_PROD)
   202→   202→    const long groups = in_tensor_ch / ker_ch;
   203→   203→    const long out_ch_per_group = out_tensor_ch / groups;
   204→   204→
   205→   205→    if (in_tensor_ch % groups != 0 || out_tensor_ch % groups != 0)
   206→   206→    {
   207→   207→        return -1;
   208→   208→    }
   209→   209→
   210→   210→    //--- calculate the buffer address ---
   211→   211→    const long align_byte = sizeof(long);
   212→   212→    // buffer-1
   213→   213→    int8_t *two_column_buf = (int8_t*)in_tmp_buf;
   214→   214→    two_column_buf = (int8_t*)nn_align((unsigned long)two_column_buf, align_byte);  // align the starting addresss
   215→   215→    // buffer-2
   216→   216→    long two_column_buf_size = 2 * ker_ch * ker_dim_x * ker_dim_y * sizeof(*in_tensor);
   217→   217→    int32_t *contri_buf = (int32_t*)(two_column_buf + two_column_buf_size);
   218→   218→    contri_buf = (int32_t*)nn_align((unsigned long)contri_buf, align_byte);         // align the starting addresss
   219→   219→
   220→   220→    const long rhs_cols = ker_dim_y * ker_dim_x * ker_ch;
   221→   221→    const long rhs_cols_aligned = nn_align((long)rhs_cols, align_byte);
   222→   222→    long maxvl_per_loop;
   223→   223→
   224→   224→    // calculate the weight contribution
   225→   225→    const int8_t *pWt = ker_weight;
   226→   226→    int32_t *pContri_buf = contri_buf;
   227→   227→    const int32_t *pBias = bias;
   228→   228→    NDS_VEC_VSETVLI(maxvl_per_loop, reg_x0, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   229→   229→    if (bias != NULL)
   230→   230→    {
   231→   231→        long out_tensor_ch2 = out_tensor_ch;
   232→   232→        long vl;
   233→   233→        NDS_VEC_VSETVLI(vl, 1, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M1);
   234→   234→        NDS_VEC_VAND_VI(NDS_VEC_V31, NDS_VEC_V31, 0);   // const zero for redsum
   235→   235→
   236→   236→        while (out_tensor_ch2 > 0)
   237→   237→        {
   238→   238→            NDS_VEC_VSETVLI(vl, rhs_cols, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   239→   239→            NDS_VEC_VAND_VI(NDS_VEC_V0, NDS_VEC_V0, 0);
   240→   240→            NDS_VEC_VSETVLI(vl, rhs_cols, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   241→   241→            NDS_VEC_VMV_V_I(NDS_VEC_V8, 1);     // const one for vd4dot
   242→   242→
   243→   243→            long rhs_cols2 = rhs_cols;
   244→   244→            while (rhs_cols2 >= maxvl_per_loop)
   245→   245→            {
   246→   246→                NDS_VEC_VSETVLI(vl, rhs_cols2, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   247→   247→                NDS_VEC_VLE8_V(NDS_VEC_V16, pWt);
   248→   248→                pWt += vl;
   249→   249→                rhs_cols2 -= vl;
   250→   250→
   251→   251→                NDS_VEC_VSETVLI(vl, vl, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   252→   252→                NDS_VEC_VD4DOTS_VV(NDS_VEC_V0, NDS_VEC_V16, NDS_VEC_V8);
   253→   253→            }
   254→   254→
   255→   255→            //clean one of the dot product source
   256→   256→            long tmp_avl;
   257→   257→            tmp_avl = rhs_cols2 + 3;
   258→   258→            NDS_VEC_VSETVLI(vl, tmp_avl, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   259→   259→            NDS_VEC_VAND_VI(NDS_VEC_V8, NDS_VEC_V8, 0);
   260→   260→            NDS_VEC_VSETVLI(vl, rhs_cols2, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   261→   261→            NDS_VEC_VADD_VI(NDS_VEC_V8, NDS_VEC_V8, 1);     // const one for vd4dot
   262→   262→
   263→   263→            // remain loop
   264→   264→            while (rhs_cols2 > 0)
   265→   265→            {
   266→   266→                NDS_VEC_VSETVLI(vl, rhs_cols2, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   267→   267→                NDS_VEC_VLE8_V(NDS_VEC_V16, pWt);
   268→   268→                pWt += vl;
   269→   269→                rhs_cols2 -= vl;
   270→   270→
   271→   271→                vl = (vl + 3) >> 2;
   272→   272→                NDS_VEC_VSETVLI(vl, vl, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   273→   273→                NDS_VEC_VD4DOTS_VV(NDS_VEC_V0, NDS_VEC_V16, NDS_VEC_V8);
   274→   274→            }
   275→   275→
   276→   276→            NDS_VEC_VSETVLI(vl, rhs_cols, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);   //note the avl here
   277→   277→            NDS_VEC_VREDSUM_VS(NDS_VEC_V0, NDS_VEC_V0, NDS_VEC_V31);
   278→   278→
   279→   279→            NDS_VEC_VSETVLI(vl, 1, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M1);
   280→   280→            NDS_VEC_VLE32_V(NDS_VEC_V1, pBias);
   281→   281→            NDS_VEC_VMUL_VX(NDS_VEC_V0, NDS_VEC_V0, in_offset);     // add input_offset contribution
   282→   282→            NDS_VEC_VADD_VV(NDS_VEC_V0, NDS_VEC_V0, NDS_VEC_V1);    // add bias
   283→   283→            NDS_VEC_VSE32_V(NDS_VEC_V0, pContri_buf);
   284→   284→            pBias += vl;
   285→   285→            pContri_buf += vl;
   286→   286→            out_tensor_ch2--;
   287→   287→        }
   288→   288→    }
   289→   289→    else
   290→   290→    {
   291→   291→        long out_tensor_ch2 = out_tensor_ch;
   292→   292→        long vl;
   293→   293→        NDS_VEC_VSETVLI(vl, 1, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M1);
   294→   294→        NDS_VEC_VAND_VI(NDS_VEC_V31, NDS_VEC_V31, 0);   // const zero for redsum
   295→   295→
   296→   296→        while (out_tensor_ch2 > 0)
   297→   297→        {
   298→   298→            NDS_VEC_VSETVLI(vl, rhs_cols, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   299→   299→            NDS_VEC_VAND_VI(NDS_VEC_V0, NDS_VEC_V0, 0);
   300→   300→            NDS_VEC_VSETVLI(vl, rhs_cols, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   301→   301→            NDS_VEC_VMV_V_I(NDS_VEC_V8, 1);     // const one for vd4dot
   302→   302→
   303→   303→            long rhs_cols2 = rhs_cols;
   304→   304→            while (rhs_cols2 >= maxvl_per_loop)
   305→   305→            {
   306→   306→                NDS_VEC_VSETVLI(vl, rhs_cols2, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   307→   307→                NDS_VEC_VLE8_V(NDS_VEC_V16, pWt);
   308→   308→                pWt += vl;
   309→   309→                rhs_cols2 -= vl;
   310→   310→
   311→   311→                NDS_VEC_VSETVLI(vl, vl, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   312→   312→                NDS_VEC_VD4DOTS_VV(NDS_VEC_V0, NDS_VEC_V16, NDS_VEC_V8);
   313→   313→            }
   314→   314→
   315→   315→            //clean one of the dot product source
   316→   316→            long tmp_avl;
   317→   317→            tmp_avl = rhs_cols2 + 3;
   318→   318→            NDS_VEC_VSETVLI(vl, tmp_avl, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   319→   319→            NDS_VEC_VAND_VI(NDS_VEC_V8, NDS_VEC_V8, 0);
   320→   320→            NDS_VEC_VSETVLI(vl, rhs_cols2, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   321→   321→            NDS_VEC_VADD_VI(NDS_VEC_V8, NDS_VEC_V8, 1);     // const one for vd4dot
   322→   322→
   323→   323→            // remain loop
   324→   324→            while (rhs_cols2 > 0)
   325→   325→            {
   326→   326→                NDS_VEC_VSETVLI(vl, rhs_cols2, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   327→   327→                NDS_VEC_VLE8_V(NDS_VEC_V16, pWt);
   328→   328→                pWt += vl;
   329→   329→                rhs_cols2 -= vl;
   330→   330→
   331→   331→                vl = (vl + 3) >> 2;
   332→   332→                NDS_VEC_VSETVLI(vl, vl, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   333→   333→                NDS_VEC_VD4DOTS_VV(NDS_VEC_V0, NDS_VEC_V16, NDS_VEC_V8);
   334→   334→            }
