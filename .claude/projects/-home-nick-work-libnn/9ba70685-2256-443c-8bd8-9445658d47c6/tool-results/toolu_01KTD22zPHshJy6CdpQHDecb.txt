    25→    25→//// Convolution Functions
    26→    26→int32_t riscv_nn_conv_HWC_s8_s8_s8_RGB_sym_bias_fast(const q7_t * in_tensor,
    27→    27→                                            const uint16_t in_tensor_dim,
    28→    28→                                            const q7_t * ker_weight,
    29→    29→                                            const uint16_t out_tensor_ch,
    30→    30→                                            const uint16_t ker_dim,
    31→    31→                                            const uint16_t pad,
    32→    32→                                            const uint16_t stride,
    33→    33→                                            const q31_t * bias,
    34→    34→                                            const uint16_t pre_rshift,
    35→    35→                                            const uint16_t out_scale,
    36→    36→                                            const uint16_t post_rshift,
    37→    37→                                            q7_t * out_tensor,
    38→    38→                                            const uint16_t out_tensor_dim,
    39→    39→                                            q15_t * in_tmp_buf,
    40→    40→                                            q15_t * wt_tmp_buf)
    41→    41→{
    42→    42→#ifdef ENA_EMPTY_FUN
    43→    43→    //empty body
    44→    44→#elif defined(ENA_VEC_ISA) && defined(ENA_VEC_INTRINSIC)
    45→    45→    long   i_out_y, i_out_x, i_ker_y, i_ker_x;
    46→    46→
    47→    47→    /*
    48→    48→     *  Here we use in_tmp_buf as q15_t internally as computation are done with q15_t level
    49→    49→     *  im2col are done to output in q15_t format from q7_t input
    50→    50→     */
    51→    51→    q7_t *pBuffer = (q7_t *)in_tmp_buf;
    52→    52→    q7_t *pOut = out_tensor;
    53→    53→
    54→    54→    long buf_len = (3 * ker_dim * ker_dim);
    55→    55→    long vl;
    56→    56→
    57→    57→    // This part implements the im2col function
    58→    58→    for (i_out_y = 0; i_out_y < out_tensor_dim; i_out_y++)
    59→    59→    {
    60→    60→        for (i_out_x = 0; i_out_x < out_tensor_dim; i_out_x++)
    61→    61→        {
    62→    62→            // init in_tmp_buf
    63→    63→            long len = buf_len;
    64→    64→            q7_t *pBuffer2 = pBuffer;
    65→    65→            vl = __riscv_vsetvl_e8m8(len);
    66→    66→            vint8m8_t vZero = __riscv_vand(__riscv_vundefined_i8m8(), 0x0, vl);     //const zero for redsum
    67→    67→            while(len > 0)
    68→    68→            {
    69→    69→                __riscv_vse8(pBuffer2, vZero, vl);
    70→    70→                len -= vl;
    71→    71→                pBuffer2 += vl;
    72→    72→                vl = __riscv_vsetvl_e8m8(len);
    73→    73→            }
    74→    74→
    75→    75→            // if (i_ker_y < 0)
    76→    76→            for (i_ker_y = i_out_y * stride - pad; i_ker_y < 0; i_ker_y++)
    77→    77→            {
    78→    78→                pBuffer += ker_dim * 3;
    79→    79→            }
    80→    80→
    81→    81→            long bound_y = ((i_out_y * stride - pad + ker_dim) < in_tensor_dim)? (i_out_y * stride - pad + ker_dim) : in_tensor_dim;
    82→    82→            for (; i_ker_y < bound_y; i_ker_y++)
    83→    83→            {
    84→    84→                // if (i_ker_x < 0)
    85→    85→                for (i_ker_x = i_out_x * stride - pad; i_ker_x < 0; i_ker_x++)
    86→    86→                {
    87→    87→                    pBuffer += 3;
    88→    88→                }
    89→    89→
    90→    90→                long bound_x = (i_out_x * stride - pad + ker_dim < in_tensor_dim)? (i_out_x * stride - pad + ker_dim) : in_tensor_dim;
    91→    91→                const q7_t *pPixel = in_tensor + (i_ker_y * in_tensor_dim + i_ker_x) * 3;
    92→    92→                long loop = (bound_x - i_ker_x)*3;
    93→    93→                i_ker_x = bound_x;
    94→    94→                while(loop > 0)
    95→    95→                {
    96→    96→                    vl = __riscv_vsetvl_e8m8(loop);
    97→    97→                    vint8m8_t vpPixel = __riscv_vle8_v_i8m8(pPixel, vl);
    98→    98→                    __riscv_vse8(pBuffer, vpPixel, vl);
    99→    99→                    pPixel += vl;
   100→   100→                    pBuffer += vl;
   101→   101→                    loop -= vl;
   102→   102→                }
   103→   103→
   104→   104→                // if (i_ker_x >= in_tensor_dim)
   105→   105→                for ( ; i_ker_x < i_out_x * stride - pad + ker_dim; i_ker_x++)
   106→   106→                {
   107→   107→                    pBuffer += 3;
   108→   108→                }
   109→   109→            }
   110→   110→
   111→   111→            // if (i_ker_y >= in_tensor_dim)
   112→   112→            for (; i_ker_y < i_out_y * stride - pad + ker_dim; i_ker_y++)
   113→   113→            {
   114→   114→                pBuffer += ker_dim * 3;
   115→   115→            }
   116→   116→
   117→   117→            if (pBuffer == (q7_t *)in_tmp_buf + 2 * (3 * ker_dim * ker_dim))
   118→   118→            {
   119→   119→                pOut = nn_mat_mul_kernel_q7_bias_2sft((q7_t *)ker_weight, (q7_t *)in_tmp_buf, out_tensor_ch, 3 * ker_dim * ker_dim, pre_rshift, out_scale, post_rshift, bias, pOut);
   120→   120→                pBuffer = (q7_t *)in_tmp_buf;
   121→   121→            }
   122→   122→        }
   123→   123→    }
   124→   124→
   125→   125→#if (NDS_VEC_RVV_VERSION >= 100) && defined(ENA_NDS_V5_VEC_DOT_PROD)
   126→   126→    if (pBuffer != (q7_t *)in_tmp_buf)
   127→   127→    {
   128→   128→        unsigned vl;
   129→   129→        const q7_t *p_wt = ker_weight;
   130→   130→        int       i;
   131→   131→        uint16_t  colCnt = 3 * ker_dim * ker_dim;
   132→   132→        // long vlenb = NDS_VEC_CSR_GET_VLENB();
   133→   133→
   134→   134→#if (__clang__)
   135→   135→        register const long zero asm ("x0") = 0;
   136→   136→#else
   137→   137→        register const long zero asm ("x0");
   138→   138→#endif
   139→   139→
   140→   140→        vl = __riscv_vsetvl_e32m1(1);
   141→   141→        vint32m1_t vZero = __riscv_vand(__riscv_vundefined_i32m1(), zero, vl);     //const zero for redsum
   142→   142→
   143→   143→        for (i = 0; i < out_tensor_ch; i++)
   144→   144→        {
   145→   145→            q7_t      *p_in = (q7_t*)in_tmp_buf;
   146→   146→            uint16_t colCnt2 = colCnt;
   147→   147→
   148→   148→            //clear v0~v7
   149→   149→            vl = __riscv_vsetvl_e32m8(colCnt);
   150→   150→            vint32m8_t vAccumulator = __riscv_vand(__riscv_vundefined_i32m8(), zero, vl);
   151→   151→
   152→   152→            // while (colCnt2 >= (vlenb * NDS_VEC_VTYPE_LMUL_ID_M8))
   153→   153→            // {
   154→   154→            //     NDS_VEC_VSETVLI(vl, colCnt2, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   155→   155→            //     //load src1
   156→   156→            //     NDS_VEC_VLB_V(NDS_VEC_V8, p_wt);
   157→   157→            //     //load src2
   158→   158→            //     NDS_VEC_VLB_V(NDS_VEC_V16, p_in);
   159→   159→            //     //bump pointers and update loop counter
   160→   160→            //     colCnt2 -= vl;
   161→   161→            //     p_wt += vl;
   162→   162→            //     p_in += vl;
   163→   163→            //     //set the proper avl/SEW/LMUL values for vd4dot
   164→   164→            //     vl >>= 2;
   165→   165→            //     NDS_VEC_VSETVLI(vl, vl, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   166→   166→            //     //acc += src1 * src2
   167→   167→            //     NDS_VEC_VD4DOTS_VV(NDS_VEC_V0, NDS_VEC_V8, NDS_VEC_V16);
   168→   168→            // }
   169→   169→
   170→   170→            //extra process for the case of elements to be dot producted is not a mutiple of 4
   171→   171→            while(colCnt2 > 0)
   172→   172→            {
   173→   173→                //clean one of the dot product source
   174→   174→                vl = __riscv_vsetvlmax_e8m8();
   175→   175→                vint8m8_t vP_wt_i8m8 = __riscv_vand(__riscv_vundefined_i8m8(), zero, vl);
   176→   176→
   177→   177→                vl = __riscv_vsetvl_e8m8(colCnt2);
   178→   178→                //load src1
   179→   179→                vP_wt_i8m8 = __riscv_vle8_v_i8m8_tu(vP_wt_i8m8, p_wt, vl);
   180→   180→                //load src2
   181→   181→                vint8m8_t vP_in_i8m8 = __riscv_vle8_v_i8m8(p_in, vl);
   182→   182→                //bump pointers and update loop counter
   183→   183→                colCnt2 -= vl;
   184→   184→                p_wt += vl;
   185→   185→                p_in += vl;
   186→   186→                //set the proper avl/SEW/LMUL values for vd4dot
   187→   187→                //remember to deal with the avl of vd4dot is not a multiple of 4 case
   188→   188→                vl = __riscv_vsetvl_e32m8(vl);
   189→   189→                //acc += src1 * src2
   190→   190→                vAccumulator = __riscv_vd4dots_vv_i32m8(vAccumulator, vP_wt_i8m8, vP_in_i8m8, vl);
   191→   191→            }
   192→   192→
   193→   193→            vl = __riscv_vsetvl_e32m8(colCnt);  //note the avl here
   194→   194→            vint32m1_t vSum_i32 = __riscv_vredsum(vAccumulator, vZero, vl);
   195→   195→
   196→   196→            //add bias back
   197→   197→            vl = __riscv_vsetvl_e32m1(1);
   198→   198→            vint32m1_t vBias_i32 = __riscv_vle32_v_i32m1(bias, vl);
   199→   199→            vSum_i32 = __riscv_vadd(vSum_i32, vBias_i32, vl);
   200→   200→
   201→   201→            //2-stage shift
   202→   202→            vSum_i32 = __riscv_vsra(vSum_i32, pre_rshift, vl);
   203→   203→            vSum_i32 = __riscv_vmul(vSum_i32, out_scale, vl);
   204→   204→            vSum_i32 = __riscv_vadd(vSum_i32, NN_ROUND(post_rshift), vl);
   205→   205→
   206→   206→            //scale down, clip and store the results
   207→   207→            vl = __riscv_vsetvl_e16mf2(1);
   208→   208→            vint16mf2_t vSum_i16 = __riscv_vnsra(vSum_i32, post_rshift, vl);
   209→   209→            vl = __riscv_vsetvl_e8mf4(1);
   210→   210→            vint8mf4_t vSum_i8 = __riscv_vnclip(vSum_i16, zero, __RISCV_VXRM_RNU, vl);
   211→   211→            __riscv_vse8(pOut, vSum_i8, vl);
   212→   212→            bias += vl;
   213→   213→            pOut += vl;
   214→   214→        }
   215→   215→    }
   216→   216→#elif (NDS_VEC_RVV_VERSION >= 100) && defined(ENA_VEC_ISA_ZVQMAC)
   217→   217→    /* left-over because odd number of output pixels */
   218→   218→    if (pBuffer != (q7_t *)in_tmp_buf)
   219→   219→    {
   220→   220→        const q7_t *pA = ker_weight;
   221→   221→        long i;
   222→   222→        /* basically each time it process 4 entries */
   223→   223→        long colCnt = (3 * ker_dim * ker_dim);
   224→   224→
   225→   225→        //set SEW for 1st time VMV_S_X
   226→   226→        //LMUL=8 is just for clear v8~v15 (vmv ignores LMUL)
   227→   227→        vl = __riscv_vsetvl_e32m8(colCnt);
   228→   228→
   229→   229→        for (i = 0; i < out_tensor_ch; i++)
   230→   230→        {
   231→   231→            q31_t sum = bias[i];
   232→   232→            q7_t *pB = (q7_t*)in_tmp_buf;
   233→   233→            long colCnt2 = colCnt;
   234→   234→
   235→   235→            //init
   236→   236→            vint32m1_t vBias = __riscv_vmv_s_x_i32m1(sum, vl);
   237→   237→            //clear v8~v15 (acc)
   238→   238→            vint32m8_t vAcc = __riscv_vundefined_i32m8();
   239→   239→            vAcc = __riscv_vand_tu(vAcc, vAcc, 0, vl);
   240→   240→
   241→   241→            while(colCnt2 > 0)
   242→   242→            {
   243→   243→                vl = __riscv_vsetvl_e8m2(colCnt2);
   244→   244→                vint8m2_t vPa_i8 = __riscv_vundefined_i8m2();
   245→   245→                vint8m2_t vPb_i8 = __riscv_vundefined_i8m2();
   246→   246→                vPa_i8 = __riscv_vle8_tu(vPa_i8, pA, vl);
   247→   247→                vPb_i8 = __riscv_vle8_tu(vPb_i8, pB, vl);
   248→   248→                vAcc = __riscv_vqmacc_vv_i32m8_tu(vAcc, vPa_i8, vPb_i8, vl);
   249→   249→                //bump pointers and update loop counter
   250→   250→                colCnt2 -= vl;
   251→   251→                pA += vl;
   252→   252→                pB += vl;
   253→   253→            }
   254→   254→
   255→   255→            vl = __riscv_vsetvl_e32m8(colCnt);    //note the avl here
   256→   256→            vint32m1_t vAcc_sum = __riscv_vredsum(vAcc, vBias, vl);
   257→   257→            //move vr results to gpr
   258→   258→            sum = __riscv_vmv_x(vAcc_sum);
   259→   259→
   260→   260→            sum  = (sum  >> pre_rshift) * out_scale + NN_ROUND(post_rshift);
   261→   261→            *pOut++ = NDS_ISA_SATS((sum >> post_rshift), 8);
   262→   262→        }
   263→   263→    }
   264→   264→#endif
   265→   265→#elif defined(ENA_VEC_ISA)
   266→   266→#ifdef ENA_TILING
   267→   267→#if (NDS_VEC_RVV_VERSION >= 1000)
   268→   268→    /*
   269→   269→     * Direct convolution with fused bias + requantize.
   270→   270→     * Vectorizes across out_tensor_ch: vqmacc.vx accumulates one input scalar
   271→   271→     * against the full transposed weight row (all output channels).
   272→   272→     * Processes 4 output pixels at a time, sharing weight loads.
   273→   273→     * Eliminates im2col buffer and q32 intermediate buffer.
   274→   274→     *
   275→   275→     * Register plan:
   276→   276→     *   v0-v3   (M4 e32): Accumulator pixel 1
   277→   277→     *   v4-v7   (M4 e32): Accumulator pixel 2
   278→   278→     *   v8-v11  (M4 e32): Accumulator pixel 3
   279→   279→     *   v12-v15 (M4 e32): Accumulator pixel 4
   280→   280→     *   v16     (M1 e8):  Weight vector load
   281→   281→     */
   282→   282→    const int in_tensor_ch = 3;
   283→   283→    long i_ker_y, i_ker_x, ch;
   284→   284→
   285→   285→    const long align_byte = 64;
   286→   286→    q7_t *pOut = out_tensor;
   287→   287→    long vl;
   288→   288→
   289→   289→    /* Allocate transposed weight buffer from in_tmp_buf */
   290→   290→    q7_t *src2 = (q7_t *)in_tmp_buf;
   291→   291→    src2 = (q7_t*)nn_align((unsigned long)src2, align_byte);
   292→   292→
   293→   293→    /* Transpose weights once: from [out_ch x K] to [K x out_ch] */
   294→   294→    const long row = out_tensor_ch;
   295→   295→    const long col = ker_dim * ker_dim * in_tensor_ch;
   296→   296→    nn_mat_mul_kernel_tiling_transpose_q7(ker_weight, src2, row, col);
   297→   297→
   298→   298→    /* Total output pixels */
   299→   299→    long total_pixels = (long)out_tensor_dim * out_tensor_dim;
   300→   300→    long pixel_idx = 0;
   301→   301→
   302→   302→    /* Process 4 output pixels at a time */
   303→   303→    for (; pixel_idx + 3 < total_pixels; pixel_idx += 4)
   304→   304→    {
   305→   305→        long out_y1 = pixel_idx / out_tensor_dim;
   306→   306→        long out_x1 = pixel_idx % out_tensor_dim;
   307→   307→        long out_y2 = (pixel_idx + 1) / out_tensor_dim;
   308→   308→        long out_x2 = (pixel_idx + 1) % out_tensor_dim;
   309→   309→        long out_y3 = (pixel_idx + 2) / out_tensor_dim;
   310→   310→        long out_x3 = (pixel_idx + 2) % out_tensor_dim;
   311→   311→        long out_y4 = (pixel_idx + 3) / out_tensor_dim;
   312→   312→        long out_x4 = (pixel_idx + 3) % out_tensor_dim;
   313→   313→
   314→   314→        /* Load bias into 4 M4 accumulators at e32/M4 */
   315→   315→        NDS_VEC_VSETVLI(vl, out_tensor_ch, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M4);
   316→   316→        NDS_VEC_VLE32_V(NDS_VEC_V0, bias);
   317→   317→        NDS_VEC_VLE32_V(NDS_VEC_V4, bias);
   318→   318→        NDS_VEC_VLE32_V(NDS_VEC_V8, bias);
   319→   319→        NDS_VEC_VLE32_V(NDS_VEC_V12, bias);
   320→   320→
   321→   321→        /* Set e8/M1 once for all vqmacc accumulation (e8*e8->e32, M1 input, M4 acc) */
   322→   322→        NDS_VEC_VSETVLI(vl, out_tensor_ch, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M1);
   323→   323→
   324→   324→        /* Accumulate over all K positions */
   325→   325→        if (pad == 0)
   326→   326→        {
   327→   327→            /* Fast path: no padding means all pixels are interior — skip boundary checks */
   328→   328→            for (i_ker_y = 0; i_ker_y < ker_dim; i_ker_y++)
   329→   329→            {
   330→   330→                long in_row1 = out_y1 * stride + i_ker_y;
   331→   331→                long in_row2 = out_y2 * stride + i_ker_y;
   332→   332→                long in_row3 = out_y3 * stride + i_ker_y;
   333→   333→                long in_row4 = out_y4 * stride + i_ker_y;
   334→   334→
   335→   335→                for (i_ker_x = 0; i_ker_x < ker_dim; i_ker_x++)
   336→   336→                {
   337→   337→                    long in_col1 = out_x1 * stride + i_ker_x;
   338→   338→                    long in_col2 = out_x2 * stride + i_ker_x;
   339→   339→                    long in_col3 = out_x3 * stride + i_ker_x;
   340→   340→                    long in_col4 = out_x4 * stride + i_ker_x;
   341→   341→                    long k_base = (i_ker_y * ker_dim + i_ker_x) * in_tensor_ch;
   342→   342→
   343→   343→                    for (ch = 0; ch < in_tensor_ch; ch++)
   344→   344→                    {
   345→   345→                        long k_pos = k_base + ch;
   346→   346→
   347→   347→                        /* Load transposed weight row into v16 at e8/M1 */
   348→   348→                        NDS_VEC_VLB_V(NDS_VEC_V16, src2 + k_pos * out_tensor_ch);
   349→   349→
   350→   350→                        /* Get input scalars — no boundary checks needed */
   351→   351→                        q7_t a1 = in_tensor[(in_row1 * in_tensor_dim + in_col1) * in_tensor_ch + ch];
   352→   352→                        q7_t a2 = in_tensor[(in_row2 * in_tensor_dim + in_col2) * in_tensor_ch + ch];
   353→   353→                        q7_t a3 = in_tensor[(in_row3 * in_tensor_dim + in_col3) * in_tensor_ch + ch];
   354→   354→                        q7_t a4 = in_tensor[(in_row4 * in_tensor_dim + in_col4) * in_tensor_ch + ch];
   355→   355→
   356→   356→                        /* vqmacc.vx: acc[e32/M4] += scalar * wt[e8/M1] */
   357→   357→                        NDS_VEC_VQMACC_VX(NDS_VEC_V0, a1, NDS_VEC_V16);
   358→   358→                        NDS_VEC_VQMACC_VX(NDS_VEC_V4, a2, NDS_VEC_V16);
   359→   359→                        NDS_VEC_VQMACC_VX(NDS_VEC_V8, a3, NDS_VEC_V16);
   360→   360→                        NDS_VEC_VQMACC_VX(NDS_VEC_V12, a4, NDS_VEC_V16);
   361→   361→                    }
   362→   362→                }
   363→   363→            }
   364→   364→        }
   365→   365→        else
