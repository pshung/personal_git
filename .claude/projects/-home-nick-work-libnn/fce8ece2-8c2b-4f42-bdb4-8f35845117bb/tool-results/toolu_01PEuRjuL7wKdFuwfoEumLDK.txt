    25→//// Convolution Functions
    26→int32_t riscv_nn_conv_HWC_s8_s8_s8_asym_bias_any_dilated(const int8_t * in_tensor,
    27→                                                         const uint16_t in_tensor_dim_x,
    28→                                                         const uint16_t in_tensor_dim_y,
    29→                                                         const uint16_t in_tensor_ch,
    30→                                                         const uint16_t in_tensor_batch,
    31→                                                         const int8_t * ker_weight,
    32→                                                         const uint16_t out_tensor_ch,
    33→                                                         const uint16_t ker_dim_x,
    34→                                                         const uint16_t ker_dim_y,
    35→                                                         const uint16_t ker_ch,
    36→                                                         const uint16_t pad_x,
    37→                                                         const uint16_t pad_y,
    38→                                                         const uint16_t stride_x,
    39→                                                         const uint16_t stride_y,
    40→                                                         const int32_t * bias,
    41→                                                         int8_t * out_tensor,
    42→                                                         const int32_t * out_shift,
    43→                                                         const int32_t * out_scale,
    44→                                                         const int32_t out_offset,    //value is in the range of [-128, 127]
    45→                                                         const int32_t in_offset,     //value is in the range of [-127, 128]
    46→                                                         const int32_t act_min,
    47→                                                         const int32_t act_max,
    48→                                                         const uint16_t out_tensor_dim_x,
    49→                                                         const uint16_t out_tensor_dim_y,
    50→                                                         const int32_t dilation_x,
    51→                                                         const int32_t dilation_y,
    52→                                                         int16_t * in_tmp_buf)
    53→{
    54→#ifdef ENA_EMPTY_FUN
    55→    //empty body
    56→#elif defined(ENA_VEC_ISA) && (NDS_VEC_RVV_VERSION >= 100) && defined(ENA_NDS_V5_VEC_DOT_PROD) && defined(ENA_VEC_ELEN64) && defined(ENA_VEC_INTRINSIC)
    57→    int i_batch;
    58→    for (i_batch = 0; i_batch < in_tensor_batch; i_batch++)
    59→    {
    60→        int32_t i_out_y, i_out_x, i_ker_y, i_ker_x;
    61→
    62→        /* Generate two columns from the input tensor a GEMM computation */
    63→        int8_t *two_column_buf = (int8_t *)in_tmp_buf;
    64→        int8_t *out = out_tensor;
    65→        long vl;
    66→
    67→        /* This part implements the im2col function */
    68→        for (i_out_y = 0; i_out_y < out_tensor_dim_y; i_out_y++)
    69→        {
    70→            for (i_out_x = 0; i_out_x < out_tensor_dim_x; i_out_x++)
    71→            {
    72→                for (i_ker_y = i_out_y * stride_y - pad_y; i_ker_y < i_out_y * stride_y - pad_y + (ker_dim_y - 1) * dilation_y + 1; i_ker_y += dilation_y)
    73→                {
    74→                    for (i_ker_x = i_out_x * stride_x - pad_x; i_ker_x < i_out_x * stride_x - pad_x + (ker_dim_x - 1) * dilation_x + 1; i_ker_x += dilation_x)
    75→                    {
    76→                        if (i_ker_y < 0 || i_ker_y >= in_tensor_dim_y || i_ker_x < 0 || i_ker_x >= in_tensor_dim_x)
    77→                        {
    78→                            nn_set_val_s8(two_column_buf, -in_offset, in_tensor_ch);
    79→                        }
    80→                        else
    81→                        {
    82→                            /* Copying the pixel data to column */
    83→                            nn_dup_s8(in_tensor + (i_ker_y * in_tensor_dim_x + i_ker_x) * in_tensor_ch, two_column_buf, in_tensor_ch);
    84→                        }
    85→                        two_column_buf += in_tensor_ch;
    86→                    }
    87→                }
    88→
    89→                /* Computation is filed for every 2 columns */
    90→                if (two_column_buf == (int8_t *)in_tmp_buf + 2 * in_tensor_ch * ker_dim_y * ker_dim_x)
    91→                {
    92→                    out = nn_mat_mult_kernel_s8_offset(ker_weight,
    93→                                                       (int8_t *)in_tmp_buf,
    94→                                                       out_tensor_ch,
    95→                                                       out_shift,
    96→                                                       out_scale,
    97→                                                       in_offset,
    98→                                                       out_offset,
    99→                                                       act_min,
   100→                                                       act_max,
   101→                                                       in_tensor_ch * ker_dim_y * ker_dim_x,
   102→                                                       bias,
   103→                                                       out);
   104→
   105→                    /* counter reset */
   106→                    two_column_buf = (int8_t *)in_tmp_buf;
   107→                }
   108→            }
   109→        }
   110→
   111→        /* left-over because odd number of output pixels */
   112→        if (two_column_buf != (int8_t *)in_tmp_buf)
   113→        {
   114→            const int8_t *ker_a = ker_weight;
   115→            int i;
   116→            long col_count = (in_tensor_ch * ker_dim_y * ker_dim_x);
   117→            long vlenb = NDS_VEC_CSR_GET_VLENB();
   118→
   119→            vl = __riscv_vsetvl_e8m4(col_count);
   120→            vint8m4_t vOne_i8 = __riscv_vmv_v_x_i8m4(1, vl);                   //const one for vd4dot
   121→
   122→            //set SEW for 1st time VMV_S_X
   123→            vl = __riscv_vsetvl_e32m4(col_count);
   124→
   125→            for (i = 0; i < out_tensor_ch; i++)
   126→            {
   127→                int32_t sum = 0;
   128→                if (bias)
   129→                {
   130→                    sum = bias[i];
   131→                }
   132→
   133→                /* Point to the beginning of the im2col buffer where the input is available as a rearranged column */
   134→                const int8_t *ip_as_col = (int8_t *)in_tmp_buf;
   135→                long col_count2 = col_count;
   136→                int32_t sum_ker_a = 0;
   137→
   138→                //init
   139→                vint32m1_t vSum_e0 = __riscv_vmv_s_x_i32m1(sum, 1);
   140→                vint32m1_t vSum_ker_a_e0 = __riscv_vmv_s_x_i32m1(sum_ker_a, 1);
   141→
   142→                //clear the accumulator
   143→                vint32m4_t vAcc0 = __riscv_vand(__riscv_vundefined_i32m4(), 0, vl);
   144→                vint32m4_t vAcc1 = __riscv_vand(__riscv_vundefined_i32m4(), 0, vl);
   145→
   146→                while (col_count2 >= (vlenb * NDS_VEC_VTYPE_LMUL_ID_M4))
   147→                {
   148→                    vint8m4_t vWt_i8, vIn_i8;
   149→                    vl = __riscv_vsetvl_e8m4(col_count2);
   150→                    vWt_i8 = __riscv_vle8_v_i8m4(ker_a, vl);
   151→                    vIn_i8 = __riscv_vle8_v_i8m4(ip_as_col, vl);
   152→                    col_count2 -= vl;
   153→                    ker_a += vl;
   154→                    ip_as_col += vl;
   155→
   156→                    vl = __riscv_vsetvl_e32m4(vl);
   157→                    vAcc0 = __riscv_vd4dots_vv_i32m4_tu(vAcc0, vWt_i8, vIn_i8, vl);
   158→                    vAcc1 = __riscv_vd4dots_vv_i32m4_tu(vAcc1, vWt_i8, vOne_i8, vl);
   159→                }
   160→
   161→                if(col_count2 > 0)
   162→                {
   163→                    vint8m4_t vWt_i8, vIn_i8;
   164→                    vl = __riscv_vsetvl_e8m4(col_count+3);
   165→                    vWt_i8 = __riscv_vand(__riscv_vundefined_i8m4(), 0, vl);
   166→
   167→                    vl = __riscv_vsetvl_e8m4(col_count2);
   168→                    vWt_i8 = __riscv_vle8_v_i8m4_tu(vWt_i8, ker_a, vl);
   169→                    vIn_i8 = __riscv_vle8_v_i8m4(ip_as_col, vl);
   170→                    col_count2 -= vl;
   171→                    ker_a += vl;
   172→                    ip_as_col += vl;
   173→
   174→                    vl = (vl + 3) >> 2;
   175→                    vl = __riscv_vsetvl_e32m4(vl);
   176→                    vAcc0 = __riscv_vd4dots_vv_i32m4_tu(vAcc0, vWt_i8, vIn_i8, vl);
   177→                    vAcc1 = __riscv_vd4dots_vv_i32m4_tu(vAcc1, vWt_i8, vOne_i8, vl);
   178→                }
   179→
   180→                vl = __riscv_vsetvl_e32m4(col_count);       //note the avl here
   181→                vSum_e0 = __riscv_vredsum_tu(vSum_e0, vAcc0, vSum_e0, vl);
   182→                vSum_ker_a_e0 = __riscv_vredsum_tu(vSum_ker_a_e0, vAcc1, vSum_ker_a_e0, vl);
   183→
   184→                //move vr results to gpr
   185→                sum = __riscv_vmv_x(vSum_e0);
   186→                sum_ker_a = __riscv_vmv_x(vSum_ker_a_e0);
   187→
   188→                //add offset back
   189→                sum += in_offset * sum_ker_a;
   190→
   191→                sum = nn_requantize(sum, out_scale[i], out_shift[i]);
   192→                sum += out_offset;
   193→                sum = MAX(sum, act_min);
   194→                sum = MIN(sum, act_max);
   195→                *out++ = (int8_t)sum;
   196→            }
   197→        }
   198→        in_tensor += (in_tensor_dim_x * in_tensor_dim_y * in_tensor_ch);
   199→        out_tensor += (out_tensor_dim_x * out_tensor_dim_y * out_tensor_ch);
   200→    }
   201→#elif defined(ENA_VEC_ISA) && (NDS_VEC_RVV_VERSION >= 100) && defined(ENA_NDS_V5_VEC_DOT_PROD)
   202→    const long groups = in_tensor_ch / ker_ch;
   203→    const long out_ch_per_group = out_tensor_ch / groups;
   204→
   205→    if (in_tensor_ch % groups != 0 || out_tensor_ch % groups != 0)
   206→    {
   207→        return -1;
   208→    }
   209→
   210→    //--- calculate the buffer address ---
   211→    const long align_byte = sizeof(long);
   212→    // buffer-1: use 4-column buffer when groups==1 for unroll4 path, else 2-column
   213→    const long col_batch = (groups == 1) ? 4 : 2;
   214→    int8_t *two_column_buf = (int8_t*)in_tmp_buf;
   215→    two_column_buf = (int8_t*)nn_align((unsigned long)two_column_buf, align_byte);  // align the starting addresss
   216→    // buffer-2
   217→    long two_column_buf_size = col_batch * ker_ch * ker_dim_x * ker_dim_y * sizeof(*in_tensor);
   218→    int32_t *contri_buf = (int32_t*)(two_column_buf + two_column_buf_size);
   219→    contri_buf = (int32_t*)nn_align((unsigned long)contri_buf, align_byte);         // align the starting addresss
   220→
   221→    const long rhs_cols = ker_dim_y * ker_dim_x * ker_ch;
   222→    const long rhs_cols_aligned = nn_align((long)rhs_cols, align_byte);
   223→    long maxvl_per_loop;
   224→
   225→    // calculate the weight contribution
   226→    const int8_t *pWt = ker_weight;
   227→    int32_t *pContri_buf = contri_buf;
   228→    const int32_t *pBias = bias;
   229→    NDS_VEC_VSETVLI(maxvl_per_loop, reg_x0, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   230→    if (bias != NULL)
   231→    {
   232→        long out_tensor_ch2 = out_tensor_ch;
   233→        long vl;
   234→        NDS_VEC_VSETVLI(vl, 1, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M1);
   235→        NDS_VEC_VAND_VI(NDS_VEC_V31, NDS_VEC_V31, 0);   // const zero for redsum
   236→
   237→        while (out_tensor_ch2 > 0)
   238→        {
   239→            NDS_VEC_VSETVLI(vl, rhs_cols, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   240→            NDS_VEC_VAND_VI(NDS_VEC_V0, NDS_VEC_V0, 0);
   241→            NDS_VEC_VSETVLI(vl, rhs_cols, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   242→            NDS_VEC_VMV_V_I(NDS_VEC_V8, 1);     // const one for vd4dot
   243→
   244→            long rhs_cols2 = rhs_cols;
   245→            while (rhs_cols2 >= maxvl_per_loop)
   246→            {
   247→                NDS_VEC_VSETVLI(vl, rhs_cols2, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   248→                NDS_VEC_VLE8_V(NDS_VEC_V16, pWt);
   249→                pWt += vl;
   250→                rhs_cols2 -= vl;
   251→
   252→                NDS_VEC_VSETVLI(vl, vl, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   253→                NDS_VEC_VD4DOTS_VV(NDS_VEC_V0, NDS_VEC_V16, NDS_VEC_V8);
   254→            }
   255→
   256→            //clean one of the dot product source
   257→            long tmp_avl;
   258→            tmp_avl = rhs_cols2 + 3;
   259→            NDS_VEC_VSETVLI(vl, tmp_avl, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   260→            NDS_VEC_VAND_VI(NDS_VEC_V8, NDS_VEC_V8, 0);
   261→            NDS_VEC_VSETVLI(vl, rhs_cols2, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   262→            NDS_VEC_VADD_VI(NDS_VEC_V8, NDS_VEC_V8, 1);     // const one for vd4dot
   263→
   264→            // remain loop
   265→            while (rhs_cols2 > 0)
   266→            {
   267→                NDS_VEC_VSETVLI(vl, rhs_cols2, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   268→                NDS_VEC_VLE8_V(NDS_VEC_V16, pWt);
   269→                pWt += vl;
   270→                rhs_cols2 -= vl;
   271→
   272→                vl = (vl + 3) >> 2;
   273→                NDS_VEC_VSETVLI(vl, vl, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   274→                NDS_VEC_VD4DOTS_VV(NDS_VEC_V0, NDS_VEC_V16, NDS_VEC_V8);
   275→            }
   276→
   277→            NDS_VEC_VSETVLI(vl, rhs_cols, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);   //note the avl here
   278→            NDS_VEC_VREDSUM_VS(NDS_VEC_V0, NDS_VEC_V0, NDS_VEC_V31);
   279→
   280→            NDS_VEC_VSETVLI(vl, 1, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M1);
   281→            NDS_VEC_VLE32_V(NDS_VEC_V1, pBias);
   282→            NDS_VEC_VMUL_VX(NDS_VEC_V0, NDS_VEC_V0, in_offset);     // add input_offset contribution
   283→            NDS_VEC_VADD_VV(NDS_VEC_V0, NDS_VEC_V0, NDS_VEC_V1);    // add bias
   284→            NDS_VEC_VSE32_V(NDS_VEC_V0, pContri_buf);
   285→            pBias += vl;
   286→            pContri_buf += vl;
   287→            out_tensor_ch2--;
   288→        }
   289→    }
   290→    else
   291→    {
   292→        long out_tensor_ch2 = out_tensor_ch;
   293→        long vl;
   294→        NDS_VEC_VSETVLI(vl, 1, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M1);
   295→        NDS_VEC_VAND_VI(NDS_VEC_V31, NDS_VEC_V31, 0);   // const zero for redsum
   296→
   297→        while (out_tensor_ch2 > 0)
   298→        {
   299→            NDS_VEC_VSETVLI(vl, rhs_cols, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   300→            NDS_VEC_VAND_VI(NDS_VEC_V0, NDS_VEC_V0, 0);
   301→            NDS_VEC_VSETVLI(vl, rhs_cols, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   302→            NDS_VEC_VMV_V_I(NDS_VEC_V8, 1);     // const one for vd4dot
   303→
   304→            long rhs_cols2 = rhs_cols;
   305→            while (rhs_cols2 >= maxvl_per_loop)
   306→            {
   307→                NDS_VEC_VSETVLI(vl, rhs_cols2, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   308→                NDS_VEC_VLE8_V(NDS_VEC_V16, pWt);
   309→                pWt += vl;
   310→                rhs_cols2 -= vl;
   311→
   312→                NDS_VEC_VSETVLI(vl, vl, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   313→                NDS_VEC_VD4DOTS_VV(NDS_VEC_V0, NDS_VEC_V16, NDS_VEC_V8);
   314→            }
   315→
   316→            //clean one of the dot product source
   317→            long tmp_avl;
   318→            tmp_avl = rhs_cols2 + 3;
   319→            NDS_VEC_VSETVLI(vl, tmp_avl, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   320→            NDS_VEC_VAND_VI(NDS_VEC_V8, NDS_VEC_V8, 0);
   321→            NDS_VEC_VSETVLI(vl, rhs_cols2, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   322→            NDS_VEC_VADD_VI(NDS_VEC_V8, NDS_VEC_V8, 1);     // const one for vd4dot
   323→
   324→            // remain loop
   325→            while (rhs_cols2 > 0)
   326→            {
   327→                NDS_VEC_VSETVLI(vl, rhs_cols2, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   328→                NDS_VEC_VLE8_V(NDS_VEC_V16, pWt);
   329→                pWt += vl;
   330→                rhs_cols2 -= vl;
   331→
   332→                vl = (vl + 3) >> 2;
   333→                NDS_VEC_VSETVLI(vl, vl, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   334→                NDS_VEC_VD4DOTS_VV(NDS_VEC_V0, NDS_VEC_V16, NDS_VEC_V8);
   335→            }
   336→
   337→            NDS_VEC_VSETVLI(vl, rhs_cols, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);   //note the avl here
   338→            NDS_VEC_VREDSUM_VS(NDS_VEC_V0, NDS_VEC_V0, NDS_VEC_V31);
   339→
   340→            NDS_VEC_VSETVLI(vl, 1, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M1);
   341→            NDS_VEC_VMUL_VX(NDS_VEC_V0, NDS_VEC_V0, in_offset);
   342→            NDS_VEC_VSE32_V(NDS_VEC_V0, pContri_buf);
   343→            pContri_buf++;
   344→            out_tensor_ch2--;
   345→        }
   346→    }
   347→
   348→    for (long cur_batch = 0; cur_batch < in_tensor_batch; cur_batch++)
   349→    {
   350→        int8_t *two_column_buf2 = two_column_buf;
   351→        const int8_t *pKer_wt = &ker_weight[0];
   352→        const int32_t *pOut_scale = &out_scale[0];
   353→        const int32_t *pOut_shift = &out_shift[0];
   354→        const int32_t *pContri_buf = &contri_buf[0];
   355→
   356→        if (groups == 1)
   357→        {
   358→            /* === groups==1 fast path: 4-pixel batching with _unroll4 === */
   359→            int8_t *out = out_tensor;
   360→            for (long cur_out_y = 0; cur_out_y < out_tensor_dim_y; cur_out_y++)
   361→            {
   362→                for (long cur_out_x = 0; cur_out_x < out_tensor_dim_x; cur_out_x++)
   363→                {
   364→                    for (long cur_ker_y = cur_out_y * stride_y - pad_y; cur_ker_y < cur_out_y * stride_y - pad_y + (ker_dim_y - 1) * dilation_y + 1; cur_ker_y += dilation_y)
   365→                    {
   366→                        for (long cur_ker_x = cur_out_x * stride_x - pad_x; cur_ker_x < cur_out_x * stride_x - pad_x + (ker_dim_x - 1) * dilation_x + 1; cur_ker_x += dilation_x)
   367→                        {
   368→                            if (cur_ker_y < 0 || cur_ker_y >= in_tensor_dim_y || cur_ker_x < 0 || cur_ker_x >= in_tensor_dim_x)
   369→                            {
   370→                                nn_set_val_s8(two_column_buf2, -in_offset, ker_ch);
   371→                            }
   372→                            else
   373→                            {
   374→                                nn_dup_s8(in_tensor + (cur_ker_y * in_tensor_dim_x + cur_ker_x) * in_tensor_ch,
   375→                                                two_column_buf2,
   376→                                                ker_ch * sizeof(*in_tensor));
   377→                            }
   378→                            two_column_buf2 += ker_ch;
   379→                        }
   380→                    }
   381→
   382→                    /* Computation is filed for every 4 columns */
   383→                    if (two_column_buf2 == two_column_buf + 4 * rhs_cols)

   400→                    }
   401→                }
   402→            }
   403→
   404→            /* Handle left-over pixels (1..3 remaining) */

   414→                                                          out_tensor_ch,
   415→                                                          out_tensor_ch,
   416→                                                          pOut_shift,

   420→                                                          act_min,
   421→                                                          act_max,
   422→                                                          rhs_cols,
   423→                                                          rhs_cols_aligned,
   424→                                                          pContri_buf,
   425→                                                          out);

   429→                /* Handle 1 remaining pixel with scalar path */
   430→                if (leftover_bytes > 0)
   431→                {
   432→                    uint32_t vl;
   433→                    const int8_t *pKer_wt2 = pKer_wt;
   434→                    const int32_t *pOut_scale2 = pOut_scale;
   435→                    const int32_t *pOut_shift2 = pOut_shift;

   460→                            NDS_VEC_VD4DOTS_VV(NDS_VEC_V0, NDS_VEC_V8, NDS_VEC_V16);

   462→

   505→                        pOut_scale2++;
   506→                        pOut_shift2++;
   507→                        pContri_buf2++;
   508→                    }
   509→                }
   510→                /* counter reset */
   511→                two_column_buf2 = two_column_buf;
   512→            }
   513→        }
   514→        else
   515→        {
   516→            /* === groups>1 path: 2-pixel batching with _v3 (group-aware) === */
   517→            for (long cur_group = 0; cur_group < groups; cur_group++)
   518→            {
   519→                int8_t *out = out_tensor + cur_group * out_ch_per_group;

   537→                                }
   538→                                two_column_buf2 += ker_ch;
   539→                            }
   540→                        }
   541→
   542→                        /* Computation is filed for every 2 columns */
   543→                        if (two_column_buf2 == (int8_t *)in_tmp_buf + 2 * ker_ch * ker_dim_y * ker_dim_x)

   546→                                                                  two_column_buf,

   548→                                                                  out_tensor_ch,

   550→                                                                  pOut_scale,

   553→                                                                  act_min,
   554→                                                                  act_max,
   555→                                                                  rhs_cols,
   556→                                                                  rhs_cols_aligned,
   557→                                                                  pContri_buf,
   558→                                                                  out);
   559→
   560→                            /* counter reset */
   561→                            two_column_buf2 = (int8_t *)two_column_buf;
   562→                        }
   563→                    }
   564→                }
   565→
   566→                /* left-over because odd number of output pixels */
   567→                if (two_column_buf2 != (int8_t *)two_column_buf)
   568→                {
   569→                    uint32_t vl;
   570→                    const int8_t *pKer_wt2 = pKer_wt;
   571→                    const int32_t *pOut_scale2 = pOut_scale;
   572→                    const int32_t *pOut_shift2 = pOut_shift;
   573→                    const int32_t *pContri_buf2 = pContri_buf;
   574→
   575→                    NDS_VEC_VSETVLI(vl, 1, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M1);
   576→                    NDS_VEC_VAND_VI(NDS_VEC_V31, NDS_VEC_V31, 0);   // const zero for redsum
   577→
   578→                    for (long i = 0; i < out_ch_per_group; i++)
   579→                    {
   580→                        uint32_t rhs_cols2 = rhs_cols;
   581→                        const int8_t *pIn = (int8_t *)two_column_buf;
   582→
   583→                        // init the accumulators
   584→                        NDS_VEC_VSETVLI(vl, rhs_cols2, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   585→                        NDS_VEC_VAND_VI(NDS_VEC_V0, NDS_VEC_V0, 0x0);
   586→
   587→                        while (rhs_cols2 >= maxvl_per_loop)
   588→                        {
   589→                            NDS_VEC_VSETVLI(vl, rhs_cols2, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   590→                            NDS_VEC_VLE8_V(NDS_VEC_V8, pIn);
   591→                            NDS_VEC_VLE8_V(NDS_VEC_V16, pKer_wt2);
   592→                            rhs_cols2 -= vl;
   593→                            pIn += vl;
   594→                            pKer_wt2 += vl;

   596→                            NDS_VEC_VSETVLI(vl, vl, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);

   605→                            tmp_avl = rhs_cols2 + 3;
   606→                            NDS_VEC_VSETVLI(vl, tmp_avl, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);

   608→

   611→                            NDS_VEC_VLE8_V(NDS_VEC_V16, pKer_wt2);
   612→                            rhs_cols2 -= vl;
   613→                            pIn += vl;
   614→                            pKer_wt2 += vl;
   615→
   616→                            vl = (vl + 3) >> 2;
   617→                            NDS_VEC_VSETVLI(vl, vl, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);

   621→                        NDS_VEC_VSETVLI(vl, rhs_cols, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);   //note the avl here

   640→                        out++;
   641→                        pOut_scale2++;
   642→                        pOut_shift2++;

   646→                    two_column_buf2 = (int8_t *)two_column_buf;

   649→                pOut_scale += out_ch_per_group;
   650→                pOut_shift += out_ch_per_group;
   651→                pContri_buf += out_ch_per_group;

   655→        out_tensor += (out_tensor_dim_x * out_tensor_dim_y * out_tensor_ch);
   656→    }
   657→#elif defined(ENA_VEC_ISA) && (NDS_VEC_RVV_VERSION >= 100) && defined(ENA_VEC_ISA_ZVQMAC) && defined(ENA_VEC_ELEN64) //backup obsolete algo
   658→    int i_batch;
   659→    for (i_batch = 0; i_batch < in_tensor_batch; i_batch++)
   660→    {
   661→        int32_t i_out_y, i_out_x, i_ker_y, i_ker_x;
   662→

   665→        int8_t *out = out_tensor;
   666→        long vl;
   667→
   668→        /* This part implements the im2col function */
   669→        for (i_out_y = 0; i_out_y < out_tensor_dim_y; i_out_y++)
   670→        {
   671→            for (i_out_x = 0; i_out_x < out_tensor_dim_x; i_out_x++)

   694→                                                       (int8_t *)in_tmp_buf,

   696→                                                       out_shift,
   697→                                                       out_scale,

   699→                                                       out_offset,

   709→            }
   710→        }

   712→        /* left-over because odd number of output pixels */
   713→        if (two_column_buf != (int8_t *)in_tmp_buf)
   714→        {

   778→#elif defined(ENA_DSP_ISA_64)

   786→
   787→    //--- calculate the buffer address ---
