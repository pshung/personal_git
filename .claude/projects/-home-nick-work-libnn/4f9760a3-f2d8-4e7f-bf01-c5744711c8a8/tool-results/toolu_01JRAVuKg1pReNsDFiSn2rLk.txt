    25→//// Convolution Functions
    26→int32_t riscv_nn_conv_HWC_s8_s8_s8_RGB_sym_bias_fast(const q7_t * in_tensor,
    27→                                            const uint16_t in_tensor_dim,
    28→                                            const q7_t * ker_weight,
    29→                                            const uint16_t out_tensor_ch,
    30→                                            const uint16_t ker_dim,
    31→                                            const uint16_t pad,
    32→                                            const uint16_t stride,
    33→                                            const q31_t * bias,
    34→                                            const uint16_t pre_rshift,
    35→                                            const uint16_t out_scale,
    36→                                            const uint16_t post_rshift,
    37→                                            q7_t * out_tensor,
    38→                                            const uint16_t out_tensor_dim,
    39→                                            q15_t * in_tmp_buf,
    40→                                            q15_t * wt_tmp_buf)
    41→{
    42→#ifdef ENA_EMPTY_FUN
    43→    //empty body
    44→#elif defined(ENA_VEC_ISA) && defined(ENA_VEC_INTRINSIC)
    45→    long   i_out_y, i_out_x, i_ker_y, i_ker_x;
    46→
    47→    /*
    48→     *  Here we use in_tmp_buf as q15_t internally as computation are done with q15_t level
    49→     *  im2col are done to output in q15_t format from q7_t input
    50→     */
    51→    q7_t *pBuffer = (q7_t *)in_tmp_buf;
    52→    q7_t *pOut = out_tensor;
    53→
    54→    long buf_len = (3 * ker_dim * ker_dim);
    55→    long vl;
    56→
    57→    // This part implements the im2col function
    58→    for (i_out_y = 0; i_out_y < out_tensor_dim; i_out_y++)
    59→    {
    60→        for (i_out_x = 0; i_out_x < out_tensor_dim; i_out_x++)
    61→        {
    62→            // init in_tmp_buf
    63→            long len = buf_len;
    64→            q7_t *pBuffer2 = pBuffer;
    65→            vl = __riscv_vsetvl_e8m8(len);
    66→            vint8m8_t vZero = __riscv_vand(__riscv_vundefined_i8m8(), 0x0, vl);     //const zero for redsum
    67→            while(len > 0)
    68→            {
    69→                __riscv_vse8(pBuffer2, vZero, vl);
    70→                len -= vl;
    71→                pBuffer2 += vl;
    72→                vl = __riscv_vsetvl_e8m8(len);
    73→            }
    74→
    75→            // if (i_ker_y < 0)
    76→            for (i_ker_y = i_out_y * stride - pad; i_ker_y < 0; i_ker_y++)
    77→            {
    78→                pBuffer += ker_dim * 3;
    79→            }
    80→
    81→            long bound_y = ((i_out_y * stride - pad + ker_dim) < in_tensor_dim)? (i_out_y * stride - pad + ker_dim) : in_tensor_dim;
    82→            for (; i_ker_y < bound_y; i_ker_y++)
    83→            {
    84→                // if (i_ker_x < 0)
    85→                for (i_ker_x = i_out_x * stride - pad; i_ker_x < 0; i_ker_x++)
    86→                {
    87→                    pBuffer += 3;
    88→                }
    89→
    90→                long bound_x = (i_out_x * stride - pad + ker_dim < in_tensor_dim)? (i_out_x * stride - pad + ker_dim) : in_tensor_dim;
    91→                const q7_t *pPixel = in_tensor + (i_ker_y * in_tensor_dim + i_ker_x) * 3;
    92→                long loop = (bound_x - i_ker_x)*3;
    93→                i_ker_x = bound_x;
    94→                while(loop > 0)
    95→                {
    96→                    vl = __riscv_vsetvl_e8m8(loop);
    97→                    vint8m8_t vpPixel = __riscv_vle8_v_i8m8(pPixel, vl);
    98→                    __riscv_vse8(pBuffer, vpPixel, vl);
    99→                    pPixel += vl;
   100→                    pBuffer += vl;
   101→                    loop -= vl;
   102→                }
   103→
   104→                // if (i_ker_x >= in_tensor_dim)
   105→                for ( ; i_ker_x < i_out_x * stride - pad + ker_dim; i_ker_x++)
   106→                {
   107→                    pBuffer += 3;
   108→                }
   109→            }
   110→
   111→            // if (i_ker_y >= in_tensor_dim)
   112→            for (; i_ker_y < i_out_y * stride - pad + ker_dim; i_ker_y++)
   113→            {
   114→                pBuffer += ker_dim * 3;
   115→            }
   116→
   117→            if (pBuffer == (q7_t *)in_tmp_buf + 2 * (3 * ker_dim * ker_dim))
   118→            {
   119→                pOut = nn_mat_mul_kernel_q7_bias_2sft((q7_t *)ker_weight, (q7_t *)in_tmp_buf, out_tensor_ch, 3 * ker_dim * ker_dim, pre_rshift, out_scale, post_rshift, bias, pOut);
   120→                pBuffer = (q7_t *)in_tmp_buf;
   121→            }
   122→        }
   123→    }
   124→
   125→#if (NDS_VEC_RVV_VERSION >= 100) && defined(ENA_NDS_V5_VEC_DOT_PROD)
   126→    if (pBuffer != (q7_t *)in_tmp_buf)
   127→    {
   128→        unsigned vl;
   129→        const q7_t *p_wt = ker_weight;
   130→        int       i;
   131→        uint16_t  colCnt = 3 * ker_dim * ker_dim;
   132→        // long vlenb = NDS_VEC_CSR_GET_VLENB();
   133→
   134→#if (__clang__)
   135→        register const long zero asm ("x0") = 0;
   136→#else
   137→        register const long zero asm ("x0");
   138→#endif
   139→
   140→        vl = __riscv_vsetvl_e32m1(1);
   141→        vint32m1_t vZero = __riscv_vand(__riscv_vundefined_i32m1(), zero, vl);     //const zero for redsum
   142→
   143→        for (i = 0; i < out_tensor_ch; i++)
   144→        {
   145→            q7_t      *p_in = (q7_t*)in_tmp_buf;
   146→            uint16_t colCnt2 = colCnt;
   147→
   148→            //clear v0~v7
   149→            vl = __riscv_vsetvl_e32m8(colCnt);
   150→            vint32m8_t vAccumulator = __riscv_vand(__riscv_vundefined_i32m8(), zero, vl);
   151→
   152→            // while (colCnt2 >= (vlenb * NDS_VEC_VTYPE_LMUL_ID_M8))
   153→            // {
   154→            //     NDS_VEC_VSETVLI(vl, colCnt2, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M8);
   155→            //     //load src1
   156→            //     NDS_VEC_VLB_V(NDS_VEC_V8, p_wt);
   157→            //     //load src2
   158→            //     NDS_VEC_VLB_V(NDS_VEC_V16, p_in);
   159→            //     //bump pointers and update loop counter
   160→            //     colCnt2 -= vl;
   161→            //     p_wt += vl;
   162→            //     p_in += vl;
   163→            //     //set the proper avl/SEW/LMUL values for vd4dot
   164→            //     vl >>= 2;
   165→            //     NDS_VEC_VSETVLI(vl, vl, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   166→            //     //acc += src1 * src2
   167→            //     NDS_VEC_VD4DOTS_VV(NDS_VEC_V0, NDS_VEC_V8, NDS_VEC_V16);
   168→            // }
   169→
   170→            //extra process for the case of elements to be dot producted is not a mutiple of 4
   171→            while(colCnt2 > 0)
   172→            {
   173→                //clean one of the dot product source
   174→                vl = __riscv_vsetvlmax_e8m8();
   175→                vint8m8_t vP_wt_i8m8 = __riscv_vand(__riscv_vundefined_i8m8(), zero, vl);
   176→
   177→                vl = __riscv_vsetvl_e8m8(colCnt2);
   178→                //load src1
   179→                vP_wt_i8m8 = __riscv_vle8_v_i8m8_tu(vP_wt_i8m8, p_wt, vl);
   180→                //load src2
   181→                vint8m8_t vP_in_i8m8 = __riscv_vle8_v_i8m8(p_in, vl);
   182→                //bump pointers and update loop counter
   183→                colCnt2 -= vl;
   184→                p_wt += vl;
   185→                p_in += vl;
   186→                //set the proper avl/SEW/LMUL values for vd4dot
   187→                //remember to deal with the avl of vd4dot is not a multiple of 4 case
   188→                vl = __riscv_vsetvl_e32m8(vl);
   189→                //acc += src1 * src2
   190→                vAccumulator = __riscv_vd4dots_vv_i32m8(vAccumulator, vP_wt_i8m8, vP_in_i8m8, vl);
   191→            }
   192→
   193→            vl = __riscv_vsetvl_e32m8(colCnt);  //note the avl here
   194→            vint32m1_t vSum_i32 = __riscv_vredsum(vAccumulator, vZero, vl);
   195→
   196→            //add bias back
   197→            vl = __riscv_vsetvl_e32m1(1);
   198→            vint32m1_t vBias_i32 = __riscv_vle32_v_i32m1(bias, vl);
   199→            vSum_i32 = __riscv_vadd(vSum_i32, vBias_i32, vl);
   200→
   201→            //2-stage shift
   202→            vSum_i32 = __riscv_vsra(vSum_i32, pre_rshift, vl);
   203→            vSum_i32 = __riscv_vmul(vSum_i32, out_scale, vl);
   204→            vSum_i32 = __riscv_vadd(vSum_i32, NN_ROUND(post_rshift), vl);
   205→
   206→            //scale down, clip and store the results
   207→            vl = __riscv_vsetvl_e16mf2(1);
   208→            vint16mf2_t vSum_i16 = __riscv_vnsra(vSum_i32, post_rshift, vl);
   209→            vl = __riscv_vsetvl_e8mf4(1);
   210→            vint8mf4_t vSum_i8 = __riscv_vnclip(vSum_i16, zero, __RISCV_VXRM_RNU, vl);
   211→            __riscv_vse8(pOut, vSum_i8, vl);
   212→            bias += vl;
   213→            pOut += vl;
   214→        }
   215→    }
   216→#elif (NDS_VEC_RVV_VERSION >= 100) && defined(ENA_VEC_ISA_ZVQMAC)
   217→    /* left-over because odd number of output pixels */
   218→    if (pBuffer != (q7_t *)in_tmp_buf)
   219→    {
   220→        const q7_t *pA = ker_weight;
   221→        long i;
   222→        /* basically each time it process 4 entries */
   223→        long colCnt = (3 * ker_dim * ker_dim);
   224→
   225→        //set SEW for 1st time VMV_S_X
   226→        //LMUL=8 is just for clear v8~v15 (vmv ignores LMUL)
   227→        vl = __riscv_vsetvl_e32m8(colCnt);
   228→
   229→        for (i = 0; i < out_tensor_ch; i++)
   230→        {
   231→            q31_t sum = bias[i];
   232→            q7_t *pB = (q7_t*)in_tmp_buf;
   233→            long colCnt2 = colCnt;
   234→
   235→            //init
   236→            vint32m1_t vBias = __riscv_vmv_s_x_i32m1(sum, vl);
   237→            //clear v8~v15 (acc)
   238→            vint32m8_t vAcc = __riscv_vundefined_i32m8();
   239→            vAcc = __riscv_vand_tu(vAcc, vAcc, 0, vl);
   240→
   241→            while(colCnt2 > 0)
   242→            {
   243→                vl = __riscv_vsetvl_e8m2(colCnt2);
   244→                vint8m2_t vPa_i8 = __riscv_vundefined_i8m2();
   245→                vint8m2_t vPb_i8 = __riscv_vundefined_i8m2();
   246→                vPa_i8 = __riscv_vle8_tu(vPa_i8, pA, vl);
   247→                vPb_i8 = __riscv_vle8_tu(vPb_i8, pB, vl);
   248→                vAcc = __riscv_vqmacc_vv_i32m8_tu(vAcc, vPa_i8, vPb_i8, vl);
   249→                //bump pointers and update loop counter
   250→                colCnt2 -= vl;
   251→                pA += vl;
   252→                pB += vl;
   253→            }
   254→
   255→            vl = __riscv_vsetvl_e32m8(colCnt);    //note the avl here
   256→            vint32m1_t vAcc_sum = __riscv_vredsum(vAcc, vBias, vl);
   257→            //move vr results to gpr
   258→            sum = __riscv_vmv_x(vAcc_sum);
   259→
   260→            sum  = (sum  >> pre_rshift) * out_scale + NN_ROUND(post_rshift);
   261→            *pOut++ = NDS_ISA_SATS((sum >> post_rshift), 8);
   262→        }
   263→    }
   264→#endif
   265→#elif defined(ENA_VEC_ISA)
   266→#ifdef ENA_TILING
   267→#if (NDS_VEC_RVV_VERSION >= 1000)
   268→    /*
   269→     * Direct convolution with fused bias + requantize.
   270→     * Vectorizes across out_tensor_ch: vqmacc.vx accumulates one input scalar
   271→     * against the full transposed weight row (all output channels).
   272→     * Processes 2 output pixels at a time, sharing weight loads.
   273→     * Eliminates im2col buffer and q32 intermediate buffer.
   274→     *
   275→     * Register plan:
   276→     *   v0-v7  (M8): Accumulator pixel 1 (e32)
   277→     *   v8-v15 (M8): Accumulator pixel 2 (e32)
   278→     *   v16-v17 (M2): Weight vector load (e8)
   279→     */
   280→    const int in_tensor_ch = 3;
   281→    long i_ker_y, i_ker_x, ch;
   282→
   283→    const long align_byte = 64;
   284→    q7_t *pOut = out_tensor;
   285→    long vl;
   286→
   287→    /* Allocate transposed weight buffer from in_tmp_buf */
   288→    q7_t *src2 = (q7_t *)in_tmp_buf;
   289→    src2 = (q7_t*)nn_align((unsigned long)src2, align_byte);
   290→
   291→    /* Transpose weights once: from [out_ch x K] to [K x out_ch] */
   292→    const long row = out_tensor_ch;
   293→    const long col = ker_dim * ker_dim * in_tensor_ch;
   294→    nn_mat_mul_kernel_tiling_transpose_q7(ker_weight, src2, row, col);
   295→
   296→    /* Total output pixels */
   297→    long total_pixels = (long)out_tensor_dim * out_tensor_dim;
   298→    long pixel_idx = 0;
   299→
   300→    /* Process 2 output pixels at a time */
   301→    for (; pixel_idx + 1 < total_pixels; pixel_idx += 2)
   302→    {
   303→        long out_y1 = pixel_idx / out_tensor_dim;
   304→        long out_x1 = pixel_idx % out_tensor_dim;
   305→        long out_y2 = (pixel_idx + 1) / out_tensor_dim;
   306→        long out_x2 = (pixel_idx + 1) % out_tensor_dim;
   307→
   308→        /* Load bias into accumulators: v0 (pixel1), v8 (pixel2) at e32/M8 */
   309→        NDS_VEC_VSETVLI(vl, out_tensor_ch, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   310→        NDS_VEC_VLE32_V(NDS_VEC_V0, bias);
   311→        NDS_VEC_VLE32_V(NDS_VEC_V8, bias);
   312→
   313→        /* Set e8/M2 once for all vqmacc accumulation (e8*e8->e32, M2 input, M8 acc) */
   314→        NDS_VEC_VSETVLI(vl, out_tensor_ch, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M2);
   315→
   316→        /* Accumulate over all K positions */
   317→        if (pad == 0)
   318→        {
   319→            /* Fast path: no padding means all pixels are interior — skip boundary checks */
   320→            for (i_ker_y = 0; i_ker_y < ker_dim; i_ker_y++)
   321→            {
   322→                long in_row1 = out_y1 * stride + i_ker_y;
   323→                long in_row2 = out_y2 * stride + i_ker_y;
   324→
   325→                for (i_ker_x = 0; i_ker_x < ker_dim; i_ker_x++)
   326→                {
   327→                    long in_col1 = out_x1 * stride + i_ker_x;
   328→                    long in_col2 = out_x2 * stride + i_ker_x;
   329→                    long k_base = (i_ker_y * ker_dim + i_ker_x) * in_tensor_ch;
   330→
   331→                    for (ch = 0; ch < in_tensor_ch; ch++)
   332→                    {
   333→                        long k_pos = k_base + ch;
   334→
   335→                        /* Load transposed weight row into v16 at e8/M2 */
   336→                        NDS_VEC_VLB_V(NDS_VEC_V16, src2 + k_pos * out_tensor_ch);
   337→
   338→                        /* Get input scalars — no boundary checks needed */
   339→                        q7_t a1 = in_tensor[(in_row1 * in_tensor_dim + in_col1) * in_tensor_ch + ch];
   340→                        q7_t a2 = in_tensor[(in_row2 * in_tensor_dim + in_col2) * in_tensor_ch + ch];
   341→
   342→                        /* vqmacc.vx: acc[e32/M8] += scalar * wt[e8/M2] */
   343→                        NDS_VEC_VQMACC_VX(NDS_VEC_V0, a1, NDS_VEC_V16);
   344→                        NDS_VEC_VQMACC_VX(NDS_VEC_V8, a2, NDS_VEC_V16);
   345→                    }
   346→                }
   347→            }
   348→        }
   349→        else
   350→        {
   351→            /* General path: check boundaries for padding */
   352→            for (i_ker_y = 0; i_ker_y < ker_dim; i_ker_y++)
   353→            {
   354→                long in_row1 = out_y1 * stride - pad + i_ker_y;
   355→                long in_row2 = out_y2 * stride - pad + i_ker_y;
   356→
   357→                for (i_ker_x = 0; i_ker_x < ker_dim; i_ker_x++)
   358→                {
   359→                    long in_col1 = out_x1 * stride - pad + i_ker_x;
   360→                    long in_col2 = out_x2 * stride - pad + i_ker_x;
   361→                    long k_base = (i_ker_y * ker_dim + i_ker_x) * in_tensor_ch;
   362→
   363→                    for (ch = 0; ch < in_tensor_ch; ch++)
   364→                    {
   365→                        long k_pos = k_base + ch;
   366→
   367→                        /* Load transposed weight row into v16 at e8/M2 */
   368→                        NDS_VEC_VLB_V(NDS_VEC_V16, src2 + k_pos * out_tensor_ch);
   369→
   370→                        /* Get input scalars */
   371→                        q7_t a1 = 0, a2 = 0;
   372→                        if (in_row1 >= 0 && in_row1 < in_tensor_dim &&
   373→                            in_col1 >= 0 && in_col1 < in_tensor_dim)
   374→                            a1 = in_tensor[(in_row1 * in_tensor_dim + in_col1) * in_tensor_ch + ch];
   375→                        if (in_row2 >= 0 && in_row2 < in_tensor_dim &&
   376→                            in_col2 >= 0 && in_col2 < in_tensor_dim)
   377→                            a2 = in_tensor[(in_row2 * in_tensor_dim + in_col2) * in_tensor_ch + ch];
   378→
   379→                        /* vqmacc.vx: acc[e32/M8] += scalar * wt[e8/M2] */
   380→                        NDS_VEC_VQMACC_VX(NDS_VEC_V0, a1, NDS_VEC_V16);
   381→                        NDS_VEC_VQMACC_VX(NDS_VEC_V8, a2, NDS_VEC_V16);
   382→                    }
   383→                }
   384→            }
   385→        }
   386→
   387→        /* Requantize pixel 1: shift, scale, narrow, clip, store */
   388→        NDS_VEC_VSETVLI(vl, out_tensor_ch, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   389→        NDS_VEC_VSRA_VX(NDS_VEC_V0, NDS_VEC_V0, pre_rshift);
   390→        NDS_VEC_VMUL_VX(NDS_VEC_V0, NDS_VEC_V0, out_scale);
   391→        NDS_VEC_VSETVLI(vl, out_tensor_ch, NDS_VEC_VTYPE_SEW_E16, NDS_VEC_VTYPE_LMUL_M4);
   392→        NDS_VEC_VNCLIP_WX(NDS_VEC_V0, NDS_VEC_V0, post_rshift);
   393→        NDS_VEC_VSETVLI(vl, out_tensor_ch, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M2);
   394→        NDS_VEC_VNCLIP_WX(NDS_VEC_V0, NDS_VEC_V0, 0);
   395→        NDS_VEC_VSB_V(NDS_VEC_V0, pOut);
   396→        pOut += out_tensor_ch;
   397→
   398→        /* Requantize pixel 2 */
   399→        NDS_VEC_VSETVLI(vl, out_tensor_ch, NDS_VEC_VTYPE_SEW_E32, NDS_VEC_VTYPE_LMUL_M8);
   400→        NDS_VEC_VSRA_VX(NDS_VEC_V8, NDS_VEC_V8, pre_rshift);
   401→        NDS_VEC_VMUL_VX(NDS_VEC_V8, NDS_VEC_V8, out_scale);
   402→        NDS_VEC_VSETVLI(vl, out_tensor_ch, NDS_VEC_VTYPE_SEW_E16, NDS_VEC_VTYPE_LMUL_M4);
   403→        NDS_VEC_VNCLIP_WX(NDS_VEC_V8, NDS_VEC_V8, post_rshift);
   404→        NDS_VEC_VSETVLI(vl, out_tensor_ch, NDS_VEC_VTYPE_SEW_E8, NDS_VEC_VTYPE_LMUL_M2);
   405→        NDS_VEC_VNCLIP_WX(NDS_VEC_V8, NDS_VEC_V8, 0);
   406→        NDS_VEC_VSB_V(NDS_VEC_V8, pOut);
   407→        pOut += out_tensor_ch;
   408→    }
   409→
   410→    /* Handle the last pixel if total_pixels is odd */

   812→        uint16_t colCnt = (3 * ker_dim * ker_dim);
   813→
   814→        //set SEW for 1st time VMV_S_X
   815→        //LMUL=2 is just for clear v2~v3 (vmv ignores LMUL)

   817→
   818→        for (i = 0; i < out_tensor_ch; i++)
   819→        {
   820→            q31_t sum = bias[i];
   821→            q7_t *pB = (q7_t*)in_tmp_buf;
   822→            uint16_t colCnt2 = colCnt;
   823→
   824→            //init
   825→            NDS_VEC_VMV_S_X(NDS_VEC_V4, sum);
   826→            //clear v2~v3 (acc)
   827→            NDS_VEC_VAND_VI(NDS_VEC_V2, NDS_VEC_V2, 0x0);
   828→
   829→            while(colCnt2 > 0)
   830→            {
   831→                NDS_VEC_VSETVLI_E16(vl, colCnt2);
   832→                NDS_VEC_VLB_V(NDS_VEC_V0, pA);
   833→                NDS_VEC_VLB_V(NDS_VEC_V1, pB);
   834→                NDS_VEC_VWMACC_VV(NDS_VEC_V2, NDS_VEC_V1, NDS_VEC_V0);
   835→                //bump pointers and update loop counter
   836→                colCnt2 -= vl;
   837→                pA += vl;
   838→                pB += vl;
   839→            }
   840→
   841→            NDS_VEC_VSETVLI_E32_M2(vl, colCnt);
   842→            NDS_VEC_VREDSUM_VS(NDS_VEC_V4, NDS_VEC_V2, NDS_VEC_V4);
   843→            //move vr results to gpr
   844→            NDS_VEC_VMV_X_S(sum, NDS_VEC_V4);
   845→
   846→            sum  = (sum  >> pre_rshift) * out_scale + NN_ROUND(post_rshift);
   847→            *pOut++ = (q7_t) NDS_ISA_SATS((sum >> post_rshift), 8);
   848→        }
   849→    }
   850→#endif
   851→#endif
   852→#elif defined(ENA_DSP_ISA_V5)
   853→    int16_t   i_out_y, i_out_x, i_ker_y, i_ker_x;
   854→
   855→    /*
   856→     *  Here we use in_tmp_buf as q15_t internally as computation are done with q15_t level
   857→     *  im2col are done to output in q15_t format from q7_t input
   858→     */
   859→    q7_t *wt_tmp = (q7_t*)ker_weight;
   860→
   861→    q7_t    *pBuffer = (q7_t *)in_tmp_buf;
   862→    q7_t     *pOut = out_tensor;
   863→
   864→    uint32_t i;
   865→    q7_t    *pBufferB = (q7_t *)wt_tmp_buf;
   866→
   867→    uint16_t totalsize = 3 * ker_dim * ker_dim;
   868→    for (i = 0; i < out_tensor_ch; i++)
   869→    {
   870→        uint32_t j;
   871→        for (j = 0; j < totalsize; j++)
   872→        {
   873→            *pBufferB++ = *wt_tmp++;
   874→        }
   875→        *pBufferB++ = 0;
   876→    }
   877→
   878→    // This part implements the im2col function
   879→    for (i_out_y = 0; i_out_y < out_tensor_dim; i_out_y++)
   880→    {
   881→        for (i_out_x = 0; i_out_x < out_tensor_dim; i_out_x++)
   882→        {
   883→            for (i_ker_y = i_out_y * stride - pad; i_ker_y < i_out_y * stride - pad + ker_dim; i_ker_y++)
   884→            {
   885→                for (i_ker_x = i_out_x * stride - pad; i_ker_x < i_out_x * stride - pad + ker_dim; i_ker_x++)
   886→                {
   887→                    if (i_ker_y < 0 || i_ker_y >= in_tensor_dim || i_ker_x < 0 || i_ker_x >= in_tensor_dim)
   888→                    {
   889→                        *pBuffer++ = 0;
   890→                        *pBuffer++ = 0;
   891→                        *pBuffer++ = 0;
   892→                    }
   893→                    else
   894→                    {
   895→                        const q7_t *pPixel = in_tensor + (i_ker_y * in_tensor_dim + i_ker_x) * 3;
   896→                        *pBuffer++ = *pPixel++;
   897→                        *pBuffer++ = *pPixel++;
   898→                        *pBuffer++ = *pPixel++;
   899→                    }
   900→                }
   901→            }
   902→            *pBuffer++ = 0;
   903→
   904→            if (pBuffer == (q7_t *)in_tmp_buf + 2 * (3 * ker_dim * ker_dim + 1))
   905→            {
   906→                pOut = nn_mat_mul_kernel_q7_bias_2sft((q7_t *)wt_tmp_buf, (q7_t *)in_tmp_buf, out_tensor_ch, 3 * ker_dim * ker_dim + 1, pre_rshift, out_scale, post_rshift, bias, pOut);
   907→                pBuffer = (q7_t *)in_tmp_buf;
   908→            }
   909→        }
   910→    }
   911→
   912→    /* left-over because odd number of output pixels */
   913→#ifdef ENA_DSP_ISA_64
   914→    if (pBuffer != (q7_t *)in_tmp_buf)
   915→    {
   916→        const q7_t *pA = ker_weight;
   917→        int i;
   918→
   919→        for (i = 0; i < out_tensor_ch; i++)
   920→        {
   921→            long sum = 0;
   922→            q7_t *pB = (q7_t*)in_tmp_buf;
   923→
   924→            /* basically each time it process 8 entries */
   925→            uint16_t  colCnt = 3 * ker_dim * ker_dim >> 3;
   926→

   928→            {
   929→                long inA1, inB1;
   930→                inA1 = *__SIMD64(pA)++;
   931→                inB1 = *__SIMD64(pB)++;
   932→                sum = NDS_DSP_SMAQA(sum, inA1, inB1);
   933→                colCnt--;
   934→            }

   936→            sum = (q31_t)(sum >> 32) + (q31_t)sum;
   937→

   939→            while (colCnt)
   940→            {
   941→                q7_t inA1 = *pA++;
   942→                q7_t inB1 = *pB++;
   943→                sum += inA1 * inB1;
   944→                colCnt--;
   945→            }
   946→
   947→            //add bias back
   948→            sum += *bias++;
   949→
   950→            sum  = (sum  >> pre_rshift) * out_scale + NN_ROUND(post_rshift);
   951→            *pOut++ = NDS_DSP_SATS((sum >> post_rshift), 8);
   952→        }
   953→    }
   954→#else

   956→    {
   957→        const q7_t *pA = ker_weight;

   959→
   960→        for (i = 0; i < out_tensor_ch; i++)
   961→        {
   962→            long sum = *bias++;
   963→            q7_t *pB = (q7_t*)in_tmp_buf;
   964→
   965→            uint16_t  colCnt = 3 * ker_dim * ker_dim >> 2;
   966→            while (colCnt)
   967→            {
   968→                long inA1, inB1;
   969→                inA1 = *__SIMD32(pA)++;
   970→                inB1 = *__SIMD32(pB)++;
   971→                sum = NDS_DSP_SMAQA(sum, inA1, inB1);
   972→                colCnt--;
   973→            }
   974→
   975→            colCnt = 3 * ker_dim * ker_dim & 0x3;
   976→            while (colCnt)
   977→            {
   978→                q7_t inA1 = *pA++;
   979→                q7_t inB1 = *pB++;
   980→                sum += inA1 * inB1;
   981→                colCnt--;
   982→            }
   983→
   984→            sum  = (sum  >> pre_rshift) * out_scale + NN_ROUND(post_rshift);
   985→            *pOut++ = NDS_DSP_SATS((sum >> post_rshift), 8);
   986→        }
   987→    }
   988→#endif
   989→#elif defined(ENA_DSP_ISA)
   990→    int16_t   i_out_y, i_out_x, i_ker_y, i_ker_x;
   991→
   992→    /*
   993→     *  Here we use in_tmp_buf as q15_t internally as computation are done with q15_t level
   994→     *  im2col are done to output in q15_t format from q7_t input
   995→     */
   996→    q7_t *wt_tmp = (q7_t*)ker_weight;
   997→    q15_t *pBuffer = in_tmp_buf;
   998→    q7_t *pOut = out_tensor;
   999→
  1000→    uint32_t i;
  1001→    uint16_t size;
  1002→    uint16_t size2 = (3 * ker_dim * ker_dim) >> 2;
  1003→    q31_t tmp;
  1004→    q31_t *pwt2 = (q31_t *)wt_tmp_buf;
  1005→    for (i = 0; i < out_tensor_ch; i++)
  1006→    {
  1007→        size = size2;
  1008→        while (size != 0)
  1009→        {
  1010→            tmp = *__SIMD32(wt_tmp)++;;
  1011→            *pwt2++ = NDS_DSP_SUNPKD810(tmp);
  1012→            *pwt2++ = NDS_DSP_SUNPKD810(tmp >> 16);
  1013→            size--;
  1014→        }
  1015→        q7_t tmp1 = *wt_tmp++;
  1016→        q7_t tmp2 = *wt_tmp++;
  1017→        *pwt2++ = NDS_DSP_PKBB16(tmp2, tmp1);
  1018→        tmp1 = *wt_tmp++;
  1019→        *pwt2++ = (u16_t)tmp1;
  1020→    }
  1021→
  1022→    // This part implements the im2col function
  1023→    for (i_out_y = 0; i_out_y < out_tensor_dim; i_out_y++)
  1024→    {
  1025→        for (i_out_x = 0; i_out_x < out_tensor_dim; i_out_x++)
  1026→        {
  1027→            for (i_ker_y = i_out_y * stride - pad; i_ker_y < i_out_y * stride - pad + ker_dim; i_ker_y++)
  1028→            {
  1029→                for (i_ker_x = i_out_x * stride - pad; i_ker_x < i_out_x * stride - pad + ker_dim; i_ker_x++)
  1030→                {
  1031→                    if (i_ker_y < 0 || i_ker_y >= in_tensor_dim || i_ker_x < 0 || i_ker_x >= in_tensor_dim)
  1032→                    {
  1033→                        *pBuffer++ = 0;

  1051→                pBuffer = in_tmp_buf;

  1056→    if (pBuffer != in_tmp_buf)
  1057→    {

  1060→

  1062→        {

  1065→

  1068→
  1069→            while (colCnt)
  1070→            {
  1071→                q31_t     inA1, inA2;
  1072→                q31_t     inB1, inB2;

  1094→    }
  1095→#else
