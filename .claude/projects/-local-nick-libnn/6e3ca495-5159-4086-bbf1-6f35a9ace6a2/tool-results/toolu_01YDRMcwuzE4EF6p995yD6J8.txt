     8тЖТ     8тЖТ#   $1: Source code path (without / tailed)

    11тЖТ    11тЖТ#   $2: Compiler name (with prefix)
    12тЖТ    12тЖТ#       Set it to the compiler name without full path, e.g., nds32le-elf-gcc,
    13тЖТ    13тЖТ#       riscv32-elf-clang, riscv64-linux-gcc, etc. This script will try to find
    14тЖТ    14тЖТ#       the full path of this specified compiler from the PATH environment
    15тЖТ    15тЖТ#       variable; therefore, please export the proper toolcahin path into PATH
    16тЖТ    16тЖТ#       environment variable before using this script.
    17тЖТ    17тЖТ#   $3: Simulator name
    18тЖТ    18тЖТ#       In DAILY test, please give the full path of simulator.
    19тЖТ    19тЖТ#   $4: Simulation target
    20тЖТ    20тЖТ#       It could be GDB, SID or BOARD.
    21тЖТ    21тЖТ#   $5: BS3 / DAILY
    22тЖТ    22тЖТ#       BS3:   Verify the accuracy with the built-in multi-lib. (note the sat
    23тЖТ    23тЖТ#              extension and code coverage won't be verified here)
    24тЖТ    24тЖТ#       DAILY: Use the user specified compiler and compilation options to build
    25тЖТ    25тЖТ#              the library, and this built library will be used in the accuracy
    26тЖТ    26тЖТ#              tests later. Not only the accuracy but also the code coverage
    27тЖТ    27тЖТ#              will be verified in this mode.
    28тЖТ    28тЖТ#   $6: Extra flags for selecting multi-lib or the algorithm to be built as library
    29тЖТ    29тЖТ#       As the 5th argument is set to BS3, these flags are used to select the
    30тЖТ    30тЖТ#       built-in multi-lib. For example, if "-mext-dsp" is contained in this
    31тЖТ    31тЖТ#       argument, the DSP version library will be used for linking. By default,
    32тЖТ    32тЖТ#       the plain C version library will be used for linking.
    33тЖТ    33тЖТ#       As the 5th argument is set to DAILY, these flags will be used to select
    34тЖТ    34тЖТ#       the proper algorithm and build them into library. For example, if
    35тЖТ    35тЖТ#       "-mext-dsp" is contained in this argument, the DSP algorithm will be
    36тЖТ    36тЖТ#       used as building the library. By default, the plain C algorithm will be
    37тЖТ    37тЖТ#       used to build the library.
    38тЖТ    38тЖТ#   $7: Specified linker
    39тЖТ    39тЖТ#       Use the specified linker instead of the default one when building the
    40тЖТ    40тЖТ#       executable. If this argument is left to be empty, the default linker
    41тЖТ    41тЖТ#       will be used. (e.g., ld, lld, etc.)
    42тЖТ    42тЖТ#
    43тЖТ    43тЖТ# Example:
    44тЖТ    44тЖТ#   mkdir build_dir && cd build_dir
    45тЖТ    45тЖТ#
    46тЖТ    46тЖТ#   Ex-1: use default linker then test plain C algorithm with V3 GDB run
    47тЖТ    47тЖТ#   ../test.sh "" nds32le-elf-gcc nds32le-elf-run GDB BS3
    48тЖТ    48тЖТ#
    49тЖТ    49тЖТ#   Ex-2: use default linker then test DSP algorithm with RV32 SID

    54тЖТ    54тЖТ#
    55тЖТ    55тЖТ#   Ex-4: use default linker then test v-ext 0.8 algorithm with SID
    56тЖТ    56тЖТ#   ../test.sh "" riscv64-elf-gcc riscv-sim-wrapper SID BS3 "-mext-vector -DNDS_VEC_RVV_VERSION=80"
    57тЖТ    57тЖТ#
    58тЖТ    58тЖТ#   Ex-5: use default linker then test v-ext 1.0 algorithm with SID
    59тЖТ    59тЖТ#   ../test.sh "" riscv64-elf-gcc riscv-sim-wrapper SID BS3 "-mext-vector -DNDS_VEC_RVV_VERSION=1000"

    72тЖТ    72тЖТfunction check_coverage {
    73тЖТ    73тЖТ    local gcda_list called_fun_list fun_coverage fun_name coverage_status
    74тЖТ    74тЖТ    # remember change directory to the folder used to keep gcov outputs
    75тЖТ    75тЖТ    cd ${GCOV_DIR}
    76тЖТ    76тЖТ
    77тЖТ    77тЖТ    gcda_list=($(ls ${LIB_OBJ_DIR}/*.gcda))
    78тЖТ    78тЖТ    for gcda in ${gcda_list[@]}; do
    79тЖТ    79тЖТ        called_fun_list=$(${GCOV} -f ${gcda} | grep -w "Function" -A1)
    80тЖТ    80тЖТ        IFS=-
    81тЖТ    81тЖТ        called_fun_list=(${called_fun_list})
    82тЖТ    82тЖТ        IFS=
    83тЖТ    83тЖТ        coverage_status=""
    84тЖТ    84тЖТ        is_coverage_fail=0
    85тЖТ    85тЖТ
    86тЖТ    86тЖТ        # let the function with riscv_nn prefix be shown first
    87тЖТ    87тЖТ        for called_fun in "${called_fun_list[@]}"; do
    88тЖТ    88тЖТ            if [[ "${called_fun}" == *"${FUN_PREFIX}"* ]]; then
    89тЖТ    89тЖТ                fun_coverage=$(grep -Eo "[[:digit:]]{1,3}.[[:digit:]]{2}%" <<< ${called_fun})
    90тЖТ    90тЖТ                # remember to ignore those functions with 0.00% coverage
    91тЖТ    91тЖТ                if [[ "${fun_coverage}" != 0.00% ]]; then
    92тЖТ    92тЖТ                    fun_name=$(grep -w "Function" <<< ${called_fun} | cut -d "'" -f2)
    93тЖТ    93тЖТ                    coverage_status+="${fun_name},${fun_coverage},"
    94тЖТ    94тЖТ                    # coverage_status+=""
    95тЖТ    95тЖТ                    if [[ "${fun_coverage}" != "100.00%" ]]; then
    96тЖТ    96тЖТ                        is_coverage_fail=1
    97тЖТ    97тЖТ                    fi
    98тЖТ    98тЖТ                fi
    99тЖТ    99тЖТ
   100тЖТ   100тЖТ                called_fun_list=("${called_fun_list[@]/${called_fun}}")
   101тЖТ   101тЖТ                break
   102тЖТ   102тЖТ            fi
   103тЖТ   103тЖТ        done
   104тЖТ   104тЖТ
   105тЖТ   105тЖТ        for called_fun in ${called_fun_list[@]}; do
   106тЖТ   106тЖТ            fun_coverage=$(grep -Eo "[[:digit:]]{1,3}.[[:digit:]]{2}%" <<< ${called_fun})
   107тЖТ   107тЖТ            # remember to ignore those functions with 0.00% coverage
   108тЖТ   108тЖТ            if [[ "${fun_coverage}" != 0.00% ]]; then
   109тЖТ   109тЖТ                fun_name=$(grep -w "Function" <<< ${called_fun} | cut -d "'" -f2)
   110тЖТ   110тЖТ                coverage_status+="${fun_name},${fun_coverage},"
   111тЖТ   111тЖТ                if [[ "${fun_coverage}" != "100.00%" ]]; then
   112тЖТ   112тЖТ                    is_coverage_fail=1
   113тЖТ   113тЖТ                fi
   114тЖТ   114тЖТ            fi
   115тЖТ   115тЖТ        done
   116тЖТ   116тЖТ
   117тЖТ   117тЖТ        if [ ${is_coverage_fail} -eq 1 ]; then
   118тЖТ   118тЖТ            coverage_fail_list+=("${coverage_status}")
   119тЖТ   119тЖТ            echo ${coverage_status} >> ${COVERAGE_FAIL_LIST}
   120тЖТ   120тЖТ            cp $(basename ${gcda/gcda/c.gcov}) ${FAILED_GCOV_DIR}
   121тЖТ   121тЖТ        fi
   122тЖТ   122тЖТ
   123тЖТ   123тЖТ        # write coverage results into file
   124тЖТ   124тЖТ        echo ${coverage_status} >> ${COV_CSV}
   125тЖТ   125тЖТ    done
   126тЖТ   126тЖТ
   127тЖТ   127тЖТ    # go back to previous path
   128тЖТ   128тЖТ    cd ${CURRENT_PATH}
   129тЖТ   129тЖТ}

   135тЖТ   135тЖТcalculate_code_size () {
   136тЖТ   136тЖТ    local log_file build_cmd tmp_size_log
   137тЖТ   137тЖТ
   138тЖТ   138тЖТ    echo -n "${1}," >> ${SIZE_CSV}
   139тЖТ   139тЖТ    log_file="${LOG_DIR}/${1}_empty.log"
   140тЖТ   140тЖТ    build_cmd="${CC} ${DEFAULT_APP_CFLAGS} -O3 ${EXTRA_FLAG} -DENA_DUMP_GOLDEN=${ENA_DUMP_GOLDEN} -DENA_MEASURE_PERF=${ENA_MEASURE_PERF} -DGETENV_TESTDIR=${UNIT_TEST_PATH} ${INC_DIR} -o ${ADX_DIR}/t_$1_empty.adx ${UNIT_TEST_PATH}/t_$1.c ${EMPTY_LIB} -lm"
   141тЖТ   141тЖТ    #use the linker specified by the user
   142тЖТ   142тЖТ    if [[ -n ${LD} ]]; then
   143тЖТ   143тЖТ        build_cmd="${build_cmd} -fuse-ld=${LD}"
   144тЖТ   144тЖТ    fi
   145тЖТ   145тЖТ
   146тЖТ   146тЖТ    # build the executable
   147тЖТ   147тЖТ    ${build_cmd} 2>>${log_file} | tee -a ${log_file}
   148тЖТ   148тЖТ
   149тЖТ   149тЖТ    # get the size info
   150тЖТ   150тЖТ    tmp_size_log="${LOG_DIR}/size.log"
   151тЖТ   151тЖТ    ${SIZE} ${ADX_DIR}/t_${1}.adx       | tail -1 > ${tmp_size_log}
   152тЖТ   152тЖТ    ${SIZE} ${ADX_DIR}/t_${1}_empty.adx | tail -1 >> ${tmp_size_log}
   153тЖТ   153тЖТ
   154тЖТ   154тЖТ    # calculate the size of each section
   155тЖТ   155тЖТ    awk '
   156тЖТ   156тЖТ        NR==1 {split($0,non_empty)}
   157тЖТ   157тЖТ        NR==2 {split($0,empty)}
   158тЖТ   158тЖТ        END {for(i=1;i<=5;i++) printf "%u,", non_empty[i] - empty[i]}
   159тЖТ   159тЖТ    ' ${tmp_size_log} >> ${SIZE_CSV}
   160тЖТ   160тЖТ    echo >> ${SIZE_CSV}  #new line
   161тЖТ   161тЖТ}

   168тЖТ   168тЖТfunction run_and_check {
   169тЖТ   169тЖТ    # skip the fp16 functions when the zfh extension is not enabled
   170тЖТ   170тЖТ    if [ ${ENA_FP16} -eq 0 ] && [[ "${1}" == *"f16"* ]]; then
   171тЖТ   171тЖТ        accuracy_unsupport_num=$((accuracy_unsupport_num + 1))
   172тЖТ   172тЖТ        echo "$2,$1,x,x," >> ${PERF_CSV}
   173тЖТ   173тЖТ        return
   174тЖТ   174тЖТ    fi
   175тЖТ   175тЖТ
   176тЖТ   176тЖТ    # skip the integer 4-bit functions when the v-ext enabled but the VLN8 not enabled
   177тЖТ   177тЖТ    if [[ ${EXTRA_FLAG} =~ -mext-vector ]] && [[ ${ENA_VL4} -eq 0 ]] && [[ ${1} =~ s4 ]]; then
   178тЖТ   178тЖТ        accuracy_unsupport_num=$((accuracy_unsupport_num + 1))
   179тЖТ   179тЖТ        echo "$2,$1,x,x," >> ${PERF_CSV}
   180тЖТ   180тЖТ        return
   181тЖТ   181тЖТ    fi
   182тЖТ   182тЖТ
   183тЖТ   183тЖТ    log "--- $1 ---"
   184тЖТ   184тЖТ    local log_file="${LOG_DIR}/${1}.log"
   185тЖТ   185тЖТ    build_cmd="${CC} ${DEFAULT_APP_CFLAGS} -O3 -DENA_DUMP_GOLDEN=${ENA_DUMP_GOLDEN} -DENA_MEASURE_PERF=${ENA_MEASURE_PERF} -DGETENV_TESTDIR=${UNIT_TEST_PATH} ${INC_DIR} -o ${ADX_DIR}/t_$1.adx ${UNIT_TEST_PATH}/t_$1.c ${LINKED_LIB} ${EXTRA_FLAG} -lm"
   186тЖТ   186тЖТ    #use the linker specified by the user
   187тЖТ   187тЖТ    if [[ -n ${LD} ]]; then
   188тЖТ   188тЖТ        build_cmd="${build_cmd} -fuse-ld=${LD}"
   189тЖТ   189тЖТ    fi
   190тЖТ   190тЖТ
   191тЖТ   191тЖТ    echo ${build_cmd} > ${log_file}
   192тЖТ   192тЖТ    ${build_cmd} 2>>${log_file} | tee -a ${log_file}
   193тЖТ   193тЖТ    if [ ${PIPESTATUS[0]} == 0 ]; then  #check the compiling successful or not
   194тЖТ   194тЖТ        # dump the disassembly
   195тЖТ   195тЖТ        ${OBJDUMP} "${ADX_DIR}/t_${1}.adx" -S > "${OBJDUMP_DIR}/t_${1}.objdump"
   196тЖТ   196тЖТ
   197тЖТ   197тЖТ        log "Start at $(date)"
   198тЖТ   198тЖТ        ${SIM_WRAPPER} ${ADX_DIR}/t_$1.adx >> ${log_file} 2>&1
   199тЖТ   199тЖТ        ret=$?
   200тЖТ   200тЖТ        log "End at $(date)"
   201тЖТ   201тЖТ
   202тЖТ   202тЖТ        # check correctness
   203тЖТ   203тЖТ        if [ ${ret} == 0 ]; then
   204тЖТ   204тЖТ            # check whehter the program is executed (For the case of AndeSim
   205тЖТ   205тЖТ            # doesn't launch successfully. BTW, AndeSim doesn't return error
   206тЖТ   206тЖТ            # code even if it enounters some issues.)
   207тЖТ   207тЖТ            # Here we use the ALWAYS present string as the criteria whether
   208тЖТ   208тЖТ            # the test program is executed.
   209тЖТ   209тЖТ            grep -w 'pass\|fail' ${log_file} &> /dev/null
   210тЖТ   210тЖТ            if [ $? -ne 0 ]; then
   211тЖТ   211тЖТ                accuracy_fail_list+=("${1}")
   212тЖТ   212тЖТ                echo "Warning! There is no 'pass' or 'fail' string is parsed in this log file; therefore, the program may not executed." >> ${log_file}
   213тЖТ   213тЖТ            else
   214тЖТ   214тЖТ                accuracy_pass_num=$((accuracy_pass_num + 1))
   215тЖТ   215тЖТ            fi
   216тЖТ   216тЖТ
   217тЖТ   217тЖТ            if [ ${ENA_GREP_PF_TO_CSV} == 1 ]; then
   218тЖТ   218тЖТ                #dump inst/cycle into csv
   219тЖТ   219тЖТ                echo -e "$2,$1,\c" >> ${PERF_CSV}
   220тЖТ   220тЖТ                grep -w 'inst' ${log_file}  | awk '{printf "%ld,", $5}' >> ${PERF_CSV}
   221тЖТ   221тЖТ                grep -w 'cycle' ${log_file} | awk '{printf "%ld,", $5}' >> ${PERF_CSV}
   222тЖТ   222тЖТ                echo >> ${PERF_CSV}  #new line
   223тЖТ   223тЖТ            fi
   224тЖТ   224тЖТ
   225тЖТ   225тЖТ            if [ ${ENA_GREP_PARAM_TO_CSV} == 1 ]; then
   226тЖТ   226тЖТ                # following parser will work properly only as below assumptions
   227тЖТ   227тЖТ                # are valid
   228тЖТ   228тЖТ                #   (1) each parameter is seperated with semicolon except the
   229тЖТ   229тЖТ                #       last one
   230тЖТ   230тЖТ                #   (2) the parameters are surrounded by the "param_hint_begin"
   231тЖТ   231тЖТ                #       and "param_hint_end"
   232тЖТ   232тЖТ
   233тЖТ   233тЖТ                #dump the parameters into csv
   234тЖТ   234тЖТ                local param_hint_begin="___PARAMETERS_BEGIN___"
   235тЖТ   235тЖТ                local param_hint_end="___PARAMETERS_END___"
   236тЖТ   236тЖТ                local tmp_param_str=$(sed -n "/${param_hint_begin}/,/${param_hint_end}/p" ${log_file})
   237тЖТ   237тЖТ
   238тЖТ   238тЖТ                # remove all newlines
   239тЖТ   239тЖТ                tmp_param_str=${tmp_param_str//$'\r'/}
   240тЖТ   240тЖТ                tmp_param_str=${tmp_param_str//$'\n'/}
   241тЖТ   241тЖТ                # remove hint string
   242тЖТ   242тЖТ                tmp_param_str=${tmp_param_str//${param_hint_begin}/}
   243тЖТ   243тЖТ                tmp_param_str=${tmp_param_str//${param_hint_end}/,}
   244тЖТ   244тЖТ                # make each parameter being in one line
   245тЖТ   245тЖТ                tmp_param_str=${tmp_param_str//;/;$'\n'}
   246тЖТ   246тЖТ                echo -e "$2,$1,${tmp_param_str}" >> ${PARAM_CSV}
   247тЖТ   247тЖТ            fi
   248тЖТ   248тЖТ        else
   249тЖТ   249тЖТ            accuracy_fail_list+=("${1}")
   250тЖТ   250тЖТ        fi
   251тЖТ   251тЖТ    else
   252тЖТ   252тЖТ        # build the executable fail
   253тЖТ   253тЖТ        accuracy_fail_list+=("${1}")
   254тЖТ   254тЖТ    fi
   255тЖТ   255тЖТ
   256тЖТ   256тЖТ    if [[ ${ENA_MEASURE_CODE_SIZE} -eq 1 ]]; then
   257тЖТ   257тЖТ        calculate_code_size "${1}"
   258тЖТ   258тЖТ    fi
   259тЖТ   259тЖТ}
