     1→#!/usr/bin/env bash
     2→################################################################################
     3→#
     4→# This script is used to verify libnn accuracy, code coverage and measure
     5→# performance.
     6→#
     7→# Input:
     8→#   $1: Source code path (without / tailed)
     9→#       This is a dummy argument, just for making the API be the same as test.sh
    10→#       of libdsp.
    11→#   $2: Compiler name (with prefix)
    12→#       Set it to the compiler name without full path, e.g., nds32le-elf-gcc,
    13→#       riscv32-elf-clang, riscv64-linux-gcc, etc. This script will try to find
    14→#       the full path of this specified compiler from the PATH environment
    15→#       variable; therefore, please export the proper toolcahin path into PATH
    16→#       environment variable before using this script.
    17→#   $3: Simulator name
    18→#       In DAILY test, please give the full path of simulator.
    19→#   $4: Simulation target
    20→#       It could be GDB, SID or BOARD.
    21→#   $5: BS3 / DAILY
    22→#       BS3:   Verify the accuracy with the built-in multi-lib. (note the sat
    23→#              extension and code coverage won't be verified here)
    24→#       DAILY: Use the user specified compiler and compilation options to build
    25→#              the library, and this built library will be used in the accuracy
    26→#              tests later. Not only the accuracy but also the code coverage
    27→#              will be verified in this mode.
    28→#   $6: Extra flags for selecting multi-lib or the algorithm to be built as library
    29→#       As the 5th argument is set to BS3, these flags are used to select the
    30→#       built-in multi-lib. For example, if "-mext-dsp" is contained in this
    31→#       argument, the DSP version library will be used for linking. By default,
    32→#       the plain C version library will be used for linking.
    33→#       As the 5th argument is set to DAILY, these flags will be used to select
    34→#       the proper algorithm and build them into library. For example, if
    35→#       "-mext-dsp" is contained in this argument, the DSP algorithm will be
    36→#       used as building the library. By default, the plain C algorithm will be
    37→#       used to build the library.
    38→#   $7: Specified linker
    39→#       Use the specified linker instead of the default one when building the
    40→#       executable. If this argument is left to be empty, the default linker
    41→#       will be used. (e.g., ld, lld, etc.)
    42→#
    43→# Example:
    44→#   mkdir build_dir && cd build_dir
    45→#
    46→#   Ex-1: use default linker then test plain C algorithm with V3 GDB run
    47→#   ../test.sh "" nds32le-elf-gcc nds32le-elf-run GDB BS3
    48→#
    49→#   Ex-2: use default linker then test DSP algorithm with RV32 SID
    50→#   ../test.sh "" riscv32-elf-gcc riscv-sim-wrapper SID BS3 "-mext-dsp"
    51→#
    52→#   Ex-3: use lld as linker, clang as compiler then test DSP algorithm with RV64 SID
    53→#   ../test.sh "" riscv64-linux-clang riscv-sim-wrapper SID BS3 "-mext-dsp" lld
    54→#
    55→#   Ex-4: use default linker then test v-ext 0.8 algorithm with SID
    56→#   ../test.sh "" riscv64-elf-gcc riscv-sim-wrapper SID BS3 "-mext-vector -DNDS_VEC_RVV_VERSION=80"
    57→#
    58→#   Ex-5: use default linker then test v-ext 1.0 algorithm with SID
    59→#   ../test.sh "" riscv64-elf-gcc riscv-sim-wrapper SID BS3 "-mext-vector -DNDS_VEC_RVV_VERSION=1000"
    60→#
    61→# Note:
    62→#   In build-system-3, the 3rd parameter is allways "riscv-sim-wrapper".
    63→#   build-system-3 will prepare a proper warpper named "riscv-sim-wrapper"
    64→#   according to the specified simulation target by users, and this wrapper will
    65→#   be used in the accuray test later.
    66→#

    69→#######################################################
    70→# check the coverage of executed functions
    71→#######################################################
    72→function check_coverage {
    73→    local gcda_list called_fun_list fun_coverage fun_name coverage_status
    74→    # remember change directory to the folder used to keep gcov outputs
    75→    cd ${GCOV_DIR}
    76→
    77→    gcda_list=($(ls ${LIB_OBJ_DIR}/*.gcda))
    78→    for gcda in ${gcda_list[@]}; do
    79→        called_fun_list=$(${GCOV} -f ${gcda} | grep -w "Function" -A1)
    80→        IFS=-
    81→        called_fun_list=(${called_fun_list})
    82→        IFS=
    83→        coverage_status=""
    84→        is_coverage_fail=0
    85→
    86→        # let the function with riscv_nn prefix be shown first
    87→        for called_fun in "${called_fun_list[@]}"; do
    88→            if [[ "${called_fun}" == *"${FUN_PREFIX}"* ]]; then
    89→                fun_coverage=$(grep -Eo "[[:digit:]]{1,3}.[[:digit:]]{2}%" <<< ${called_fun})
    90→                # remember to ignore those functions with 0.00% coverage
    91→                if [[ "${fun_coverage}" != 0.00% ]]; then
    92→                    fun_name=$(grep -w "Function" <<< ${called_fun} | cut -d "'" -f2)
    93→                    coverage_status+="${fun_name},${fun_coverage},"
    94→                    # coverage_status+=""
    95→                    if [[ "${fun_coverage}" != "100.00%" ]]; then
    96→                        is_coverage_fail=1
    97→                    fi
    98→                fi
    99→
   100→                called_fun_list=("${called_fun_list[@]/${called_fun}}")
   101→                break
   102→            fi
   103→        done
   104→
   105→        for called_fun in ${called_fun_list[@]}; do
   106→            fun_coverage=$(grep -Eo "[[:digit:]]{1,3}.[[:digit:]]{2}%" <<< ${called_fun})
   107→            # remember to ignore those functions with 0.00% coverage
   108→            if [[ "${fun_coverage}" != 0.00% ]]; then
   109→                fun_name=$(grep -w "Function" <<< ${called_fun} | cut -d "'" -f2)
   110→                coverage_status+="${fun_name},${fun_coverage},"
   111→                if [[ "${fun_coverage}" != "100.00%" ]]; then
   112→                    is_coverage_fail=1
   113→                fi
   114→            fi
   115→        done
   116→
   117→        if [ ${is_coverage_fail} -eq 1 ]; then
   118→            coverage_fail_list+=("${coverage_status}")
   119→            echo ${coverage_status} >> ${COVERAGE_FAIL_LIST}
   120→            cp $(basename ${gcda/gcda/c.gcov}) ${FAILED_GCOV_DIR}
   121→        fi
   122→
   123→        # write coverage results into file
   124→        echo ${coverage_status} >> ${COV_CSV}
   125→    done
   126→
   127→    # go back to previous path
   128→    cd ${CURRENT_PATH}
   129→}

   131→#######################################################
   132→# collect the code size of each section
   133→#   $1: test function name
   134→#######################################################
   135→calculate_code_size () {
   136→    local log_file build_cmd tmp_size_log
   137→
   138→    echo -n "${1}," >> ${SIZE_CSV}
   139→    log_file="${LOG_DIR}/${1}_empty.log"
   140→    build_cmd="${CC} ${DEFAULT_APP_CFLAGS} -O3 ${EXTRA_FLAG} -DENA_DUMP_GOLDEN=${ENA_DUMP_GOLDEN} -DENA_MEASURE_PERF=${ENA_MEASURE_PERF} -DGETENV_TESTDIR=${UNIT_TEST_PATH} ${INC_DIR} -o ${ADX_DIR}/t_$1_empty.adx ${UNIT_TEST_PATH}/t_$1.c ${EMPTY_LIB} -lm"
   141→    #use the linker specified by the user
   142→    if [[ -n ${LD} ]]; then
   143→        build_cmd="${build_cmd} -fuse-ld=${LD}"
   144→    fi
   145→
   146→    # build the executable
   147→    ${build_cmd} 2>>${log_file} | tee -a ${log_file}
   148→
   149→    # get the size info
   150→    tmp_size_log="${LOG_DIR}/size.log"
   151→    ${SIZE} ${ADX_DIR}/t_${1}.adx       | tail -1 > ${tmp_size_log}
   152→    ${SIZE} ${ADX_DIR}/t_${1}_empty.adx | tail -1 >> ${tmp_size_log}
   153→
   154→    # calculate the size of each section
   155→    awk '
   156→        NR==1 {split($0,non_empty)}
   157→        NR==2 {split($0,empty)}
   158→        END {for(i=1;i<=5;i++) printf "%u,", non_empty[i] - empty[i]}
   159→    ' ${tmp_size_log} >> ${SIZE_CSV}
   160→    echo >> ${SIZE_CSV}  #new line
   161→}

   163→#######################################################
   164→# run the unit test and check the accuracy results
   165→#   $1: test function name
   166→#   $2: test function group name (for grouping excel results)
   167→#######################################################
   168→function run_and_check {
   169→    # skip the fp16 functions when the zfh extension is not enabled
   170→    if [ ${ENA_FP16} -eq 0 ] && [[ "${1}" == *"f16"* ]]; then
   171→        accuracy_unsupport_num=$((accuracy_unsupport_num + 1))
   172→        echo "$2,$1,x,x," >> ${PERF_CSV}
   173→        return
   174→    fi
   175→
   176→    # skip the integer 4-bit functions when the v-ext enabled but the VLN8 not enabled
   177→    if [[ ${EXTRA_FLAG} =~ -mext-vector ]] && [[ ${ENA_VL4} -eq 0 ]] && [[ ${1} =~ s4 ]]; then
   178→        accuracy_unsupport_num=$((accuracy_unsupport_num + 1))
   179→        echo "$2,$1,x,x," >> ${PERF_CSV}
   180→        return
   181→    fi
   182→
   183→    log "--- $1 ---"
   184→    local log_file="${LOG_DIR}/${1}.log"
   185→    build_cmd="${CC} ${DEFAULT_APP_CFLAGS} -O3 -DENA_DUMP_GOLDEN=${ENA_DUMP_GOLDEN} -DENA_MEASURE_PERF=${ENA_MEASURE_PERF} -DGETENV_TESTDIR=${UNIT_TEST_PATH} ${INC_DIR} -o ${ADX_DIR}/t_$1.adx ${UNIT_TEST_PATH}/t_$1.c ${LINKED_LIB} ${EXTRA_FLAG} -lm"
   186→    #use the linker specified by the user
   187→    if [[ -n ${LD} ]]; then
   188→        build_cmd="${build_cmd} -fuse-ld=${LD}"
   189→    fi
   190→
   191→    echo ${build_cmd} > ${log_file}
   192→    ${build_cmd} 2>>${log_file} | tee -a ${log_file}
   193→    if [ ${PIPESTATUS[0]} == 0 ]; then  #check the compiling successful or not
   194→        # dump the disassembly
   195→        ${OBJDUMP} "${ADX_DIR}/t_${1}.adx" -S > "${OBJDUMP_DIR}/t_${1}.objdump"
   196→
   197→        log "Start at $(date)"
   198→        ${SIM_WRAPPER} ${ADX_DIR}/t_$1.adx >> ${log_file} 2>&1
   199→        ret=$?
   200→        log "End at $(date)"
   201→
   202→        # check correctness
   203→        if [ ${ret} == 0 ]; then
   204→            # check whehter the program is executed (For the case of AndeSim
   205→            # doesn't launch successfully. BTW, AndeSim doesn't return error
   206→            # code even if it enounters some issues.)
   207→            # Here we use the ALWAYS present string as the criteria whether
   208→            # the test program is executed.
   209→            grep -w 'pass\|fail' ${log_file} &> /dev/null
   210→            if [ $? -ne 0 ]; then
   211→                accuracy_fail_list+=("${1}")
   212→                echo "Warning! There is no 'pass' or 'fail' string is parsed in this log file; therefore, the program may not executed." >> ${log_file}
   213→            else
   214→                accuracy_pass_num=$((accuracy_pass_num + 1))
   215→            fi
   216→
   217→            if [ ${ENA_GREP_PF_TO_CSV} == 1 ]; then
   218→                #dump inst/cycle into csv
   219→                echo -e "$2,$1,\c" >> ${PERF_CSV}
   220→                grep -w 'inst' ${log_file}  | awk '{printf "%ld,", $5}' >> ${PERF_CSV}
   221→                grep -w 'cycle' ${log_file} | awk '{printf "%ld,", $5}' >> ${PERF_CSV}
   222→                echo >> ${PERF_CSV}  #new line
   223→            fi
   224→
   225→            if [ ${ENA_GREP_PARAM_TO_CSV} == 1 ]; then
   226→                # following parser will work properly only as below assumptions
   227→                # are valid
   228→                #   (1) each parameter is seperated with semicolon except the
   229→                #       last one
   230→                #   (2) the parameters are surrounded by the "param_hint_begin"
   231→                #       and "param_hint_end"
   232→
   233→                #dump the parameters into csv
   234→                local param_hint_begin="___PARAMETERS_BEGIN___"
   235→                local param_hint_end="___PARAMETERS_END___"
   236→                local tmp_param_str=$(sed -n "/${param_hint_begin}/,/${param_hint_end}/p" ${log_file})
   237→
   238→                # remove all newlines
   239→                tmp_param_str=${tmp_param_str//$'\r'/}
   240→                tmp_param_str=${tmp_param_str//$'\n'/}
   241→                # remove hint string
   242→                tmp_param_str=${tmp_param_str//${param_hint_begin}/}
   243→                tmp_param_str=${tmp_param_str//${param_hint_end}/,}
   244→                # make each parameter being in one line
   245→                tmp_param_str=${tmp_param_str//;/;$'\n'}
   246→                echo -e "$2,$1,${tmp_param_str}" >> ${PARAM_CSV}
   247→            fi
   248→        else
   249→            accuracy_fail_list+=("${1}")
   250→        fi
   251→    else
   252→        # build the executable fail
   253→        accuracy_fail_list+=("${1}")
   254→    fi
   255→
   256→    if [[ ${ENA_MEASURE_CODE_SIZE} -eq 1 ]]; then
   257→        calculate_code_size "${1}"
   258→    fi
   259→}

   265→function target_cfg_diagnosis {

   271→
   272→    ${build_cmd} 2>>${log_file} | tee -a ${log_file}
   273→    if [ ${PIPESTATUS[0]} == 0 ]; then  #check the compiling successful or not

   275→        ${SIM_WRAPPER} ${ADX_DIR}/${src_fname}.adx >> ${log_file} 2>&1
   276→        ret=$?
   277→        log "End at $(date)"

   303→#######################################################
   304→# Some function's algorithm implementaions are hardware
   305→# configuration dependant; thus, there are mutiple
   306→# algoirhtms for different hardware configurations and
   307→# the code coverage is impossible to be 100% as executing
   308→# the function under one hardware configuration. Here we
   309→# execute these functions with all hardware configuration
   310→# simulator warpper to make the coverage reach 100%.
   311→#######################################################
   312→function run_special_case_for_rvv () {
   313→    local sim_wrapper_list=($(ls ${SIM_WRAPPER%v*}*))
   314→    local log_file="${LOG_DIR}/special_case.log"
   315→
   316→    for sim_wrapper in "${sim_wrapper_list[@]}"; do
   317→      for fun in "${SPECIAL_CASE_RVV[@]}"; do
   318→          echo "${sim_wrapper} ${ADX_DIR}/t_${fun}.adx" >> ${log_file}
   319→          ${sim_wrapper} ${ADX_DIR}/t_${fun}.adx >> ${log_file}
   320→      done
   321→    done
   322→}

   324→#######################################################
   325→# Build the library
   326→#   $1: flags used to build the library
   327→#######################################################
   328→build_lib () {
   329→    local build_lib_stdout build_lib_stderr
   330→    build_lib_stdout=${CURRENT_PATH}/${TEST_LOG_DIR}/build_lib.log
   331→    build_lib_stderr=${CURRENT_PATH}/${TEST_LOG_DIR}/build_lib_error.log
   332→
   333→    log "****************************************************"
   334→    log "Build libnn ..."
   335→    start_time
   336→    log "    LIB_CFLAGS= ${1}"
   337→    log "    Output=${build_lib_stdout}"
   338→    log "    Error=${build_lib_stderr}"
   339→    log
   340→
   341→    ${LIB_ROOT_PATH}/build.sh "${CPU_NAME}" "${1}" "${CC}" "${CURRENT_PATH}/${TEST_LOG_DIR}" > "${build_lib_stdout}" 2> "${build_lib_stderr}"
   342→    if [[ ${CC} =~ "linux" ]] && [[ ! ${EXTRA_FLAG} =~ "-static" ]]; then
   343→        # build the dynamic library for glibc toolcahin ...
   344→        local all_obj_list=$(ls ${LIB_OBJ_DIR}/*.o)
   345→        local build_shared_lib_cmd="${CC} -shared -o ${LIB_ROOT_PATH}/${LIB_NAME}.so ${all_obj_list}"
   346→        echo ${build_shared_lib_cmd} >> "${build_lib_stdout}"
   347→        ${build_shared_lib_cmd} >> "${build_lib_stdout}" 2> "${build_lib_stderr}"
   348→    fi
   349→    build_lib_result=$?
   350→
   351→    if [ ${build_lib_result} == 0 ]; then
   352→        log "Build library done."
   353→    else
   354→        log "Build library FAIL !!!"
   355→    fi
   356→    end_time
   357→    log "****************************************************"
   358→}
