    41→class TransformSpace(object):
    42→    """Base class for transform space
    43→    TransformSpace is the node in the computation graph of axes
    44→
    45→    .. note::
    46→
    47→        We can regard our schedule code as a combination of transformation of axes.
    48→        Starting from raw axes in the definition of MLIR operation, we can transform
    49→        these axes by some operators.
    50→        Each operator has some tunable parameters.
    51→        Then the tuning process is just to find good parameters of these op.
    52→
    53→    So all the combinations of the parameters of these op form our search space.
    54→
    55→    Naming convention:
    56→    We call the set of all possible values as XXXSpace. (XXX can be Tile ...)
    57→    We call a specific entity in a space as XXXEntity.
    58→    """
    59→
    60→    def __init__(self):
    61→        self.ins = []
    62→        self.num_output = 0
    63→        self.entities = []
    64→
    65→    def __len__(self):
    66→        return len(self.entities)
    67→
    68→    def __getitem__(self, index):
    69→        """Get an entity of the space by index
    70→
    71→        Parameters
    72→        ----------
    73→        index: int
    74→
    75→        Returns
    76→        -------
    77→        transform entity
    78→        """
    79→        return self.entities[index]
    80→
    81→    @staticmethod
    82→    def get_num_output() -> int:
    83→        """get number of output axes after this transform
    84→
    85→        Returns
    86→        -------
    87→        n: int
    88→            number of output axes
    89→        """
    90→        return 0
    91→
    92→
    93→def get_factors(n):
    94→    """return all factors of an integer
    95→
    96→    Parameters
    97→    ----------
    98→    n: int
    99→        integer to factorize
   100→
   101→    Returns
   102→    -------
   103→    factors: list
   104→        List of all factors
   105→    """
   106→    if n == 0:
   107→        return [0]
   108→    step = 2 if n % 2 else 1
   109→    ret = list(
   110→        set(
   111→            functools.reduce(
   112→                list.__add__,
   113→                ([i, n // i] for i in range(1, int(math.sqrt(n)) + 1, step) if n % i == 0),
   114→            )
   115→        )
   116→    )
   117→    ret.sort()
   118→    return ret
   119→
   120→
   121→def get_pow2s(n):
   122→    """return all power-of-two numbers that are less or equal than the integer
   123→
   124→    Parameters
   125→    ----------
   126→    n: int
   127→        integer for reference
   128→
   129→    Returns
   130→    -------
   131→    factors: list
   132→        List of all power-of-two numbers
   133→    """
   134→    if n == 0:
   135→        return [0]
   136→    return [2**x for x in range(math.floor(math.log2(n)) + 1)]

   138→PARALLEL = "parallel"
   139→REDUCTION = "reduction"

   141→class Axis(TransformSpace):
   142→    '''
   143→    An axis represents loop, iteration, mlir scf.for
   144→
   145→    no: No of axis of opname
   146→
   147→    trip_count: iteration trip count
   148→
   149→    type: parallel or reduce
   150→    '''
   151→    def __init__(self,
   152→                 opid: int,
   153→                 no: int,
   154→                 trip_count: int,
   155→                 type: str):
   156→        super().__init__()
   157→        self.opid = opid
   158→        self.no = no
   159→        self.trip_count = trip_count
   160→        self.type = type
   161→        if opid < 0:
   162→            raise ValueError
   163→        if no < 0:
   164→            raise ValueError
   165→        if trip_count < 1:
   166→            raise ValueError
   167→        if type != PARALLEL and type != REDUCTION:
   168→            raise ValueError
   169→    
   170→    def __eq__(self, value: object) -> bool:
   171→        assert isinstance(value, Axis) 
   172→        return self.opid == value.opid and \
   173→               self.no == value.no and \
   174→               self.trip_count == value.trip_count and \
   175→               self.type == value.type
   176→    
   177→    def __hash__(self) -> int:
   178→        text = self.__repr__()
   179→        return text.__hash__()
   180→
   181→    @staticmethod
   182→    def get_num_output() -> int:
   183→        return 1
   184→    
   185→    def __repr__(self) -> str:
   186→        return (f"Axis(opid={self.opid}, no={self.no}, "
   187→                f"trip_count={self.trip_count}, type={self.type})")
   188→    
   189→    @property
   190→    def record_format(self):
   191→        return f"{self.opid}, {self.no}, {self.trip_count}, {self.type}"
   192→
   193→    @staticmethod
   194→    def get_axes(opid: int, trip_counts: list[int], types: list[str]):
   195→        assert len(trip_counts) == len(types)
   196→        axes = []
   197→        count = 0
   198→        for trip_count, type in zip(trip_counts, types):
   199→            axes.append(Axis(opid, count, trip_count, type))
   200→            count += 1
   201→        return axes
   202→

   204→class OpInfo():
   205→    def __init__(self,
   206→                 no: int,
   207→                 opname: str,
   208→                 iteration_space: list[int],
   209→                 iteration_type: list[str],
   210→                 tile_levels: tuple[int, int]) -> None:
   211→        assert len(iteration_space) == len(iteration_type)
   212→        self.no = no
   213→        self.opname = opname
   214→        self.iteration_space = iteration_space
   215→        self.iteration_type = iteration_type
   216→        self.tile_levels = tile_levels
   217→        self.axes: list[Axis] = Axis.get_axes(no, iteration_space, iteration_type)
   218→    
   219→    def serialize(self):
   220→        return tuple(self.__dict__.values())
   221→    
   222→    @staticmethod
   223→    def deserialize(args: tuple):
   224→        assert len(args) == 5
   225→        return OpInfo(args[0], args[1], args[2], args[3], args[4])

   228→class TileSpace(TransformSpace):
   229→    """
   230→    Combinations of IREE tiling sizes of a axis
   231→    """
   232→    def __init__(self,
   233→                 trip_count: int,
   234→                 isParallel: bool,
   235→                 policy: str,
   236→                 tiling_level: int,
   237→                 max_tile_size: int,
   238→                 candidates: list[list[int]] | None = None,
   239→                 zero_vec_common = False,
   240→                 zero_vec_inner = False,
   241→                 fil = lambda x : True):
   242→        """
   243→        trip_count :
   244→            Loop size
   245→        isParallel :
   246→            Is a parallel axis
   247→        policy :
   248→            method to generate size number
   249→        tiling_level :
   250→            Number of tiles
   251→        max_tile_size :
   252→            Extra limit of tile size
   253→        candidates :
   254→            Fixed tile size, only available if policy is 'candidate'
   255→        zero_vec_common, zero_vec_inner :
   256→            In IREE, a parallel axis can not have non-zero size in VectorCommonTile and VectorInnerTile 
   257→            at the same time, these args can limit which tile level should be zero.
   258→        """
   259→        super().__init__()
   260→        self.policy = policy
   261→        self.entities: list[TileEntity] = []
   262→        self.product = trip_count
   263→        self.num_output = tiling_level
   264→        self.isParallel = isParallel
   265→        self.zero_vec_common = zero_vec_common
   266→        self.zero_vec_inner = zero_vec_inner
   267→
   268→        if policy == "candidate":
   269→            assert isinstance(candidates, list)
   270→            for candidate in candidates:
   271→                assert len(candidate) == self.num_output
   272→                self.entities.append(TileEntity(candidate))
   273→        else:
   274→            if policy == "factors":
   275→                factors = [0] + [x for x in get_factors(self.product) if x <= max_tile_size]
   276→            elif policy == "zero_plus_power2":
   277→                factors = [0] + [x for x in get_pow2s(self.product) if x <= max_tile_size]
   278→            else:
   279→                raise RuntimeError("Invalid policy: %s" % policy)
   280→
   281→            self.factors = factors
   282→            self._generate_space(0, [0] * self.num_output)         
   283→
   284→        self.entities = list(filter(fil, self.entities))   
   285→
   286→    def _generate_space(self, now, tmp_stack: list[int]):
   287→        """Generate space by DFS"""
   288→        if now == self.num_output:
   289→            if self.isParallel:
   290→                if now == 2 and self.zero_vec_common and tmp_stack[0] != 0:
   291→                    return
   292→                if now >= 3:
   293→                    if self.zero_vec_common and tmp_stack[1] != 0:
   294→                        return
   295→                    if self.zero_vec_inner and tmp_stack[0] != 0:
   296→                        return
   297→                    # vector related level can have at most one non-zero
   298→                    if tmp_stack[0] != 0 and tmp_stack[1] != 0:
   299→                        return
   300→                
   301→            prod = tmp_stack[0]
   302→            tile_sizes = [prod]
   303→            for i in range(1, now):
   304→                n = tmp_stack[i]
   305→                if n != 0:
   306→                    prod = prod * n if prod != 0 else n
   307→                    tile_sizes.append(prod)
   308→                else:
   309→                    tile_sizes.append(0)                
   310→            
   311→            if prod > self.product: # invalid
   312→                return
   313→
   314→            tile_sizes.reverse() # outer to inner
   315→            self.entities.append(TileEntity(tile_sizes))
   316→        else:
   317→            for factor in self.factors:
   318→                tmp_stack[now] = factor
   319→                self._generate_space(now + 1, tmp_stack)
   320→

   332→
   333→class ConfigSpace(object):
   334→    """
   335→    The collection of all MLIR operations and their transform spaces
   336→    """
   337→
   338→    def __init__(self):
   339→        self.space_map: OrderedDict[SpaceKey, TileSpace] = OrderedDict()  # name -> space
   340→        self._ops: OrderedDict[str, OpInfo] = OrderedDict()
   341→
   342→        self._length: int | None = None
   343→        self._range_length: int | None = None
   344→        self._dims: list[int] | None = None
   345→        self._cross_filters = []
   346→        self._canonicalized = False
   347→    
   348→    def add_op(self, opinfo: OpInfo):
   349→        """
   350→        Op must be added with order. space_map sorting is base on Op order
   351→        """
   352→        opname = opinfo.opname
   353→        iteration_space = opinfo.iteration_space
   354→        iteration_type = opinfo.iteration_type
   355→        tile_levels = opinfo.tile_levels
   356→        axes = opinfo.axes
   357→        
   358→        assert opname not in self._ops
   359→        logger.debug("Add Op %d: name=%s num_axis=%d", len(self._ops), opname, len(axes))
   360→        logger.debug("loop_size=%s", str(iteration_space))
   361→        logger.debug("loop_type=%s", str(iteration_type))
   362→        logger.debug("tile level=%d, %d", tile_levels[0], tile_levels[1])
   363→        self._ops[opname] = opinfo
   364→        self._canonicalized = False
   365→    
   366→    def get_op_axes(self, opanme: str) -> list[Axis]:
   367→        return self._ops[opanme].axes
   368→
   369→    def define_tile(self,
   370→                    axis: Axis,
   371→                    policy: str,
   372→                    num_splits: int,
   373→                    candidates: list[list[int]] | None = None,
   374→                    max_tile_size = 1 << 31,
   375→                    fil: Callable = lambda x : True):
   376→        if axis.trip_count == 0:
   377→            logger.debug("Skip definong TileSpace with axis trip count 0, assume it's a fake axis")
   378→            return
   379→        key: SpaceKey = (axis, "ts")
   380→        if key in self.space_map:
   381→            logger.error(f"tile space already exist for axis: {str(axis)}")
   382→            return
   383→
   384→        policies = policy.split(",")
   385→        zero_vector_common = policies.count("zero_vector_common") > 0
   386→        # since all root op are zero_vec_inner and only non-root may have zero_vector_common
   387→        # zero_vec_inner is on by default
   388→        zero_vec_inner = not zero_vector_common
   389→
   390→        max_tile_size = min(max_tile_size, axis.trip_count)
   391→        space: TransformSpace = TileSpace(
   392→            axis.trip_count, axis.type == PARALLEL, policies[0], num_splits, max_tile_size, candidates,
   393→            zero_vector_common, zero_vec_inner, fil)
   394→        logger.debug("Define TileSpace: axis=%s, policy=%s, tile_levels=%d, size=%d",
   395→                     str(axis), policy, num_splits, len(space))
   396→        self.space_map[key] = space
   397→        self._canonicalized = False
   398→
   399→    def canonicalize(self):
   400→        """
   401→        Sort space map accroding to [opid, axis_id, space_type]
   402→        """
   403→        if self._canonicalized:
   404→            return
   405→
   406→        def key_func(element: tuple[SpaceKey, TileSpace]):
   407→            axis = element[0][0]
   408→            opid = axis.opid
   409→            axis_id = axis.no
   410→            space_type = 0 # currently only TileSpace
   411→            return [opid, axis_id, space_type]
   412→        
   413→        logger.debug("Space before sort: %s", str(self.space_map.keys()))
   414→        self.space_map = OrderedDict(sorted(self.space_map.items(), key=key_func))
   415→        logger.debug("Space after sort: %s", str(self.space_map.keys()))
   416→        self._canonicalized = True
   417→
   418→    def is_knob_valid(self, knob: list[int]):
   419→        self.canonicalize()
   420→        if len(knob) != len(self.dims):
   421→            return False
   422→        entities: OrderedDict[SpaceKey, TileEntity] = OrderedDict()
   423→        for idx, (key, space) in zip(knob, self.space_map.items()):
   424→            if idx >= len(space):
   425→                return False
   426→            entities[key] = space[idx]
   427→            
   428→        for fil in self._cross_filters:
   429→            if not fil(entities):
   430→                return False
   431→
   432→        return True
   433→
   434→    def is_index_valid(self, index):
   435→        """Checks if the index satisfies the multi_filter condition,
   436→        if no multi_filter is set, return true
   437→
   438→        Parameters
   439→        ----------
   440→        index: int
   441→            index from the range of the space
   442→
   443→        Returns
   444→        -------
   445→        valid: bool
   446→            whether the index meets all the constraints
   447→        """
   448→        assert 0 <= index < self.range_length
   449→        self.canonicalize()
   450→        if len(self._cross_filters) == 0:
   451→            return True
   452→        
   453→        knob = self.point2knob(index)
   454→        return self.is_knob_valid(knob)
   455→
   456→    def multi_filter(self, filter):  # pylint: disable=redefined-builtin
   457→        """
   458→        Add multi_filer
   459→
   460→        multi_filer format:
   461→
   462→        def fil(arg: OrderedDict[SpaceKey, TileEntity]) -> Bool
   463→        """
   464→        self._cross_filters.append(filter)
   465→
   466→    @property
   467→    def range_length(self):
   468→        """Length of the index range in the space (product of all combinations)"""
   469→        self.canonicalize()
   470→        if self._range_length is None:
   471→            if len(self.space_map) > 0:
   472→                self._range_length = int(
   473→                    np.prod([len(x) for x in self.space_map.values()]))
   474→            else:
   475→                self._range_length = 0
   476→        return self._range_length
   477→
   478→    @property
   479→    def dims(self):
   480→        """Dimensions in the space"""
   481→        self.canonicalize()
   482→        if self._dims is None:
   483→            self._dims = [len(x) for x in self.space_map.values()]
   484→        return self._dims
   485→
   486→    def clear_cache(self):
   487→        """Clears the cache of index validity"""
   488→        self._dims = None
   489→        self._length = None
   490→        self._range_length = None
   491→
   492→    def point2knob(self, point):
   493→        """Convert point form (single integer) to knob (vector of index)
   494→
   495→        Parameters
   496→        ----------
   497→        point: int
   498→            point to convert
   499→
   500→        Returns
   501→        -------
   502→        knob: list
   503→            knob representation of the point
   504→        """
   505→        self.canonicalize()
   506→        knob = []
   507→        for dim in self.dims:
   508→            knob.append(point % dim)
   509→            point //= dim
   510→        
   511→        return knob
   512→
   513→    def knob2point(self, knob) -> int:
   514→        """Convert knob form (vector of index) to point form (single integer)
   515→
   516→        Parameters
   517→        ----------
   518→        knob: list
   519→            knob to convert
   520→
   521→        Returns
   522→        -------
   523→        point: int
   524→            point of the knob representation
   525→        """
   526→        self.canonicalize()
   527→        point = 0
   528→        for j, k in enumerate(knob):
   529→            point += int(np.prod(self.dims[:j])) * k
   530→        
   531→        return point
   532→
   533→    def __len__(self):
   534→        if self._length is None:
   535→            self._length = int(np.prod([len(x) for x in self.space_map.values()]))
   536→        return self._length
   537→
   538→    def get(self, index):
   539→        """Get a config entity with detailed parameters from this space
   540→
   541→        Parameters
   542→        ----------
   543→        index: int
   544→            index in the space
   545→        """
   546→        self.canonicalize()
   547→        ubound = self.range_length
   548→        if index < 0 or index >= ubound:
   549→            raise IndexError(f"Index out of range: size {ubound}, got index {index}")
   550→        if not self.is_index_valid(index):
   551→            raise IndexError(
   552→                f"Index does not correspond to the multi-filter condition, got index {index}. "
   553→                f"Use is_index_valid to pre-check"
   554→            )
   555→        entities = OrderedDict()
   556→        t = index
   557→        for key, space in self.space_map.items():
   558→            entities[key] = space[t % len(space)]
   559→            t //= len(space)
   560→        ret = ConfigEntity(index, self._ops, entities)
   561→        return ret
   562→    
   563→    def _multi_fil_vector_size(self, entities: OrderedDict[SpaceKey, TileEntity]) -> bool:
   564→        """
   565→        A loose vector size limit filter that can reduce large vector size but not all invalid case.
   566→        """
   567→        def check_op_vec(opinfo: OpInfo,
   568→                         global_p_vec_sizes: list[int],
   569→                         update_global=False) -> tuple[bool, list[int]]:
   570→            axes = opinfo.axes
   571→            p_level, _ = opinfo.tile_levels
   572→            p_vec_sizes = []
   573→            r_vec_sizes = []
   574→            for axis in axes:
   575→                key: SpaceKey = (axis, "ts")
   576→                sizes = entities[key].sizes
   577→                if axis.type == PARALLEL:
   578→                    if p_level >= 3:
   579→                        p_vec_sizes.append(sizes[-1] + sizes[-2]) # assume one of two size must be zero
   580→                    elif p_level == 2:
   581→                        p_vec_sizes.append(sizes[-1])
   582→                    else:
   583→                        raise
   584→                elif axis.type == REDUCTION:
   585→                    r_vec_sizes.append(sizes[-1]) # VectorReductionTiles
   586→                else:
   587→                    raise
   588→            
   589→            all_zero = len(p_vec_sizes) == p_vec_sizes.count(0)
   590→            if not all_zero and update_global:
   591→                global_p_vec_sizes = p_vec_sizes
   592→            else:
   593→                p_vec_sizes = global_p_vec_sizes
   594→            
   595→            SEW = 8
   596→            VLEN = 512
   597→            limit = 8 * VLEN / SEW
   598→
   599→            def prod(xs: list[int]) -> int:
   600→                prod = 1
   601→                for x in xs:
   602→                    prod *= x if x > 0 else 1
   603→                return prod
   604→
   605→            if len(r_vec_sizes) == 0:
   606→                vs = prod(p_vec_sizes)
   607→                logger.debug("Inferred vector size %d, limit %d", vs, limit)
   608→                if vs > limit:
   609→                    return False, global_p_vec_sizes
   610→            else:
   611→                for i in range(1, len(p_vec_sizes)):
   612→                    left = prod(p_vec_sizes[:i] + r_vec_sizes)
   613→                    right = prod(p_vec_sizes[i:] + r_vec_sizes)
   614→                    logger.debug("Inferred vector size %d, limit %d", left, limit)
   615→                    logger.debug("Inferred vector size %d, limit %d", right, limit)
   616→                    if left > limit or right > limit:
   617→                        return False, global_p_vec_sizes
   618→            return True, global_p_vec_sizes
   619→
   620→        
   621→        opinfos = [x for x in self._ops.values()]
   622→        lastop = opinfos[-1]
   623→        global_p_vec_sizes = [0] * lastop.iteration_type.count(PARALLEL)
   624→        valid, global_p_vec_sizes = check_op_vec(lastop, global_p_vec_sizes, True)
   625→        if not valid:
   626→            return False
   627→        if len(opinfos) > 1:
   628→            startop = opinfos[0]

   641→class MultiConfigSpace(object):
   642→    """
   643→    Combine of ConfigSpaces with same dimension, provide interfaces using
   644→    outer index and outer knob.
   645→    
   646→    Outer index will be convert to inner index to access inner ConfigSpace.
   647→    Outer knob entend inner knob with index of all ConfigSpaces at front.
   648→    e.g. knob [0, ...] can be convert from index belongs to first ConfigSpace
   649→    """
   650→    def __init__(self, cfgs: list[ConfigSpace]) -> None:
   651→        dims = len(cfgs[0].dims)
   652→        for cfg in cfgs:
   653→            assert dims == len(cfg.dims), f"{dims} {len(cfg.dims)}"
   654→            cfg.canonicalize()
   655→
   656→        self._cfgs = cfgs
   657→        self._space_sizes = [x.range_length for x in self._cfgs]
   658→        if len(self) != 0:
   659→            self._weight = [float(x) / len(self) for x in self._space_sizes]
   660→        else:
   661→            self._weight = [0.0] * len(self._cfgs)
   662→    
   663→    def __len__(self):
   664→        return int(np.sum(self._space_sizes))
   665→
   666→    def point2knob(self, point: int):
   667→        assert 0 <= point < len(self)
   668→        _point = point
   669→        knob: list[int] = []
   670→        group_count = 0
   671→        for space_size, cfg in zip(self._space_sizes, self._cfgs):
   672→            if point >= space_size:
   673→                group_count += 1
   674→                point -= space_size
   675→                continue
   676→            else:
   677→                knob.append(group_count)
   678→                knob.extend(cfg.point2knob(point))
   679→                break
   680→        
   681→        logger.debug("point2knob: %d to %s", _point, str(knob))
   682→        return knob
   683→    
   684→    def knob2point(self, knob: list[int]):
   685→        assert knob[0] < len(self._cfgs)
   686→        cfg = self._cfgs[knob[0]]
   687→        assert len(knob) == len(cfg.dims) + 1
   688→        for index, ubound in zip(knob[1:], cfg.dims):
   689→            if index >= ubound:
   690→                raise IndexError
   691→
   692→        point = cfg.knob2point(knob[1:])
   693→        point += sum(self._space_sizes[:knob[0]])
   694→        logger.debug("knob2point: %s to %d", str(knob), point)
   695→        return point
   696→    
   697→    def is_knob_valid(self, knob: list[int]):
   698→        if len(knob) < 2:
   699→            return False
   700→        if knob[0] < 0 or knob[0] >= len(self._cfgs):
   701→            return False
   702→        cfg = self._cfgs[knob[0]]
   703→        return cfg.is_knob_valid(knob[1:])
   704→
   705→    def is_index_valid(self, index: int):
   706→        ubound = len(self)
   707→        if index < 0 or index >= ubound:
   708→            return False
   709→        
   710→        knob = self.point2knob(index)
   711→        return self.is_knob_valid(knob)
   712→    
   713→    def get(self, index: int):
   714→        ubound = len(self)
   715→        if index < 0 or index >= ubound:
   716→            raise IndexError(f"Index out of range: size {ubound}, got index {index}")
   717→
   718→        knob = self.point2knob(index)
   719→        assert self.is_knob_valid(knob), f"{index}, {str(knob)}"
   720→        cfg = self._cfgs[knob[0]]
   721→        inner_idx = cfg.knob2point(knob[1:])
   722→        ce = cfg.get(inner_idx)
   723→        # recreate ConfigEntity with MutiConfigSpace index instead of ConfigSpace index
   724→        return ConfigEntity(index, ce.ops, ce.entity_map)

   727→class GAMCS(MultiConfigSpace):
   728→    """
   729→    Inherit MultiConfigSpace
   730→    provide GA operation
   731→    """
   732→    def __init__(self, arg) -> None:
   733→        if isinstance(arg, MultiConfigSpace):
   734→            self.__dict__.update(arg.__dict__)
   735→        elif isinstance(arg, list):
   736→            super().__init__(arg)
   737→        else:
   738→            raise
   739→
   740→    def crossover_1p(self, g1: list[int], g2: list[int], point: int | None = None):
   741→        """
   742→        one-point crossover
   743→
   744→        return two genes(can be invalid)
   745→        """        
   746→        # auto-select point by GAMCS
   747→        if point is None:
   748→            point = np.random.randint(len(g1) - 1) + 1
   749→        if point < 1 or point >= len(g1):
   750→            raise IndexError
   751→        
   752→        ng1 = g1[:point] + g2[point:]
   753→        ng2 = g2[:point] + g1[point:]
   754→        return ng1, ng2
   755→    
   756→    def mutation(self, gene: list[int], mutation_prob: float):
   757→        ret = gene[:]
   758→        # mutation among cfg, base on their combination number
   759→        if np.random.random() < mutation_prob:
   760→            randv = np.random.random()
   761→            for i in range(len(self._weight)):
   762→                weight = self._weight[i]
   763→                if randv >= weight:
   764→                    randv -= weight
   765→                    continue
   766→                else:
   767→                    ret[0] = i
   768→
   769→        dim_sizes = self._cfgs[ret[0]].dims
   770→        for i in range(1, len(ret)):
   771→            if np.random.random() < mutation_prob:
   772→                ret[i] = np.random.randint(dim_sizes[i - 1])
   773→            else:
   774→                # mutate to another cfg may cause unmutate gene to exceed upbound
   775→                ret[i] = ret[i] % dim_sizes[i-1] 
   776→        
   777→        return ret

  1225→def _createConfigSpace_CPUBufferOpsTileAndVectorize(opinfos: list[OpInfo], policy: str): 
  1226→    # DistrbutionTiles (level 0) always use last op config
  1227→    # VectorCommonParallelTiles(VCPT) use op its own config
  1228→
  1229→    # iree-src/compiler/src/iree/compiler/Codegen/LLVMCPU/LLVMCPULowerExecutableTarget.cpp:L121
  1230→    rootOp = _getRootOp(opinfos)
  1231→    if rootOp is None:
  1232→        return None
  1233→    root_total_level = rootOp.tile_levels[0] + rootOp.tile_levels[1]
  1234→    # iree-src/compiler/src/iree/compiler/Codegen/LLVMCPU/Passes.cpp:354
  1235→    # not exist case that CPUBufferOpsTileAndVectorize root has other levels
  1236→    if root_total_level != 4 or rootOp.tile_levels[0] != 3: 
  1237→        return None
  1238→    
  1239→    # since CPUBufferOpsTileAndVectorize only allow 4 levels op, below level is fixed
  1240→    # parallel tile config = [Distribution, VectorCommonParallel, VectorInnerParallel]
  1241→    # reduction tile config = [VectorReduction]
  1242→    Distribution = 0
  1243→    VectorCommonParallel = 1
  1244→    VectorReduction = 0
  1245→    VectorInnerParallel = 2
  1246→
  1247→    start_inner = _get_vectorinnerparallel_start_axis(opinfos)
  1248→    
  1249→    # iree-src/compiler/src/iree/compiler/Codegen/LLVMCPU/LLVMCPUTile.cpp:L59
  1250→    # Note. since all opinfo has tiling config, retrieve config from fist op as fallback is NOT needed
  1251→
  1252→    cfg = ConfigSpace()
  1253→    cfg.multi_filter(cfg._multi_fil_vector_size)
  1254→    for opinfo in opinfos:
  1255→        cfg.add_op(opinfo)
  1256→        axes = cfg.get_op_axes(opinfo.opname)
  1257→        p_level, r_level = opinfo.tile_levels
  1258→        if p_level != 3 or r_level != 1: # root is lv4 and others are always lv4
  1259→            return None
  1260→
  1261→        isLast = opinfo == opinfos[-1]
  1262→
  1263→        def fil_lastop(entity: TileEntity):
  1264→            assert len(entity.sizes) == 3
  1265→            # unused level limited to zero
  1266→            if entity.sizes[VectorInnerParallel] != 0:
  1267→                return False            
  1268→            return True
  1269→
  1270→        def fil_otherop(entity: TileEntity):
  1271→            assert len(entity.sizes) == 3
  1272→            # not provide
  1273→            if entity.sizes[Distribution] != 0 or entity.sizes[VectorCommonParallel] != 0:
  1274→                return False
  1275→            # unused level limited to zero
  1276→            if entity.sizes[VectorInnerParallel] != 0:
  1277→                return False
  1278→            return True
  1279→
  1280→        try:
  1281→            for axis in axes:
  1282→                if axis.type == PARALLEL:
  1283→                    if isLast:
  1284→                        # last op has all its tile levels used
  1285→                        if axis.no >= start_inner:
  1286→                            cfg.define_tile(axis, policy + ",zero_vector_common", p_level, fil=fil_lastop)
  1287→                        else:
  1288→                            cfg.define_tile(axis, policy, p_level, fil=fil_lastop)
  1289→                    else:
  1290→                        cfg.define_tile(axis, policy, p_level, fil=fil_otherop)                    
  1291→                elif axis.type == REDUCTION:
  1292→                    # no tiling perform on reduction axis
  1293→                    cfg.define_tile(axis, "candidate", r_level, candidates=[[0]*r_level])
  1294→                else:
  1295→                    raise 
  1296→        except:
  1297→            return None
  1298→
  1299→    return MultiConfigSpace([cfg])
